79uint32_t FLASH_IAP_Erase(APP_TypeDef Application)
80{
81 uint32_t numberOfPage = 0x0;
82 uint32_t eraseCounter = 0x0;
83 uint32_t flashDestination = 0x0;
84
85 if (Application == APP1)
86 {
87 numberOfPage = (USER_APP1_END_ADDRESS - USER_APP1_START_ADDRESS) / FLASH_PAGE_SIZE;
88 flashDestination = USER_APP1_START_ADDRESS;
89 }
90 else
91 {
92 numberOfPage = (USER_APP2_END_ADDRESS - USER_APP2_START_ADDRESS) / FLASH_PAGE_SIZE;
93 flashDestination = USER_APP2_START_ADDRESS;
94 }
95
96 for(eraseCounter = 0; (eraseCounter < numberOfPage) && (FLASHStatus == FMC_STATUS_COMPLETE); eraseCounter++)
97 {
98 FLASHStatus = FMC_ErasePage(flashDestination + (FLASH_PAGE_SIZE * eraseCounter));
99 }
100
101 return (SUCCESS);
102}

120uint32_t FLASH_IAP_Write(__IO uint32_t* address, uint32_t* data, uint16_t length, APP_TypeDef Application)
121{
122 uint32_t i = 0;
123 uint32_t addressEnd;
124
125 if (Application == APP1)
126 {
127 addressEnd = USER_APP1_END_ADDRESS - 4;
128 }
129 else
130 {
131 addressEnd = USER_APP2_END_ADDRESS - 4;
132 }
133
134 for (i = 0; i < length; i++)
135 {
136 if (*address > addressEnd)
137 {
138 return (ERROR);
139 }
140
141 /* The operation will be done by word and Check the written value */
142 if ((FMC_ProgramWord(*address, *(uint32_t*)(data + i)) != FMC_STATUS_COMPLETE) \
143 || (*(uint32_t*)*address != *(uint32_t*)(data + i)))
144 {
145 return (ERROR);
146 }
147
148 *address += 4;
149 }
150
151 return (SUCCESS);
152}
