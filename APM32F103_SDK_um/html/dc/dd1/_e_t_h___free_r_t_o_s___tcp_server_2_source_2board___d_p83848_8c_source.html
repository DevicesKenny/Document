158void Ethernet_GPIOConfig(void)
159{
160 GPIO_Config_T configStruct;
161 /*
162 * ETH_RESET# PB15
163 * ETH_INT PB14
164 * ETH_MDIO PA2
165 * ETH_MDC PC1
166 * ETH_MCO PA8
167 */
168
169 /* Enable GPIOs clocks */
170 RCM_EnableAPB2PeriphClock(ETH_MDC_GPIO_CLK | ETH_MDIO_GPIO_CLK |
171 ETH_RESET_GPIO_CLK | ETH_INT_GPIO_CLK |
172 ETH_MCO_GPIO_CLK);
173
174 /* Enable SYSCFG clock */
175 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_AFIO);
176
177 /* Configure ETH_RESET# */
178 GPIO_ConfigStructInit(&configStruct);
179 configStruct.pin = ETH_INT_PIN;
180 configStruct.speed = GPIO_SPEED_50MHz;
181 configStruct.mode = GPIO_MODE_OUT_PP;
182 GPIO_Config(ETH_RESET_PORT, &configStruct);
183
184 /* Configure ETH_MII_INT */
185 configStruct.pin = ETH_INT_PIN;
186 GPIO_Config(ETH_INT_PORT, &configStruct);
187 ETH_INT_PIN_HIGH();
188
189 /* Configure ETH_MDC */
190 configStruct.pin = ETH_MDC_PIN;
191 configStruct.mode = GPIO_MODE_AF_PP;
192 GPIO_Config(ETH_MDC_PORT, &configStruct);
193
194 /* Configure ETH_MDIO */
195 configStruct.pin = ETH_MDIO_PIN;
196 GPIO_Config(ETH_MDIO_PORT, &configStruct);
197
198 /* RESET ETH PHY */
199 ETH_RESET_PIN_LOW();
200 Ethernet_ResetDelay(DP83848_RESET_DELAY);
201 ETH_RESET_PIN_HIGH();
202 Ethernet_ResetDelay(DP83848_RESET_DELAY);
203
204 /* Configure PB10 ,PB11 ,PB12 ,PB13 as alternate function push-pull */
205 configStruct.pin = GPIO_PIN_10 | GPIO_PIN_11 | GPIO_PIN_12 | GPIO_PIN_13;
206 GPIO_Config(GPIOB, &configStruct);
207
208 /* if BOARD_DP83848.h define MII_MODE, this function configurate MII_MODE */
209#ifdef MII_MODE
211 /* Enable GPIOs clocks */
212 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA | RCM_APB2_PERIPH_GPIOB |
213 RCM_APB2_PERIPH_GPIOC | RCM_APB2_PERIPH_GPIOD);
214
215 /* Configure PC2 as alternate function push-pull */
216 configStruct.pin = GPIO_PIN_8;
217 GPIO_Config(GPIOB, &configStruct);
218
219 /* Configure PC2 as alternate function push-pull */
220 configStruct.pin = GPIO_PIN_2;
221 GPIO_Config(GPIOC, &configStruct);
222
223 GPIO_ConfigPinRemap(GPIO_REMAP_ETH_MAC);
224
225 /* Configure PA1 and PA3 PA7 as input */
226 GPIO_ConfigStructInit(&configStruct);
227 configStruct.pin = GPIO_PIN_1| GPIO_PIN_3 | GPIO_PIN_7;
228 configStruct.mode = GPIO_MODE_IN_FLOATING;
229 GPIO_Config(GPIOA, &configStruct);
230
231 /* Configure PB10 as input */
232 configStruct.pin = GPIO_PIN_10;
233 GPIO_Config(GPIOB, &configStruct);
234
235 /* Configure PC3 as input */
236 configStruct.pin = GPIO_PIN_3;
237 GPIO_Config(GPIOC, &configStruct);
238
239 /* Configure PD8, PD9, PD10 PD11 PD12 as input */
240 GPIO_ConfigStructInit(&configStruct);
241 configStruct.pin = GPIO_PIN_8 | GPIO_PIN_9 | GPIO_PIN_10| GPIO_PIN_11| GPIO_PIN_12;
242 configStruct.mode = GPIO_MODE_IN_FLOATING;
243 GPIO_Config(GPIOD, &configStruct);
244
245#ifdef PHY_CLOCK_MCO
246 /* Configure ETH_MCO */
247 configStruct.pin = ETH_MCO_PIN;
248 configStruct.mode = GPIO_MODE_AF_PP;
249 configStruct.speed = GPIO_SPEED_50MHz;
250 GPIO_Config(ETH_MCO_PORT, &configStruct);
251
252 /* Get HSE clock = 25MHz on PA8 pin (MCO) */
253 RCM_ConfigMCO(RCM_MCOCLK_HSE);
254
255#endif
257 GPIO_ConfigPinRemap(GPIO_REMAP_MACEISEL_MII);
258
259#elif defined RMII_MODE /* Mode RMII */
260
261 /*
262 * ETH_RMII_REF_CLK PA1
263 * ETH_RMII_CRS_DV PD8
264 * ETH_RMII_RXD0 PD9
265 * ETH_RMII_RXD1 PD10
266 * ETH_RMII_TX_EN PB11
267 * ETH_RMII_TXD0 PB12
268 * ETH_RMII_TXD1 PB13
269 */
270
271 /* Enable GPIOs clocks */
272 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA | RCM_APB2_PERIPH_GPIOB |
273 RCM_APB2_PERIPH_GPIOC | RCM_APB2_PERIPH_GPIOD);
274
275 /* ETHERNET pins remapp in APM32107-MINI board: RX_DV and RxD[3:0] */
276 GPIO_ConfigPinRemap(GPIO_REMAP_ETH_MAC);
277
278 /* Configure PA1 and PA3 as input */
279 GPIO_ConfigStructInit(&configStruct);
280 configStruct.pin = GPIO_PIN_1;
281 configStruct.mode = GPIO_MODE_IN_FLOATING;
282 GPIO_Config(GPIOA, &configStruct);
283
284 /* Configure PD8, PD9, PD10 as input */
285 GPIO_ConfigStructInit(&configStruct);
286 configStruct.pin = GPIO_PIN_8 | GPIO_PIN_9 | GPIO_PIN_10;
287 configStruct.mode = GPIO_MODE_IN_FLOATING;
288 GPIO_Config(GPIOD, &configStruct);
289
290#ifdef PHY_CLOCK_MCO
291 /* Configure ETH_MCO */
292 configStruct.pin = ETH_MCO_PIN;
293 configStruct.mode = GPIO_MODE_AF_PP;
294 configStruct.speed = GPIO_SPEED_50MHz;
295 GPIO_Config(ETH_MCO_PORT, &configStruct);
296
297 /* Set PLL3 clock output to 50MHz (25MHz /5 *10 =50MHz) */
298 RCM_ConfigPLL3(RCM_PLL3MF_10);
299
300 /* Enable PLL3 */
301 RCM_EnablePLL3();
302
303 /* Wait till PLL3 is ready */
304 while(RCM_ReadStatusFlag(RCM_FLAG_PLL3RDY) == RESET)
305 {
306
307 }
308
309 /* Get PLL3 clock on PA8 pin (MCO) */
310 RCM_ConfigMCO(RCM_MCOCLK_PLL3CLK);
311
312#endif
314 GPIO_ConfigPinRemap(GPIO_REMAP_MACEISEL_RMII);
315#endif
316}
