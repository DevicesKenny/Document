585void QSPI_ClearIntFlag(uint32_t flag)
586{
587 volatile uint32_t dummy = 0;
588
589 if (flag & QSPI_INT_FLAG_TFO)
590 {
591 dummy = QSPI->TFOIC;
592 }
593 else if (flag & QSPI_INT_FLAG_RFO)
594 {
595 dummy = QSPI->RFOIC;
596 }
597 else if (flag & QSPI_INT_FLAG_RFU)
598 {
599 dummy = QSPI->RFUIC;
600 }
601 else if (flag & QSPI_INT_FLAG_MST)
602 {
603 dummy = QSPI->MIC;
