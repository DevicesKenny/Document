92int main(void)
93{
94 USART_Config_T usartConfig;
95
96 /* User config the different system Clock */
97 UserRCMClockConfig();
98
99 /* Configure SysTick */
100 ConfigSysTick();
101
102 /* Configure USART
103 - USARTx configured as follow:
104 - BaudRate = 115200
105 - Word Length = USART_WordLength_8b
106 - Stop Bit = USART_StopBits_1
107 - Parity = USART_Parity_No
108 - Hardware flow control disabled (RTS and CTS signals)
109 - Receive and transmit enabled
110 */
111 usartConfig.baudRate = 115200;
112 usartConfig.wordLength = USART_WORD_LEN_8B;
113 usartConfig.stopBits = USART_STOP_BIT_1;
114 usartConfig.parity = USART_PARITY_NONE ;
115 usartConfig.mode = USART_MODE_TX_RX;
116 usartConfig.hardwareFlow = USART_HARDWARE_FLOW_NONE;
117
118 APM_BOARD_COMInit(COM1, &usartConfig);
119
120 /* Configure ethernet (GPIOs, clocks, MAC, DMA) */
121 Ethernet_Config();
122
123 /* Initilaize the LwIP stack */
124 LwIP_Init();
125
126 while (1)
127 {
128 /* check if any packet received */
129 if (ETH_ReadRxPacketSize(DMARxDescToGet))
130 {
131 /* process received ethernet packet */
132 LwIP_Pkt_Handle();
133 }
134
135 /* handle periodic timers for LwIP */
136 LwIP_Periodic_Handle(ETHTimer);
137 }
138}

163void LwIP_Init(void)
164{
165 struct ip_addr ipaddr;
166 struct ip_addr netmask;
167 struct ip_addr gw;
168
169 int ipAddrTemp[4];
170 int netMaskTemp[4];
171 int gatewayTemp[4];
172
173 /* Initializes the dynamic memory heap */
174 mem_init();
175
176 /* Initializes the memory pools */
177 memp_init();
178
179 sscanf(ETH_IP_ADDR, "%d.%d.%d.%d", &ipAddrTemp[0], &ipAddrTemp[1], &ipAddrTemp[2], &ipAddrTemp[3]);
180 sscanf(ETH_NET_MASK, "%d.%d.%d.%d", &netMaskTemp[0], &netMaskTemp[1], &netMaskTemp[2], &netMaskTemp[3]);
181 sscanf(ETH_GATEWAY, "%d.%d.%d.%d", &gatewayTemp[0], &gatewayTemp[1], &gatewayTemp[2], &gatewayTemp[3]);
182
183 IP4_ADDR(&ipaddr, ipAddrTemp[0], ipAddrTemp[1], ipAddrTemp[2], ipAddrTemp[3]);
184 IP4_ADDR(&netmask, netMaskTemp[0], netMaskTemp[1], netMaskTemp[2], netMaskTemp[3]);
185 IP4_ADDR(&gw, gatewayTemp[0], gatewayTemp[1], gatewayTemp[2], gatewayTemp[3]);
186
187 /* Config MAC Address */
188 ETH_ConfigMACAddress(ETH_MAC_ADDRESS0, SetMACaddr);
189
190 /* Add a network interface to the list of lwIP netifs */
191 netif_add(&UserNetif, &ipaddr, &netmask, &gw, NULL, &ethernetif_init, &ethernet_input);
192
193 /* Registers the default network interface */
194 netif_set_default(&UserNetif);
195
196 /* When the netif is fully configured this function must be called */
197 netif_set_up(&UserNetif);
198
199#ifdef MII_MODE
200 printf("MII_MODE\r\n");
201#elif defined RMII_MODE /* Mode RMII */
202 printf("RMII_MODE\r\n");
203#endif
204 printf("Static ETH Information:\r\n");
205 printf("ETH_IP : %d.%d.%d.%d\r\n", ipAddrTemp[0], ipAddrTemp[1], ipAddrTemp[2], ipAddrTemp[3]);
206 printf("ETH_Net_Mask : %d.%d.%d.%d\r\n", netMaskTemp[0], netMaskTemp[1], netMaskTemp[2], netMaskTemp[3]);
207 printf("ETH_Gateway : %d.%d.%d.%d\r\n", gatewayTemp[0], gatewayTemp[1], gatewayTemp[2], gatewayTemp[3]);
208}

230void LwIP_Periodic_Handle(__IO uint32_t ETHTimer)
231{
232#if LWIP_TCP
233
234 /* TCP periodic process every 250 ms */
235 if (ETHTimer - TCPTimer >= TCP_TMR_INTERVAL)
236 {
237 TCPTimer = ETHTimer;
238 tcp_tmr();
239 }
240
241#endif
242
243 /* ARP periodic process every 5s */
244 if ((ETHTimer - ARPTimer) >= ARP_TMR_INTERVAL)
245 {
246 ARPTimer = ETHTimer;
247 etharp_tmr();
248 }
249}
