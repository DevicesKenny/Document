168FRESULT FATFS_ScanFiles(char* path)
169{
170 FRESULT status;
171 /* file informations */
172 FILINFO fileInfo;
173 /* file directory */
174 DIR dir;
175 /* file name */
176 char *fileName;
177 uint32_t lengthScanPath;
178
179#if _USE_LFN
180 /* Simplified Chinese need two bytes to save a word */
181 static char lfn[_MAX_LFN * 2 + 1];
182 fileInfo.lfsize = sizeof(lfn);
183 fileInfo.lfname = lfn;
184#endif
185
186 /* open directory */
187 status = f_opendir(&dir,(const TCHAR*)path);
188 if(status == FR_OK)
189 {
190 lengthScanPath = strlen(path);
191 while(1)
192 {
193 /* Read the contents of the directory */
194 status = f_readdir(&dir, &fileInfo);
195
196 /* Error or read all items complete */
197 if((status != FR_OK) || (fileInfo.fname[0] == 0))
198 {
199 break;
200 }
201#if _USE_LFN
202 fileName = *fileInfo.lfname ? fileInfo.lfname : fileInfo.fname;
203#else
204 fileName = fileInfo.fname;
205#endif
206 /* Ignore parent directory */
207 if(*fileName == '.')
208 {
209 continue;
210 }
211 /* It is a directory */
212 if(fileInfo.fattrib & AM_DIR)
213 {
214 sprintf(&path[lengthScanPath], "/%s", fileName);
215
216 /* recursive */
217 status = FATFS_ScanFiles(path);
218 path[lengthScanPath] = 0;
219 if(status != FR_OK)
220 {
221 break;
222 }
223 }
224 /* It is a file */
225 else
226 {
227 printf("%s/%s\r\n",path,fileName);
228 }
229 }
230 }
231
232 return status;
233}
