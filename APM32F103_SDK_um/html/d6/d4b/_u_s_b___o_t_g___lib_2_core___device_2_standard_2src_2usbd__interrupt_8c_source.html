73void USBD_OTG_IsrHandler(void)
74{
75 /* Handle OUT Endpoint interrupt */
76 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_ONEP))
77 {
78 USBD_OutEndpointIsrHandler();
79 }
80
81 /* Handle IN Endpoint interrupt */
82 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_INEP))
83 {
84 USBD_InEndpointIsrHandler();
85 }
86
87 /* Handle RxFIFO no empty interrupt */
88 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_RXFNONE))
89 {
90 USBD_RxFifoNoEmptyIsrHandler();
91 }
92
93 /* Handle USB Reset interrupt */
94 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_USBRST))
95 {
96 USBD_ResetIsrHandler();
97 }
98
99 /* Handle Resume/remote wakeup detected interrupt */
100 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_RWAKE))
101 {
102 USBD_ResumeIsrHandler();
103 }
104
105 /* Handle USB suspend interrupt */
106 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_USBSUS))
107 {
108 USBD_SuspendIsrHandler();
109 }
110
111 /* Handle Enumeration done interrupt */
112 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_ENUMD))
113 {
114 USBD_EnumDoneIsrHandler();
115 }
116
117 /* Handle Mode mismatch interrupt */
118 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_MMIS))
119 {
120 USBD_ModeMismatchIsrHandler();
121 }
122
123 /* Handle interrupt */
124 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_SOF))
125 {
126 USBD_SofIsrHanlder();
127 }
128
129 /* Handle Session request/new session detected interrupt */
130 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_SREQ))
131 {
132 USBD_NewSessionIsrHandler();
133 }
134
135 /* Handle OTG interrupt */
136 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_OTG))
137 {
138 USBD_OTGIsrHandler();
139 }
140
141 if (g_usbDev.intHandler)
142 {
143 g_usbDev.intHandler();
144 }

154static void USBD_OutEndpointIsrHandler(void)
155{
156 uint16_t epIntNum;
157 uint8_t ep;
158
159 epIntNum = USB_OTG_ReadOutEpIntNumber();
160
161 for (ep = 0; ep < 16; ep++)
162 {
163 if (!(epIntNum >> ep))
164 {
165 break;
166 }
167 /* Fine out all the interrupt OUT endpoint */
168 else if ((epIntNum >> ep) & 0x01)
169 {
170 /* Transfer completed */
171 if (USB_OTG_ReadOutEpIntStatus(ep, USB_INT_EP_OUT_TSFCMP))
172 {
173 USB_OTG_ClearOutEpIntStatus(ep, USB_INT_EP_OUT_TSFCMP);
174
175 /* Handle control EP */
176 if (ep == USB_EP_0)
177 {
178 /* Control OUT */
179 USBD_CtrlOutProcess();
180 }
181
182 /* Handle other OUT EP */
183 else if (g_usbDev.outEpHandler != NULL)
184 {
185 g_usbDev.outEpHandler(ep);
186 }
187 }
188
189 /* Setup completed */
190 if (USB_OTG_ReadOutEpIntStatus(ep, USB_INT_EP_OUT_SETPCMP))
191 {
192 USB_OTG_ClearOutEpIntStatus(ep, USB_INT_EP_OUT_SETPCMP);
193
194 /* SETUP */
195 USBD_SetupProcess();
196 }
197
198 /* Endpoint disable */
199 if (USB_OTG_ReadOutEpIntStatus(ep, USB_INT_EP_OUT_EPDIS))
200 {
201 USB_OTG_ClearOutEpIntStatus(ep, USB_INT_EP_OUT_EPDIS);
202 }
203
204 /* OUT token received when endpoint disabled */
205 if (USB_OTG_ReadOutEpIntStatus(ep, USB_INT_EP_OUT_RXOTDIS))
206 {
207 USB_OTG_ClearOutEpIntStatus(ep, USB_INT_EP_OUT_RXOTDIS);
208 }
209
210 /* Received Back-to-back SETUP packets over 3 */
211 if (USB_OTG_ReadOutEpIntStatus(ep, USB_INT_EP_OUT_RXBSP))
212 {
213 USB_OTG_ClearOutEpIntStatus(ep, USB_INT_EP_OUT_RXBSP);
214 }
215 }
216 }

226static void USBD_InEndpointIsrHandler(void)
227{
228 uint16_t epIntNum;
229 uint8_t ep;
230
231 epIntNum = USB_OTG_ReadInEpIntNumber();
232
233 for (ep = 0; ep < 16; ep++)
234 {
235 if (!(epIntNum >> ep))
236 {
237 break;
238 }
239 /* Fine out all the interrupt IN endpoint */
240 else if ((epIntNum >> ep) & 0x01)
241 {
242 /* Transfer completed */
243 if (USB_OTG_ReadInEpIntStatus(ep, USB_INT_EP_IN_TSFCMP))
244 {
245 USB_OTG_ClearInEpIntStatus(ep, USB_INT_EP_IN_TSFCMP);
246
247 USB_OTG_DisableInEpEmptyInterrupt(ep);
248
249 if (ep == USB_EP_0)
250 {
251 /* Control IN */
252 USBD_CtrlInProcess();
253 }
254 /* Handle other IN EP when Transfer completed */
255 else if (g_usbDev.inEpHandler != NULL)
256 {
257 g_usbDev.inEpHandler(ep);
258 }
259 }
260
261 /* TxFIFO empty */
262 if (USB_OTG_ReadInEpIntStatus(ep, USB_INT_EP_IN_TXFE))
263 {
264 USBD_PushDataToTxFIFO(ep);
265 }
266 /* Endpoint disabled */
267 if (USB_OTG_ReadInEpIntStatus(ep, USB_INT_EP_IN_EPDIS))
268 {
269 USB_OTG_ClearInEpIntStatus(ep, USB_INT_EP_IN_EPDIS);
270 }
271 /* Timeout */
272 if (USB_OTG_ReadInEpIntStatus(ep, USB_INT_EP_IN_TO))
273 {
274 USB_OTG_ClearInEpIntStatus(ep, USB_INT_EP_IN_TO);
275 }
276 /* IN endpoint NAK effective */
277 if (USB_OTG_ReadInEpIntStatus(ep, USB_INT_EP_IN_IEPNAKE))
278 {
279 USB_OTG_ClearInEpIntStatus(ep, USB_INT_EP_IN_IEPNAKE);
280 }
281 /* IN token received when TxFIFO is empty */
282 if (USB_OTG_ReadInEpIntStatus(ep, USB_INT_EP_IN_ITXEMP))
283 {
284 USB_OTG_ClearInEpIntStatus(ep, USB_INT_EP_IN_ITXEMP);
285 }
286 }
287 }

297static void USBD_RxFifoNoEmptyIsrHandler(void)
298{
299 USBD_FIFO_Status_T fifoStatus;
300 uint8_t ep;
301
302 /* Disable RxFIFO no empty Interrup */
303 USB_OTG_DisableGlobalInterrupt(USB_INT_G_RXFNONE);
304
305 /* Read and pop the RxFIFO status data */
306 fifoStatus.word = USB_OTG_PopRxFifoStatus();
307 ep = fifoStatus.bit.ep;
308
309 /* Handle RxFIFO data */
310 switch (fifoStatus.bit.status)
311 {
312 /* Received OUT data packet */
313 case FIFO_STATUS_OUT_DATA :
314 if ((fifoStatus.bit.byteCount != 0) && (g_usbDev.outBuf[ep].pBuf != NULL))
315 {
316 USB_FIFO_ReadRxFifoPacket(g_usbDev.outBuf[ep].pBuf, \
317 fifoStatus.bit.byteCount);
318
319 g_usbDev.outBuf[ep].xferCnt = fifoStatus.bit.byteCount;
320 g_usbDev.outBuf[ep].pBuf += fifoStatus.bit.byteCount;
321 g_usbDev.outBuf[ep].bufLen -= fifoStatus.bit.byteCount;
322 g_usbDev.outBuf[ep].packNum--;
323 }
324 break;
325
326 /* Received SETUP data packet */
327 case FIFO_STATUS_SETUP_DATA :
328 USB_FIFO_ReadRxFifoPacket(g_usbDev.reqData.pack, 8);
329 break;
330
331 case FIFO_STATUS_OUT_COMP :
332 break;
333 case FIFO_STATUS_SETUP_COMP :
334 break;
335 case FIFO_STATUS_G_OUT_NAK :
336 break;
337 default:
338 break;
339 }
340 /* Enable RxFIFO no empty Interrupt */
341 USB_OTG_EnableGlobalInterrupt(USB_INT_G_RXFNONE);
342}

351static void USBD_ResetIsrHandler(void)
352{
353 uint8_t ep;
354
355 /* Clear the Remote wakeup signaling */
356 USB_OTG_DisableRemoteWakeupSignal();
357
358 USB_OTG_FlushTxFIFO(0);
359
360 /* Clear endpoint interrupt flag */
361 for (ep = 0; ep < USB_MAX(USB_OUT_EP_MAX_NUM, USB_IN_EP_MAX_NUM); ep++)
362 {
363 /* Clear endpoint Flag */
364 USB_OTG_ClearOutEpIntStatus(ep, USB_INT_EP_OUT_ALL);
365 USB_OTG_ClearInEpIntStatus(ep, USB_INT_EP_IN_ALL);
366 }
367
368 /* Enable enpoint 0 global interrupt */
369 USB_OTG_EnableOutEpInterruptMask(USB_EP_0);
370 USB_OTG_EnableInEpInterruptMask(USB_EP_0);
371
372 /* Enable OUT EP interrupt */
373 USB_OTG_EnableOutEpInterrupt(USB_INT_EP_OUT_SOURCE);
374
375 /* Enable IN EP interrupt */
376 USB_OTG_EnableInEpInterrupt(USB_INT_EP_IN_SOURCE);
377
378 /* Reset device address */
379 USB_OTG_ConfigDeviceAddress(0);
380
381 USB_OTG_ReceiveSetupPacket(3);
382
383 /* EP Active */
384 USBD_OpenOutEP(USB_EP_0, USB_EP_TYPE_CONTROL, USB_EP0_PACKET_SIZE);
385 USBD_OpenInEP(USB_EP_0, USB_EP_TYPE_CONTROL, USB_EP0_PACKET_SIZE);
386
387 g_usbDev.curFeature = g_usbDev.pConfigurationDesc->pDesc[7];
388 g_usbDev.devState = USBD_DEVICE_STATE_DEFAULT;
389
390 USB_OTG_ClearGlobalIntFlag(USB_INT_G_USBRST);
391
392 /* Reset handler */
393 if (g_usbDev.resetHandler != NULL)
394 {
395 g_usbDev.resetHandler();
396 }
397
398 /* User reset callback function */
399 g_userCallback.USER_DeviceReset();
400}

448static void USBD_EnumDoneIsrHandler(void)
449{
450 uint32_t hClkFreq;
451
452 hClkFreq = RCM_ReadHCLKFreq();
453
454 if (hClkFreq < 48000000)
455 {
456 /* TurnaroundTime = 4 * (48M / hclk) + 1*/
457 USB_OTG_ConfigTurnaroundTime(((((uint32_t)48000000 << 2) + (hClkFreq - 1)) / hClkFreq) + 1);
458 }
459 else
460 {
461 USB_OTG_ConfigTurnaroundTime(0x5);
462 }
