30static rt_uint32_t _SysTick_Config(rt_uint32_t ticks)
31{
32 if ((ticks - 1) > 0xFFFFFF)
33 {
34 return 1;
35 }
36
37 _SYSTICK_LOAD = ticks - 1;
38 _SYSTICK_PRI = 0xFF;
39 _SYSTICK_VAL = 0;
40 _SYSTICK_CTRL = 0x07;
41
42 return 0;
43}

102static int uart_init(void)
103{
104 static USART_Config_T UartHandle;
105 GPIO_Config_T GPIO_ConfigStruct;
106 GPIO_ConfigStructInit(&GPIO_ConfigStruct);
107
108 /* Enable GPIO clock */
109 RCM_EnableAPB2PeriphClock((RCM_APB2_PERIPH_T)(RCM_APB2_PERIPH_GPIOA | RCM_APB2_PERIPH_USART1));
110
111 /* Configure USART Tx and Rx as alternate function push-pull */
112 GPIO_ConfigStruct.mode = GPIO_MODE_AF_PP;
113 GPIO_ConfigStruct.pin = GPIO_PIN_9;
114 GPIO_ConfigStruct.speed = GPIO_SPEED_50MHz;
115 GPIO_Config(GPIOA, &GPIO_ConfigStruct);
116
117 GPIO_Config(GPIOA, &GPIO_ConfigStruct);
118
119 GPIO_ConfigStruct.pin = GPIO_PIN_10;
120 GPIO_ConfigStruct.mode = GPIO_MODE_IN_FLOATING;
121 GPIO_Config(GPIOA, &GPIO_ConfigStruct);
122
123 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_USART1);
124
125 /* USARTx configuration */
126 /* USARTx configured as follow:
127 - BaudRate = 115200 baud
128 - Word Length = 8 Bits
129 - One Stop Bit
130 - No parity
131 - Hardware flow control disabled (RTS and CTS signals)
132 - Receive and transmit enabled
133 */
134
135 UartHandle.baudRate = 115200;
136 UartHandle.mode = USART_MODE_TX_RX;
137 UartHandle.parity = USART_PARITY_NONE;
138 UartHandle.stopBits = USART_STOP_BIT_1;
139 UartHandle.wordLength = USART_WORD_LEN_8B;
140 UartHandle.hardwareFlow = USART_HARDWARE_FLOW_NONE;
141
142 USART_Config(USART1, &UartHandle);
143 USART_Enable(USART1);
144 return 0;
145}

148void rt_hw_console_output( const char* str )
149{
150
151 rt_enter_critical();
152
153 while (*str != '\0')
154 {
155 if (*str == '\n')
156 {
157 while (USART_ReadStatusFlag( USART1, USART_FLAG_TXBE)==RESET);
158
159 USART_TxData(USART1, '\r');
160 }
161
162 while (USART_ReadStatusFlag( USART1, USART_FLAG_TXBE)==RESET);
163
164 USART_TxData(USART1, *str++ );
165 }
166
167 rt_exit_critical();
168}

170char rt_hw_console_getchar(void)
171{
172 int ch = -1;
173
174 if (USART_ReadStatusFlag(USART1, USART_FLAG_RXBNE) != RESET)
175 {
176 ch = (int)USART_RxData(USART1);
177 USART_ClearStatusFlag(USART1, USART_FLAG_RXBNE);
178 }
179 else
180 {
181 if (USART_ReadStatusFlag(USART1, USART_FLAG_OVRE) != RESET)
182 {
183 USART_ClearStatusFlag(USART1, USART_FLAG_OVRE);
184 }
185
186 rt_thread_mdelay(10);
187 }
188
189 return ch;
190}
