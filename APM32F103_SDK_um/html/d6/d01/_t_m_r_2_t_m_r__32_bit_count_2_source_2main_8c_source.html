71int main(void)
72{
73 uint32_t temp;
74 USART_Config_T usartConfig;
75 TMR_BaseConfig_T baseConfig;
76
77 APM_MINI_LEDInit(LED2);
78 APM_MINI_LEDOff(LED2);
79
80 /* USART1 Tx PA9 Rx PA10 */
81 usartConfig.baudRate = 115200;
82 usartConfig.mode = USART_MODE_TX_RX;
83 usartConfig.hardwareFlow = USART_HARDWARE_FLOW_NONE;
84 usartConfig.parity = USART_PARITY_NONE;
85 usartConfig.stopBits = USART_STOP_BIT_1;
86 usartConfig.wordLength = USART_WORD_LEN_8B;
87 APM_MINI_COMInit(COM1,&usartConfig);
88
89 /* TMR1,3,4 Clock */
90 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_TMR3 | RCM_APB1_PERIPH_TMR4);
91 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_TMR1);
92
93 /* TMR3 Configure */
94 baseConfig.division = 0;
95 baseConfig.period = 0xFFFF;
96 baseConfig.clockDivision = TMR_CLOCK_DIV_1;
97 baseConfig.countMode = TMR_COUNTER_MODE_UP;
98 TMR_ConfigTimeBase(TMR3, &baseConfig);
99 TMR_ConfigPrescaler(TMR3,0,TMR_PSC_RELOAD_UPDATE);
100 TMR_EnableUpdate(TMR3);
101
102 /* Configure TMR3 Input */
103 TMR_SelectInputTrigger(TMR3,TMR_TRIGGER_SOURCE_ITR3);
104 TMR_SelectSlaveMode(TMR3,TMR_SLAVE_MODE_EXTERNAL1);
105 TMR_EnableMasterSlaveMode(TMR3);
106 TMR_EnableAutoReload(TMR3);
107
108 /* TMR4 Configure */
109 baseConfig.division = 71;
110 baseConfig.period = 0xFFFF;
111 baseConfig.clockDivision = TMR_CLOCK_DIV_1;
112 baseConfig.countMode = TMR_COUNTER_MODE_UP;
113 TMR_ConfigTimeBase(TMR4, &baseConfig);
114 TMR_EnableAutoReload(TMR4);
115
116 /* Configure TMR4 Output */
117 TMR_SelectSlaveMode(TMR4,TMR_SLAVE_MODE_RESET);
118 TMR_ConfigUpdateRequest(TMR4,TMR_UPDATE_SOURCE_REGULAR);
119 TMR_SelectOutputTrigger(TMR4,TMR_TRGO_SOURCE_UPDATE);
120 TMR_EnableMasterSlaveMode(TMR4);
121
122 /* TMR1 Configure */
123 baseConfig.clockDivision = TMR_CLOCK_DIV_1;
124 baseConfig.countMode = TMR_COUNTER_MODE_UP;
125 baseConfig.division = 71;
126 baseConfig.period = 999;
127 TMR_ConfigTimeBase(TMR1, &baseConfig);
128
129 /* Enable TMR1 Interrupt */
130 TMR_EnableInterrupt(TMR1, TMR_INT_UPDATE);
131 NVIC_EnableIRQRequest(TMR1_UP_IRQn, 0, 0);
132
133 /* Enable TMR */
134 TMR_Enable(TMR1);
135 TMR_Enable(TMR3);
136 TMR_Enable(TMR4);
137
138 while(1)
139 {
140 temp = TMR3->CNT << 16;
141 temp |= TMR4->CNT;
142 printf(" Count Value: 0x%08x\r\n", temp);
143 APM_MINI_LEDToggle(LED2);
144 Delay();
145 }
146}

178int fputc(int ch, FILE *f)
179{
180 /* send a byte of data to the serial port */
181 USART_TxData(DEBUG_USART,(uint8_t)ch);
182
183 /* wait for the data to be send */
184 while (USART_ReadStatusFlag(DEBUG_USART, USART_FLAG_TXBE) == RESET);
185
186 return (ch);
187}
