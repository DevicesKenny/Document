49int main(void)
50{
51 ADC_Config_T ADC_configStruct;
52
53 /* System clocks configuration */
54 RCM_Configuration();
55
56 /* NVIC configuration */
57 NVIC_Configuration();
58
59 /* GPIO configuration */
60 GPIO_Configuration();
61
62 /* Configure LED GPIO Pin */
63 APM_MINI_LEDInit(LED2);
64 APM_MINI_LEDOff(LED2);
65
66 /* ADC1 Configuration */
67 ADC_configStruct.mode = ADC_MODE_INDEPENDENT;
68 ADC_configStruct.scanConvMode = DISABLE;
69 ADC_configStruct.continuosConvMode = ENABLE;
70 ADC_configStruct.externalTrigConv = ADC_EXT_TRIG_CONV_None;
71 ADC_configStruct.dataAlign = ADC_DATA_ALIGN_RIGHT;
72 ADC_configStruct.nbrOfChannel = 1;
73 ADC_Config(ADC1, &ADC_configStruct);
74
75 /* ADC1 regular channel14 configuration */
76 ADC_ConfigRegularChannel(ADC1, ADC_CHANNEL_14, 1, ADC_SAMPLETIME_13CYCLES5);
77
78 /* Configure high and low analog watchdog thresholds */
79 ADC_ConfigAnalogWatchdogThresholds(ADC1, 0x0B00, 0x0300);
80 /* Configure channel14 as the single analog watchdog guarded channel */
81 ADC_ConfigAnalogWatchdogSingleChannel(ADC1, ADC_CHANNEL_14);
82 /* Enable analog watchdog on one regular channel */
83 ADC_EnableAnalogWatchdog(ADC1, ADC_ANALOG_WATCHDOG_SINGLE_REG);
84
85 /* Enable AWD interrupt */
86 ADC_EnableInterrupt(ADC1, ADC_INT_AWD);
87 /* Enable ADC1 */
88 ADC_Enable(ADC1);
89
90 /* Enable ADC1 reset calibration register */
91 ADC_ResetCalibration(ADC1);
92 /* Check the end of ADC1 reset calibration register */
93 while(ADC_ReadResetCalibrationStatus(ADC1));
94
95 /* Start ADC1 calibration */
96 ADC_StartCalibration(ADC1);
97 /* Check the end of ADC1 calibration */
98 while(ADC_ReadCalibrationStartFlag(ADC1));
99
100 /* Start ADC1 Software Conversion */
101 ADC_EnableSoftwareStartConv(ADC1);
102
103 while (1)
104 {
105 }
106}
