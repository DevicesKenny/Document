107static void USER_InitHandler(void)
108{
109 USART_Config_T usartConfig;
110
111 /* Configure USART */
112 usartConfig.baudRate = 115200;
113 usartConfig.wordLength = USART_WORD_LEN_8B;
114 usartConfig.stopBits = USART_STOP_BIT_1;
115 usartConfig.parity = USART_PARITY_NONE ;
116 usartConfig.mode = USART_MODE_TX_RX;
117 usartConfig.hardwareFlow = USART_HARDWARE_FLOW_NONE;
118
119 APM_BOARD_COMInit(COM2,&usartConfig);
120
121 APM_MINI_PBInit(BUTTON_KEY1, BUTTON_MODE_GPIO);
122 APM_MINI_PBInit(BUTTON_KEY2, BUTTON_MODE_GPIO);
123}

174static void USER_CfgDescHandler(void *cfgDesc, void *interface)
175{
176 USBH_Interface_T *itf = interface;
177 uint8_t i;
178
179 switch (itf->itfDesc.bInterfaceClass)
180 {
181 case 0x08:
182 printf(">> This is a Mass Storage Class Device.\r\n");
183 break;
184 case 0x03:
185 printf(">> This is a HID Class Device.\r\n");
186 break;
187 case 0x0A:
188 printf(">> This is a Communications Class Device.\r\n");
189 break;
190 default :
191 printf(">> Class Code is : 0x%02X\r\n", itf->itfDesc.bInterfaceClass);
192 break;
193 }
194 printf(">> Use %d endpoint:\r\n", itf->itfDesc.bNumEndpoints);
195 for (i = 0; i < itf->itfDesc.bNumEndpoints; i++)
196 {
197 printf(" Endpoint 0x%02X: max pack size is %d bytes\r\n", \
198 itf->epDesc[i].bEndpointAddress,
199 itf->epDesc[i].wMaxPacketSize[0] | itf->epDesc[i].wMaxPacketSize[1] << 8);
200 }
201}

227static USER_STATE_T USER_ApplicationHandler(void)
228{
229 FRESULT res;
230
231 if (!APM_MINI_PBGetState(BUTTON_KEY1))
232 {
233 /* Read Disk Test */
234 if (g_usbHost.connectedFlag)
235 {
236 printf("------ Scan files on the USB disk ------ \r\n");
237
238 if (f_mount(&s_fatFs, "0:", 1) == FR_OK)
239 {
240 FATFS_ScanFiles(fileScanPath);
241 __NOP();
242 }
243 else
244 {
245 printf("fatFs mount fail!\r\n");
246 }
247 }
248 else
249 {
250 printf("Device is not connected\r\n");
251 }
252 while (!APM_MINI_PBGetState(BUTTON_KEY1));
253 }
254
255 if (!APM_MINI_PBGetState(BUTTON_KEY2))
256 {
257 if (g_usbHost.connectedFlag)
258 {
259 printf("------ Write and read USB disk file ------\r\n");
260
261 if ((res = f_mount(&s_fatFs, "0:", 1)) == FR_OK)
262 {
263 FATFS_WriteReadFile(&s_file);
264 }
265 else
266 {
267 printf("Error : %d\r\n" ,res);
268 }
269 }
270 else
271 {
272 printf("Device is not connected\r\n");
273 }
274 while (!APM_MINI_PBGetState(BUTTON_KEY2));
275 }
276
277 return USER_OK;
278}
