67int main(void)
68{
69 USART_Init();
70
71 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_CAN1);
72 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_CAN2);
73 NVIC_Configuration();
74
75 /* Configures LED 2,3 */
76 APM_MINI_LEDInit(LED2);
77 APM_MINI_LEDInit(LED3);
78
79 /* Turns selected LED Off */
80 APM_MINI_LEDOff(LED2);
81 APM_MINI_LEDOff(LED3);
82
83 APM_MINI_PBInit(BUTTON_KEY1, BUTTON_MODE_GPIO);
84 APM_MINI_PBInit(BUTTON_KEY2, BUTTON_MODE_GPIO);
85
86 CAN_GPIO_Init();
87
88 CAN_Config_Init();
89
90 CAN_Transmit_Init();
91
92 CAN_EnableInterrupt(CAN1, CAN_INT_F0MP);
93
94 CAN_EnableInterrupt(CAN2, CAN_INT_F0MP);
95
96 while (1)
97 {
98
99 if (APM_MINI_PBGetState(BUTTON_KEY1) == RESET)
100 {
101 /* Turn on led LD2 */
102 APM_MINI_LEDOn(LED2);
103 TxMessage.data[0] = 0x55;
104 printf("CAN1 Transmit 0x55 !\r\n");
105 CAN_TxMessage(CAN1, &TxMessage);
106 while (APM_MINI_PBGetState(BUTTON_KEY1) == RESET);
107 }
108
109 if (APM_MINI_PBGetState(BUTTON_KEY2) == RESET)
110 {
111 /* Turn on led LD2 */
112 APM_MINI_LEDOn(LED2);

114 printf("CAN2 Transmit 0xAA !\r\n");
115 CAN_TxMessage(CAN2, &TxMessage);
