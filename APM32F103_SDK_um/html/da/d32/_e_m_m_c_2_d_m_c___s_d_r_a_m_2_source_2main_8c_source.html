168void SystemCoreClockConfig(uint32_t pllMf, uint32_t latency)
169{
170 __IO uint32_t i;
171
172 if (!(RCM->CTRL_B.HSERDYFLG))
173 {
174 RCM->CTRL_B.HSEEN= BIT_SET;
175
176 for (i = 0; i < HSE_STARTUP_TIMEOUT; i++)
177 {
178 if (RCM->CTRL_B.HSERDYFLG)
179 {
180 break;
181 }
182 }
183 }
184
185 if (RCM->CTRL_B.HSERDYFLG)
186 {
187 /* HSelect HSE as system clock source */
188 RCM->CFG_B.SCLKSEL = 0x01;
189
190 /* Enable Prefetch Buffer */
191 FMC->CTRL1_B.PBEN = BIT_SET;
192 /* Flash latency wait state */
193 FMC->CTRL1_B.WS = latency;
194
195 /* HCLK = SYSCLK */
196 RCM->CFG_B.AHBPSC= 0X00;
197 /* PCLK2 = HCLK */
198 RCM->CFG_B.APB2PSC= 0;
199 /* PCLK1 = HCLK / 2 */
200 RCM->CFG_B.APB1PSC = 4;
201
202 /* Disable PLL */
203 RCM->CTRL_B.PLL1EN = 0;
204
205 /* PLL: HSE * PLLm */
206 RCM->CFG_B.PLL1SRCSEL = 1;
207 RCM->CFG_B.PLL1MULCFG = (pllMf - 2);
208
209 /* Enable PLL */
210 RCM->CTRL_B.PLL1EN = 1;
211
212 /* Wait PLL Ready */
213 while (RCM->CTRL_B.PLL1RDYFLG == BIT_RESET);
214
215 /* Select PLL as system clock source */
216 RCM->CFG_B.SCLKSEL = 2;
217
218 /* Wait till PLL is used as system clock source */
219 while (RCM->CFG_B.SCLKSELSTS!= 0x02);
220 }
221}
