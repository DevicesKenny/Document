113static void ENUM_GetDevDescHandler(void)
114{
115 USBH_CTRL_STATE_T ctrlState;
116
117 /* Prepare to Get device descriptor */
118 if (g_usbHost.control.xferState == USBH_XFER_START)
119 {
120 /* Get device descriptor */
121 USBH_StandardGetDescriptor(USBH_DESC_DEVICE, 0, 8);
122 g_usbHost.control.xferState = USBH_XFER_WAITING;
123 }
124 /* Do the transaction */
125 else if (g_usbHost.control.xferState == USBH_XFER_WAITING)
126 {
127 /* Polling transaction handler */
128 ctrlState = USBH_ControlXferHandler(&g_usbHost.control.reqData,
129 (uint8_t*)&g_usbHost.control.descriptor.device);
130
131 switch (ctrlState)
132 {
133 case USBH_CTRL_COMPLETE:
134 g_usbHost.control.xferState = USBH_XFER_START;
135 g_usbHost.control.maxPackSize = g_usbHost.control.descriptor.device.bMaxPacketSize;
136
137 USBH_CH_OpenChannel(g_usbHost.control.outChannelNum,
138 g_usbHost.devAddress,
139 USB_EP_TYPE_CONTROL,
140 g_usbHost.control.descriptor.device.bMaxPacketSize);
141
142 USBH_CH_OpenChannel(g_usbHost.control.inChannelNum,
143 g_usbHost.devAddress,
144 USB_EP_TYPE_CONTROL,
145 g_usbHost.control.descriptor.device.bMaxPacketSize);
146
147 /* Prepare to Set device Address */
148 USBH_ConfigEnumState(USBH_ENUM_SET_ADDR);
149 break;
150 case USBH_CTRL_STALL :
151 case USBH_CTRL_ERROR :
152 g_usbHost.control.xferState = USBH_XFER_START;
153 break;
154 default :
155 break;
156 }
157 }

210static void ENUM_SetAddressHandler(void)
211{
212 USBH_CTRL_STATE_T ctrlState;
213
214 /* Prepare to set address */
215 if (g_usbHost.control.xferState == USBH_XFER_START)
216 {
217 /* Set device Address */
218 USBH_StandardSetAddress(USBH_DEVICE_CONFIGURED_ADDRESS);
219 g_usbHost.control.xferState = USBH_XFER_WAITING;
220 }
221 /* Do the transaction */
222 else if (g_usbHost.control.xferState == USBH_XFER_WAITING)
223 {
224 /* Polling transaction handler */
225 ctrlState = USBH_ControlXferHandler(&g_usbHost.control.reqData, NULL);
226
227 switch (ctrlState)
228 {
229 /* Control transfer Complete */
230 case USBH_CTRL_COMPLETE:
231 g_usbHost.control.xferState = USBH_XFER_START;
232 g_usbHost.devAddress = USBH_DEVICE_CONFIGURED_ADDRESS;
233
234 /* Open and Config out channel */
235 USB_Delay_us(2000);
236 USB_OTG_H_ConfigDeviceAddr(g_usbHost.control.outChannelNum, g_usbHost.devAddress);
237 USB_OTG_H_ConfigDeviceAddr(g_usbHost.control.inChannelNum, g_usbHost.devAddress);
238
239 /* Prepare to Get Full device Descriptor */
240 USBH_ConfigEnumState(USBH_ENUM_GET_FULL_DEV_DESC);
241 break;
242
243 /* Control transfer Stall */
244 case USBH_CTRL_STALL :
245 /* Control transfer Error */
246 case USBH_CTRL_ERROR :
247 g_usbHost.control.xferState = USBH_XFER_START;
248 break;
249 /* Waiting */
250 default :
251 break;
252 }
253 }
