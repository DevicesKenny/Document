57int main(void)
58{
59 /* Buffer Compare state */
60 uint8_t state = FAILED;
61 /* SPI receive Buf */
62 uint8_t rxDataBufSPI[DATA_BUF_SIZE] = {0x00};
63 /* USART receive Buf */
64 uint8_t rxDataBufUSART[DATA_BUF_SIZE] = {0x00};
65
66 APM_MINI_LEDInit(LED2);
67 APM_MINI_LEDInit(LED3);
68 APM_MINI_LEDOff(LED2);
69 APM_MINI_LEDOff(LED3);
70
71 USART_FullDuplex_Init();
72 SPI_FullDuplex_Init();
73
74 Delay(0x5000);
75
76 for (int i = 0; i < DATA_BUF_SIZE; i++)
77 {
78 /* Tx send data */
79 USART_TxData(USART1, i);
80
81 /* SPI send data */
82 SPI_WriteByte(i);
83
84 /* Wait data is received by SPI */
85 while (SPI_I2S_ReadStatusFlag(SPI1, SPI_FLAG_RXBNE) == RESET);
86
87 rxDataBufSPI[i] = SPI_I2S_RxData(SPI1);
88
89 /* Wait data is received by UART1 */
90 while (USART_ReadStatusFlag(USART1, USART_FLAG_RXBNE) == RESET);
91
92 rxDataBufUSART[i] = USART_RxData(USART1);
93 }
94
95 /* Compare receive Buffer */
96 state = BufferCompare(rxDataBufUSART, rxDataBufSPI, DATA_BUF_SIZE);
97
98 /* Data is ok then turn on LED2 */
99 if (state == 1)
100 {
101 APM_MINI_LEDOn(LED2);
102 }
103
104 while (1)
105 {
106 APM_MINI_LEDToggle(LED3);
107 Delay(0x7ffff);
108 }
109}

134void SPI_FullDuplex_Init()
135{
136 GPIO_Config_T gpioConfig;
137 SPI_Config_T spiConfig;
138
139 /* Enable related Clock */
140 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_SPI1);
141 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA);
142
143 /* config PIN_6 MISO */
144 gpioConfig.pin = GPIO_PIN_6 ;
145 gpioConfig.mode = GPIO_MODE_AF_PP;
146 gpioConfig.speed = GPIO_SPEED_50MHz;
147 GPIO_Config(GPIOA, &gpioConfig);
148
149 /* config PIN_5->SCK , PIN_7->MOSI */
150 gpioConfig.pin = GPIO_PIN_5 | GPIO_PIN_7;
151 gpioConfig.mode = GPIO_MODE_IN_PU;
152 gpioConfig.speed = GPIO_SPEED_50MHz;
153 GPIO_Config(GPIOA, &gpioConfig);
154
155 SPI_ConfigStructInit(&spiConfig);
156
157 spiConfig.length = SPI_DATA_LENGTH_8B;
158
159 spiConfig.baudrateDiv = SPI_BAUDRATE_DIV_32;
160 /* 2 line full duplex */
161 spiConfig.direction = SPI_DIRECTION_2LINES_FULLDUPLEX;
162 /* LSB first */
163 spiConfig.firstBit = SPI_FIRSTBIT_LSB;
164 /* Slave mode */
165 spiConfig.mode = SPI_MODE_SLAVE;
166 /* Polarity is low */
167 spiConfig.polarity = SPI_CLKPOL_LOW;
168 /* Software select slave enable */
169 spiConfig.nss = SPI_NSS_SOFT;
170 /* Phase is 1 edge */
171 spiConfig.phase = SPI_CLKPHA_1EDGE;
172 /* SPI config */
173 SPI_Config(SPI1, &spiConfig);
174
175 SPI_ConfigDataSize(SPI1, SPI_DATA_LENGTH_8B);
176
177 SPI_Enable(SPI1);
178}

188void USART_FullDuplex_Init()
189{
190 GPIO_Config_T gpioConfig;
191 USART_Config_T usartConfig;
192 USART_ClockConfig_T usartSyncClockConfig;
193
194 /* Enable GPIO clock */
195 RCM_EnableAPB2PeriphClock(MINI_COM1_TX_GPIO_CLK);
196
197 /* Enable COM1 or COM2 clock */
198 RCM_EnableAPB2PeriphClock(MINI_COM1_CLK);
199
200 /* Configure USART Tx as alternate function push-pull */
201 gpioConfig.mode = GPIO_MODE_AF_PP;
202 gpioConfig.pin = MINI_COM1_TX_PIN;
203 gpioConfig.speed = GPIO_SPEED_50MHz;
204 GPIO_Config(MINI_COM1_TX_GPIO_PORT, &gpioConfig);
205
206 /* Configure USART Rx as input floating */
207 gpioConfig.mode = GPIO_MODE_IN_PU;
208 gpioConfig.pin = MINI_COM1_RX_PIN;
209 GPIO_Config(MINI_COM1_RX_GPIO_PORT, &gpioConfig);
210
211 /* Configure USART CLK */
212 gpioConfig.mode = GPIO_MODE_AF_PP;
213 gpioConfig.pin = GPIO_PIN_8;
214 GPIO_Config(GPIOA, &gpioConfig);
215
216 /* Enable USART Clock */
217 usartSyncClockConfig.clock = USART_CLKEN_ENABLE;
218 /* Enable last bit clock */
219 usartSyncClockConfig.lastBit = USART_LBCP_ENABLE;
220 /* Clock phase is 1 edge */
221 usartSyncClockConfig.phase = USART_CLKPHA_1EDGE;
222 /* Clock polarity is Low */
223 usartSyncClockConfig.polarity = USART_CLKPOL_LOW;
224 /* Synchronous communication clock configuration */
225 USART_ConfigClock(USART1, &usartSyncClockConfig);
226
227 /* BaudRate is 115200 */
228 usartConfig.baudRate = 115200;
229 /* Enable RX and TX */
230 usartConfig.mode = USART_MODE_TX_RX;
231 /* Parity disable */
232 usartConfig.parity = USART_PARITY_NONE;
233 /* One stop bit */
234 usartConfig.stopBits = USART_STOP_BIT_1;
235 /* Word length is 8bit */
236 usartConfig.wordLength = USART_WORD_LEN_8B;
237
238 usartConfig.hardwareFlow = USART_HARDWARE_FLOW_NONE;
239
240 /* USART1 configuration */
241 USART_Config(USART1, &usartConfig);
242
243 /* Enable USART1 */
244 USART_Enable(USART1);
245}

274uint8_t BufferCompare(uint8_t* buf1, uint8_t* buf2, uint8_t size)
275{
276 uint8_t i;
277
278 for (i = 0; i < size; i++)
279 {
280 if (buf1[i] != buf2[i])
281 {
282 return FAILED;
283 }
284 }
285
286 return PASSED;
287}
