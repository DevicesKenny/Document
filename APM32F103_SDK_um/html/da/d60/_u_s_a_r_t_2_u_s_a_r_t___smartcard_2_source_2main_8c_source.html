177uint32_t SmartcardTest(void)
178{
179 uint32_t Counter = 0;
180
181 /* Set Smartcard Reset Pin to Low */
182 GPIO_ResetBit(GPIOB, GPIO_PIN_11);
183
184 /* Set Smartcard Reset Pin to High */
185 GPIO_SetBit(GPIOB, GPIO_PIN_11);
186
187 /* Read Smartcard ATR response */
188 for (uint16_t i = 0; i < 40; i++, Counter = 0)
189 {
190 while ((USART_ReadStatusFlag(USART3, USART_FLAG_RXBNE) == RESET) \
191 && (Counter != Smartcard_Receive_Timeout))
192 {
193 Counter++;
194 }
195
196 if (Counter != Smartcard_Receive_Timeout)
197 {
198 ReceivedBuffer[i] = USART_RxData(USART3);
199 }
200 }
201
202 Counter =0;
203
204 for (uint16_t i = 0; i < 40; i++)
205 {
206 if ( ReceivedBuffer[i] == 0)
207 {
208 Counter ++;
209 }
210 }
211
212 if (Counter == 40)
213 {
214 /* The data received by the Smartcard is abnormal */
215 return ERROR;
216 }
217
218 /* Decode ATR, Test if the inserted card is ISO7816-3 T=0 compatible */
219 if (Smartcard_decode_Answer2reset(ReceivedBuffer) == 0)
220 {
221 /* Inserted card is ISO7816-3 T=0 compatible */
222 return SUCCESS;
223 }
224 else
225 {
226 /* Inserted Smartcard is not ISO7816-3 T=0 compatible */
227 return ERROR;
228 }
229}

239void SmartcardInit(void)
240{
241 GPIO_Config_T GPIO_ConfigStruct;
242 USART_Config_T USART_ConfigStruct;
243 USART_ClockConfig_T USART_ClockConfigStruct;
244
245 /* System Clocks Configuration */
246 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOB | RCM_APB2_PERIPH_AFIO);
247
248 /* Enable SC_USART clocks */
249 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_USART3);
250
251 /* Configure the GPIO ports */
252 /* USART3 GPIO Configuration
253 PB10 (USART3_TX) ---> Smartcard Tx Pin
254 PB11 (Pin) ---> Smartcard Reset Pin
255 PB12 (USART3_CK) ---> Smartcard CK Pin
256 */
257
258 /* Configure Smartcard USART (USART3) CK as alternate function push-pull */
259 GPIO_ConfigStruct.pin = GPIO_PIN_12;
260 GPIO_ConfigStruct.speed = GPIO_SPEED_50MHz;
261 GPIO_ConfigStruct.mode = GPIO_MODE_AF_PP;
262 GPIO_Config(GPIOB, &GPIO_ConfigStruct);
263
264 /* Configure Smartcard USART (USART3) Tx as alternate function open-drain */
265 GPIO_ConfigStruct.pin = GPIO_PIN_10;
266 GPIO_ConfigStruct.mode = GPIO_MODE_AF_OD;
267 GPIO_Config(GPIOB, &GPIO_ConfigStruct);
268
269 /* Configure Smartcard Reset Pin */
270 GPIO_ConfigStruct.pin = GPIO_PIN_11;
271 GPIO_ConfigStruct.mode = GPIO_MODE_OUT_OD;
272 GPIO_Config(GPIOB, &GPIO_ConfigStruct);
273
274 /* Set Smartcard Reset Pin to High */
275 GPIO_SetBit(GPIOB, GPIO_PIN_11);
276
277 /* Configure the NVIC Preemption Priority Bits */
278 NVIC_ConfigPriorityGroup(NVIC_PRIORITY_GROUP_1);
279 NVIC_EnableIRQRequest(USART3_IRQn, 1, 0);
280
281 /* Smartcard USART (USART3) configuration */
282 /* Smartcard USART (USART3) configured as follow:
283 - Word Length = 9 Bits
284 - 1.5 Stop Bit
285 - Even parity
286 - BaudRate = 9677 baud
287 - Hardware flow control disabled (RTS and CTS signals)
288 - Tx and Rx enabled
289 - USART Clock enabled
290 - USART CPOL Low
291 - USART CPHA on first edge
292 - USART Last Bit Clock Enabled
293 */
294
295 /* Smartcard USART (USART3) Clock set to 4.5MHz (PCLK1 = 36 MHZ / 10) */
296 USART_ConfigPrescaler(USART3, 0x05);
297 /* Smartcard USART (USART3) Guard Time set to 16 Bit */
298 USART_ConfigGuardTime(USART3, 16);
299
300 USART_ClockConfigStruct.clock = USART_CLKEN_ENABLE;
301 USART_ClockConfigStruct.polarity = USART_CLKPOL_LOW;
302 USART_ClockConfigStruct.phase = USART_CLKPHA_1EDGE;
303 USART_ClockConfigStruct.lastBit = USART_LBCP_ENABLE;
304 USART_ConfigClock(USART3, &USART_ClockConfigStruct);
305
306 USART_ConfigStruct.baudRate = 9677;
307 USART_ConfigStruct.wordLength = USART_WORD_LEN_9B;
308 USART_ConfigStruct.stopBits = USART_STOP_BIT_1_5;
309 USART_ConfigStruct.parity = USART_PARITY_EVEN;
310 USART_ConfigStruct.mode = USART_MODE_TX_RX;
311 USART_ConfigStruct.hardwareFlow = USART_HARDWARE_FLOW_NONE;
312 USART_Config(USART3, &USART_ConfigStruct);
313
314 /* Enable the Smartcard USART (USART3) Parity Error Interrupt */
315 USART_EnableInterrupt(USART3, USART_INT_PE);
316
317 /* Enable Smartcard USART (USART3) */
318 USART_Enable(USART3);
319
320 /* Enable the NACK Transmission */
321 USART_EnableSmartCardNACK(USART3);
322
323 /* Enable the Smartcard Interface */
324 USART_EnableSmartCard(USART3);
325}

335uint8_t Smartcard_decode_Answer2reset(__IO uint8_t* cardBuff)
336{
337 uint32_t i = 0, flag = 0, buf = 0, protocol = 0;
338
339 Smartcard_A2RStruct.TS = 0;
340 Smartcard_A2RStruct.T0 = 0;
341
342 for (i = 0; i < Smartcard_Setup_Lenght; i++)
343 {
344 Smartcard_A2RStruct.T[i] = 0;
345 }
346
347 for (i = 0; i < Smartcard_Hist_Lenght; i++)
348 {
349 Smartcard_A2RStruct.H[i] = 0;
350 }
351
352 Smartcard_A2RStruct.Tlenght = 0;
353 Smartcard_A2RStruct.Hlenght = 0;
354
355 Smartcard_A2RStruct.TS = cardBuff[0]; /* INITIAL CHARACTER */
356 Smartcard_A2RStruct.T0 = cardBuff[1]; /* FORMAT CHARACTER */
357
358 Smartcard_A2RStruct.Hlenght = Smartcard_A2RStruct.T0 & 0x0F;
359
360 if ((Smartcard_A2RStruct.T0 & 0x80) == 0x80)
361 {
362 flag = 1;
363 }
364
365 for (i = 0; i < 4; i++)
366 {
367 Smartcard_A2RStruct.Tlenght = Smartcard_A2RStruct.Tlenght + \
368 (((Smartcard_A2RStruct.T0 & 0xF0) >> (4 + i)) & 0x1);
369 }
370
371 for (i = 0; i < Smartcard_A2RStruct.Tlenght; i++)
372 {
373 Smartcard_A2RStruct.T[i] = cardBuff[i + 2];
374 }
375
376 protocol = Smartcard_A2RStruct.T[Smartcard_A2RStruct.Tlenght - 1] & 0x0F;
377
378 while (flag)
379 {
380 if ((Smartcard_A2RStruct.T[Smartcard_A2RStruct.Tlenght-1] & 0x80)== 0x80)
381 {
382 flag = 1;
383 }
384 else
385 {
386 flag = 0;
387 }
388
389 buf = Smartcard_A2RStruct.Tlenght;
390 Smartcard_A2RStruct.Tlenght = 0;
391
392 for (i = 0; i < 4; i++)
393 {
394 Smartcard_A2RStruct.Tlenght = Smartcard_A2RStruct.Tlenght + \
395 (((Smartcard_A2RStruct.T[buf - 1] & 0xF0) >> (4 + i)) & 0x1);
396 }
397
398 for (i = 0; i < Smartcard_A2RStruct.Tlenght; i++)
399 {
400 Smartcard_A2RStruct.T[buf + i] = cardBuff[i + 2 + buf];
401 }
402
403 Smartcard_A2RStruct.Tlenght += buf;
404 }
405
406 for (i = 0; i < Smartcard_A2RStruct.Hlenght; i++)
407 {
408 Smartcard_A2RStruct.H[i] = cardBuff[i + 2 + Smartcard_A2RStruct.Tlenght];
409 }
410
411 return ((uint8_t)protocol);
412}

422void SmartcardIRQHandler(void)
423{
424 /* If the SC_USART detects a parity error */
425 if (USART_ReadIntFlag(USART3, USART_INT_PE) != RESET)
426 {
427 /* Enable SC_USART RXNE Interrupt (until receiving the corrupted byte) */
428 USART_EnableInterrupt(USART3, USART_INT_RXBNE);
429 /* Flush the SC_USART DR register */
430 USART_RxData(USART3);
431 }
432
433 if (USART_ReadIntFlag(USART3, USART_INT_RXBNE) != RESET)
434 {
435 /* Disable SC_USART RXNE Interrupt */
436 USART_DisableInterrupt(USART3, USART_INT_RXBNE);
437 USART_RxData(USART3);
438 }
439}
