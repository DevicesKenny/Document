145USBH_CH_STATUS_T USBH_CH_OpenChannel(uint8_t chNum,
146 uint8_t devAddr,
147 uint8_t epType,
148 uint16_t maxPackSize)
149{
150 if (chNum >= USBH_CHANNEL_MAX_NUM)
151 return USBH_CH_FAIL;
152
153 USB_ChannelConfig_T chConfig;
154
155 /* Config transfer Channel info */
156 g_usbHost.xferPipe[chNum].maxPackSize = maxPackSize;
157 g_usbHost.xferPipe[chNum].enbalePing = g_usbHost.speed ? 0 : 1;
158
159 /* Config Channel */
160 chConfig.chNum = chNum;
161 chConfig.devAddr = devAddr;
162 chConfig.epType = epType;
163 chConfig.devSpeed = g_usbHost.speed;
164 chConfig.direction = (g_usbHost.xferPipe[chNum].epAddr >> 7) & 0x01;
165 chConfig.epNum = g_usbHost.xferPipe[chNum].epAddr & 0x7F;
166 chConfig.maxPackSize = g_usbHost.xferPipe[chNum].maxPackSize;
167 chConfig.enbalePing = g_usbHost.xferPipe[chNum].enbalePing;
168
169 USB_OTG_InitChannel(&chConfig);
170
171 return USBH_CH_OK;
172}

181uint8_t USBH_ReadEpAddrChannel(uint8_t epAddr)
