306void USART_Init(void)
307{
308 USART_Config_T usartConfigStruct;
309
310 usartConfigStruct.baudRate = 115200;
311 usartConfigStruct.hardwareFlow = USART_HARDWARE_FLOW_NONE;
312 usartConfigStruct.mode = USART_MODE_TX_RX;
313 usartConfigStruct.parity = USART_PARITY_NONE;
314 usartConfigStruct.stopBits = USART_STOP_BIT_1;
315 usartConfigStruct.wordLength = USART_WORD_LEN_8B;
316
317 APM_MINI_COMInit(COM1, &usartConfigStruct);
318 APM_MINI_COMInit(COM2, &usartConfigStruct);
319
320 Delay(0x7FFF);
321
322 /* Disable USART1 RXBNE interrput */
323 USART_DisableInterrupt(MINI_COM1,USART_INT_RXBNE);
324 USART_DisableInterrupt(MINI_COM2,USART_INT_RXBNE);
325
326 /* Disable USART2 RXBNE interrput */
327 USART_ClearStatusFlag(MINI_COM1, USART_FLAG_RXBNE);
328 USART_ClearStatusFlag(MINI_COM2, USART_FLAG_RXBNE);
329}
