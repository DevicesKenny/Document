136void DMA_Init(void)
137{
138 DMA_Config_T dmaConfig;
139
140 /* Enable DMA Clock */
141 RCM_EnableAHBPeriphClock(RCM_AHB_PERIPH_DMA1);
142
143 /* DMA config */
144 dmaConfig.peripheralBaseAddr = (uint32_t)&TMR1->CC1;
145 dmaConfig.memoryBaseAddr = (uint32_t)ConfigBuf;
146 dmaConfig.dir = DMA_DIR_PERIPHERAL_DST;
147 dmaConfig.bufferSize = 3;
148 dmaConfig.peripheralInc = DMA_PERIPHERAL_INC_DISABLE;
149 dmaConfig.memoryInc = DMA_MEMORY_INC_ENABLE;
150 dmaConfig.peripheralDataSize = DMA_PERIPHERAL_DATA_SIZE_HALFWORD;
151 dmaConfig.memoryDataSize = DMA_MEMORY_DATA_SIZE_HALFWORD;
152 dmaConfig.loopMode = DMA_MODE_CIRCULAR;
153 dmaConfig.priority = DMA_PRIORITY_HIGH;
154 dmaConfig.M2M = DMA_M2MEN_DISABLE;
155
156 /* Config TMR1 UP DMA channel */
157 DMA_Config(DMA1_Channel5, &dmaConfig);
158
159 /* Enable DMA */
160 DMA_Enable(DMA1_Channel5);
161}
