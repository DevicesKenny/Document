57int main(void)

70 /* Config TMR1 GPIO for complementary output PWM */
71 /* CH1->PA8, CH2->PA9, CH3->PA10 */
72 GPIO_ConfigStruct.pin = GPIO_PIN_8 | GPIO_PIN_9 | GPIO_PIN_10;
73 GPIO_ConfigStruct.mode = GPIO_MODE_AF_PP;
74 GPIO_ConfigStruct.speed = GPIO_SPEED_50MHz;
75 GPIO_Config(GPIOA, &GPIO_ConfigStruct);

77 /* BKIN->PB12, CH1N->PB13, CH2N->PB14, CH3N->PB15 */
78 GPIO_ConfigStruct.pin = GPIO_PIN_12 | GPIO_PIN_13 | GPIO_PIN_14| GPIO_PIN_15;
79 GPIO_Config(GPIOB, &GPIO_ConfigStruct);
80
81 /* config TMR1 */
82 TMR_TimeBaseStruct.clockDivision = TMR_CLOCK_DIV_1;
83 TMR_TimeBaseStruct.countMode = TMR_COUNTER_MODE_UP;
84 TMR_TimeBaseStruct.division = 71;
85 TMR_TimeBaseStruct.period = 2047;
86 TMR_TimeBaseStruct.repetitionCounter = 0;
87 TMR_ConfigTimeBase(TMR1, &TMR_TimeBaseStruct);
88
89 /* Configure channel */
90 OCcongigStruct.mode = TMR_OC_MODE_TMRING;
91 OCcongigStruct.idleState = TMR_OC_IDLE_STATE_SET;
92 OCcongigStruct.outputState = TMR_OC_STATE_ENABLE;
93 OCcongigStruct.polarity = TMR_OC_POLARITY_HIGH;
94 OCcongigStruct.nIdleState = TMR_OC_NIDLE_STATE_SET;
95 OCcongigStruct.outputNState = TMR_OC_NSTATE_ENABLE;
96 OCcongigStruct.nPolarity = TMR_OC_NPOLARITY_HIGH;
97 OCcongigStruct.pulse = 1023;
98
99 TMR_ConfigOC1(TMR1, &OCcongigStruct);
100
101 OCcongigStruct.pulse = 511;
102 TMR_ConfigOC2(TMR1, &OCcongigStruct);
103
104 OCcongigStruct.pulse = 255;
105 TMR_ConfigOC3(TMR1, &OCcongigStruct);
106
107 /* Configures the Break feature, dead time, Lock level, the IMOS */
108 BDTConfig.RMOS = TMR_RMOS_STATE_ENABLE;
109 BDTConfig.IMOS = TMR_IMOS_STATE_ENABLE;
110 BDTConfig.lockLevel = TMR_LOCK_LEVEL_OFF;
111 BDTConfig.deadTime = 0x01;
112 BDTConfig.BRKState = TMR_BRK_STATE_ENABLE;
113 BDTConfig.BRKPolarity = TMR_BRK_POLARITY_HIGH;
114 BDTConfig.automaticOutput = TMR_AUTOMATIC_OUTPUT_ENABLE;
115 TMR_ConfigBDT(TMR1, &BDTConfig);
116
117 /* Timer1 Commutation Interrupt */
118 TMR_EnableInterrupt(TMR1,TMR_INT_COM);
119 NVIC_EnableIRQRequest(TMR1_TRG_COM_IRQn, 0, 0);
120
121 /* Enable Capture Compare Preload Control */
122 TMR_EnableCCPreload(TMR1);
123
124 TMR_Enable(TMR1);
125 TMR_EnablePWMOutputs(TMR1);
126
127 while(1)
128 {
129 /* Generate the TMR1 Commutation event by software every 100 milliseconds */
130 Delay_ms(100);
131 TMR_GenerateEvent(TMR1, TMR_EVENT_COM);
132 }
133}
