74int main(void)
75{
76 USART_Config_T USART_ConfigStruct;
77
78 APM_MINI_LEDInit(LED2);
79
80 /* COM1(USART1) configured as follow:
81 - BaudRate = 115200 baud
82 - Word Length = 8 Bits
83 - One Stop Bit
84 - No parity
85 - Hardware flow control disabled (RTS and CTS signals)
86 - Receive and transmit enabled
87 */
88
89 USART_ConfigStruct.baudRate = 115200;
90 USART_ConfigStruct.hardwareFlow = USART_HARDWARE_FLOW_NONE;
91 USART_ConfigStruct.mode = USART_MODE_TX;
92 USART_ConfigStruct.parity = USART_PARITY_NONE;
93 USART_ConfigStruct.stopBits = USART_STOP_BIT_1;
94 USART_ConfigStruct.wordLength = USART_WORD_LEN_8B;
95
96 APM_MINI_COMInit(COM1, &USART_ConfigStruct);
97
98 IrDAInit();
99
100 SystemCoreClockUpdate();
101 SysTick_Config(SystemCoreClock / 1000);
102
103 printf("\r\nThis example provides a basic communication USARTx IrDA receive and transmit mode.\r\n");
104 printf(">> SystemCoreClock: %d\r\n",SystemCoreClock);
105
106 while (1)
107 {
108 Delay();
109
110 APM_MINI_LEDToggle(LED2);
111
112 IrDA_USART2_TxData(USART2SendData);
113 USART3ReceivedData = IrDA_USART3_RxData();
114 printf(">> USART2 Send Data : 0x%02X ,USART3 Received Data : 0x%02X. \r\n", USART2SendData, USART3ReceivedData);
115
116 IrDA_USART3_TxData(USART3SendData);
117 USART2ReceivedData = IrDA_USART2_RxData();
118 printf(">> USART3 Send Data : 0x%02X ,USART2 Received Data : 0x%02X. \r\n", USART3SendData, USART2ReceivedData);
119
120 USART2SendData++;
121 USART3SendData--;
122 }
123}

134void IrDAInit(void)
135{
136 GPIO_Config_T GPIO_ConfigStruct;
137 USART_Config_T USART_ConfigStruct;
138
139 /* System Clocks Configuration */
140 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA | RCM_APB2_PERIPH_GPIOB | RCM_APB2_PERIPH_AFIO);
141 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_USART2);
142 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_USART3);
143
144 /* Configure the GPIO ports */
145 /* USART2 GPIO Configuration
146 PA2 ------> USART2_TX
147 PA3 ------> USART2_RX
148 */
149 GPIO_ConfigStruct.pin = GPIO_PIN_2;
150 GPIO_ConfigStruct.mode = GPIO_MODE_AF_PP;
151 GPIO_ConfigStruct.speed = GPIO_SPEED_50MHz;
152 GPIO_Config(GPIOA, &GPIO_ConfigStruct);
153
154 GPIO_ConfigStruct.pin = GPIO_PIN_3;
155 GPIO_ConfigStruct.mode = GPIO_MODE_IN_FLOATING;
156 GPIO_Config(GPIOA, &GPIO_ConfigStruct);
157
158 /* USART3 GPIO Configuration
159 PB10 ------> USART3_TX
160 PB11 ------> USART3_RX
161 */
162 GPIO_ConfigStruct.pin = GPIO_PIN_10;
163 GPIO_ConfigStruct.mode = GPIO_MODE_AF_PP;
164 GPIO_ConfigStruct.speed = GPIO_SPEED_50MHz;
165 GPIO_Config(GPIOB, &GPIO_ConfigStruct);
166
167 GPIO_ConfigStruct.pin = GPIO_PIN_11;
168 GPIO_ConfigStruct.mode = GPIO_MODE_IN_FLOATING;
169 GPIO_Config(GPIOB, &GPIO_ConfigStruct);
170
171 /* USARTx configuration */
172 /* USARTx configured as follow:
173 - BaudRate = 115200 baud
174 - Word Length = 8 Bits
175 - One Stop Bit
176 - No parity
177 - Hardware flow control disabled (RTS and CTS signals)
178 - Receive and transmit enabled
179 */
180 USART_ConfigStruct.baudRate = 115200;
181 USART_ConfigStruct.hardwareFlow = USART_HARDWARE_FLOW_NONE;
182 USART_ConfigStruct.mode = USART_MODE_TX_RX;
183 USART_ConfigStruct.parity = USART_PARITY_NONE;
184 USART_ConfigStruct.stopBits = USART_STOP_BIT_1;
185 USART_ConfigStruct.wordLength = USART_WORD_LEN_8B;
186
187 /* Configure the USARTx */
188 USART_Config(USART2, &USART_ConfigStruct);
189 USART_Config(USART3, &USART_ConfigStruct);
190
191 /* Enable the USARTx */
192 USART_Enable(USART2);
193 USART_Enable(USART3);
194
195 /* Set the USARTx prescaler */
196 USART_ConfigPrescaler(USART2, 0x01);
197 USART_ConfigPrescaler(USART3, 0x01);
198
199 /* Configure the USARTx IrDA mode */
200 USART_ConfigIrDA(USART2, USART_IRDALP_NORMAL);
201 USART_ConfigIrDA(USART3, USART_IRDALP_NORMAL);
202
203 /* Enable the USARTx IrDA mode */
204 USART_EnableIrDA(USART2);
205 USART_EnableIrDA(USART3);
206}

217void IrDA_USART2_TxData(uint8_t data)
218{
219 USART_TxData(USART2, data);
220
221 /* wait for the data to be send */
222 while (USART_ReadStatusFlag(USART2, USART_FLAG_TXBE) == RESET);
