310void USBD_ConfigSerialString(uint8_t *string)
311{
312 uint32_t serialNum1 = 0;
313 uint16_t serialNum2 = 0;
314 uint8_t index, movCnt = 0;
315
316 serialNum1 = (*(uint32_t *)UID_ADDRESS1) + (*(uint32_t *)UID_ADDRESS2);
317 serialNum2 = (uint16_t)((*(uint32_t *)UID_ADDRESS1) + (*(uint32_t *)UID_ADDRESS3));
318
319 for (index = 2; index <= 16; index += 2)
320 {
321 string[index] = USBD_NumberToChar((uint8_t)(serialNum1 >> movCnt) & 0x0F);
322
323 if (movCnt < 16)
324 {
325 string[index + 16] = USBD_NumberToChar(((uint8_t)(serialNum2 >> movCnt) & 0x0F));
326 }
327
328 movCnt += 4;
329 }
