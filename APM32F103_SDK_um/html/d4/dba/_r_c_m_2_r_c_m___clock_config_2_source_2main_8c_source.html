80void SetSysClock(void)
81{
82 RCM_Reset();
83
84 RCM_ConfigHSE(RCM_HSE_OPEN);
85
86 if(RCM_WaitHSEReady() == SUCCESS)
87 {
88 FMC_EnablePrefetchBuffer();
89 FMC_ConfigLatency(FMC_LATENCY_2);
90
91 RCM_ConfigAHB(RCM_AHB_DIV_1);
92 RCM_ConfigAPB2(RCM_APB_DIV_1);
93 RCM_ConfigAPB1(RCM_APB_DIV_2);
94
95 RCM_ConfigPLL(RCM_PLLSEL_HSE, RCM_PLLMF_9);
96 RCM_EnablePLL();
97 while(RCM_ReadStatusFlag(RCM_FLAG_PLLRDY) == RESET);
98
99 RCM_ConfigSYSCLK(RCM_SYSCLK_SEL_PLL);
100 while(RCM_ReadSYSCLKSource() != RCM_SYSCLK_SEL_PLL);
101 }
102 else
103 {
104 while(1);
105 }
106}
