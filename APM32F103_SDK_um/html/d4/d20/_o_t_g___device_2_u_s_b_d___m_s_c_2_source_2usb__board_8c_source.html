241int fputc(int ch, FILE* f)
242{
243 /* send a byte of data to the serial port */
244 USART_TxData(DEBUG_USART, (uint8_t)ch);
245
246 /* wait for the data to be send */
247 while (USART_ReadStatusFlag(DEBUG_USART, USART_FLAG_TXBE) == RESET);
248
249 return (ch);
250}
