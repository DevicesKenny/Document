39void USBD_SetupProcess(void)
40{
41 uint8_t reqType;
42 uint8_t dataBuf[8];
43 USBD_DevReqData_T* pReqData = &g_usbDev.reqData;
44 uint16_t xferLen = USBD_ReadEPRxCnt(USBD_EP_0);
45
46 if (xferLen)
47 {
48 USBD_ReadDataFromEP(USBD_EP_0, (uint8_t*)dataBuf, xferLen);
49 }
50 else
51 {
52 return;
53 }
54
55 pReqData->byte.bmRequestType.byte = dataBuf[0];
56 pReqData->byte.bRequest = dataBuf[1];
57 pReqData->byte.wValue[0] = dataBuf[2];
58 pReqData->byte.wValue[1] = dataBuf[3];
59 pReqData->byte.wIndex[0] = dataBuf[4];
60 pReqData->byte.wIndex[1] = dataBuf[5];
61 pReqData->byte.wLength[0] = dataBuf[6];
62 pReqData->byte.wLength[1] = dataBuf[7];
63
64 reqType = pReqData->byte.bmRequestType.bit.type;
65
66 if (reqType == USBD_REQ_TYPE_STANDARD)
67 {
68 USBD_StandardReqeust();
69 }
70 else if (reqType == USBD_REQ_TYPE_CLASS)
71 {
72 if (g_usbDev.classReqHandler)
73 {
74 g_usbDev.classReqHandler(pReqData);
75 }
76 }
77 else if (reqType == USBD_REQ_TYPE_VENDOR)
78 {
79 if (g_usbDev.vendorReqHandler)
80 {
81 g_usbDev.vendorReqHandler(pReqData);
82 }
83 }
84 else
85 {
86 USBD_SetEPTxRxStatus(USBD_EP_0, USBD_EP_STATUS_STALL, USBD_EP_STATUS_STALL);
87 }

98void USBD_CtrlInProcess(void)
99{
100 uint32_t tmp;
101
102 if (g_usbDev.ctrlState == USBD_CTRL_STATE_DATA_IN)
103 {
104 if (g_usbDev.inBuf[0].packNum)
105 {
106 tmp = USB_MIN(g_usbDev.inBuf[0].bufLen, g_usbDev.inBuf[0].maxPackSize);
107
108 USBD_WriteDataToEP(USBD_EP_0, g_usbDev.inBuf[0].pBuf, tmp);
109 USBD_SetEPTxCnt(USBD_EP_0, tmp);
110 USBD_SetEPTxRxStatus(USBD_EP_0, USBD_EP_STATUS_VALID, USBD_EP_STATUS_NAK);
111
112 g_usbDev.inBuf[0].pBuf += tmp;
113 g_usbDev.inBuf[0].bufLen -= tmp;
114 g_usbDev.inBuf[0].packNum--;
115 }
116 else
117 {
118 if (g_usbDev.inBuf[USBD_EP_0].zeroPackFill)
119 {
120 USBD_SetEPTxCnt(USBD_EP_0, 0);
121 USBD_SetEPTxStatus(USBD_EP_0, USBD_EP_STATUS_VALID);
122 g_usbDev.inBuf[USBD_EP_0].zeroPackFill = 0;
123 }
124 else
125 {
126 if (g_usbDev.rxStatusHandler)
127 {
128 g_usbDev.rxStatusHandler();
129 }
130
131 g_usbDev.ctrlState = USBD_CTRL_STATE_WAIT_STATUS_OUT;
132 USBD_SetEPTxRxStatus(USBD_EP_0, USBD_EP_STATUS_NAK, USBD_EP_STATUS_VALID);
133 }
134
135 }
136 }
137 else if (g_usbDev.ctrlState == USBD_CTRL_STATE_WAIT_STATUS_IN)
138 {
139 if (g_usbDev.reqData.byte.bRequest == USBD_SET_ADDRESS)
140 {
141 USBD_SetDeviceAddr(g_usbDev.reqData.byte.wValue[0]);
142 }

199void USBD_CtrlInData(uint8_t* buf, uint32_t len)
200{
201 uint16_t maxPackSize = g_usbDev.inBuf[0].maxPackSize;
202 uint16_t reqLen = *(uint16_t*)g_usbDev.reqData.byte.wLength;
203
204 if (len)
205 {
206 if ((len < reqLen) && ((len % maxPackSize) == 0))
207 {
208 g_usbDev.inBuf[USBD_EP_0].zeroPackFill = 1;
209 }
210
211 if (len >= g_usbDev.inBuf[0].maxPackSize)
212 {
214 USBD_WriteDataToEP(USBD_EP_0, buf, g_usbDev.inBuf[0].maxPackSize);
215 USBD_SetEPTxCnt(USBD_EP_0, g_usbDev.inBuf[0].maxPackSize);
216 USBD_SetEPTxRxStatus(USBD_EP_0, USBD_EP_STATUS_VALID, USBD_EP_STATUS_NAK);
217
219 g_usbDev.inBuf[0].bufLen = len - g_usbDev.inBuf[0].maxPackSize;
220 g_usbDev.inBuf[0].pBuf = buf + g_usbDev.inBuf[0].maxPackSize;
221 g_usbDev.inBuf[0].packNum = (g_usbDev.inBuf[0].bufLen + (maxPackSize - 1)) / maxPackSize;
222
223 g_usbDev.ctrlState = USBD_CTRL_STATE_DATA_IN;
224 }
225 else
226 {
227 USBD_WriteDataToEP(USBD_EP_0, buf, len);
228 USBD_SetEPTxCnt(USBD_EP_0, len);
229 USBD_SetEPTxRxStatus(USBD_EP_0, USBD_EP_STATUS_VALID, USBD_EP_STATUS_NAK);
230
231 g_usbDev.ctrlState = g_usbDev.reqData.byte.bmRequestType.bit.dir ? \
232 USBD_CTRL_STATE_DATA_IN : \
233 USBD_CTRL_STATE_WAIT_STATUS_IN;
234 }
235 }
236 else
237 {
238 USBD_SetEPTxCnt(USBD_EP_0, 0);
239 USBD_SetEPTxStatus(USBD_EP_0, USBD_EP_STATUS_VALID);
240
241 g_usbDev.ctrlState = g_usbDev.reqData.byte.bmRequestType.bit.dir ? \
242 USBD_CTRL_STATE_DATA_IN : \
243 USBD_CTRL_STATE_WAIT_STATUS_IN;
244 }
