222static USBD_REG_EP_TYPE_T USBD_ConvertEPType(USBD_EP_TYPE_T epType)
223{
224 switch (epType)
225 {
226 case USBD_EP_TYPE_CONTROL :
227 return USBD_REG_EP_TYPE_CONTROL;
228 case USBD_EP_TYPE_ISO :
229 return USBD_REG_EP_TYPE_ISO;
230 case USBD_EP_TYPE_BULK :
231 return USBD_REG_EP_TYPE_BULK;
232 case USBD_EP_TYPE_INTERRUPT :
233 return USBD_REG_EP_TYPE_INTERRUPT;
234 default :
235 return USBD_REG_EP_TYPE_CONTROL;
236 }
