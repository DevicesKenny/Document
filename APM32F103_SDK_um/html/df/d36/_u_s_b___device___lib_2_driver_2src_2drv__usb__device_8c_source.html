301void USBD_SetEPRxCnt(uint8_t ep, uint32_t cnt)
302{
303 __IOM uint16_t* p;

305
306 p = USBD_ReadEPRxCntPointer(ep);
307
308 if (cnt > 62)

378void USBD_ReadDataFromEP(uint8_t ep, uint8_t* rBuf, uint32_t rLen)
379{

381 uint16_t* epAddr;
382#else

386

390

393 tmp = *epAddr++;
