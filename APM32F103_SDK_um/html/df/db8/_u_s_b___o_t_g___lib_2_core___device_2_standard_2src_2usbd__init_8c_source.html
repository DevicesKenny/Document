198static void USB_DeviceInit(void)
199{
200 /* Device mode or host mode */
201 USB_OTG_ConfigMode(USB_MODE_DEVICE);
202
203 /* PHY Clock restart */
204 USB_OTG_PWR->PCGCTRL = 0;
205
206 /* Periodic frame interval */
207 USB_OTG_ConfigPeriodicFrameInterval(USB_FRAME_INTERVAL_80);
208
209#ifdef USB_OTG_FS
210 /* Device Speed */
211 USB_OTG_ConfigDeviceSpeed(USB_DEV_SPEED_FS);
212
213 /* RxFIFO Configuration */
214 USB_OTG_ConfigRxFifoDepth(USB_FS_RX_FIFO_SIZE);
215
216 uint16_t txFifoAddr[4];
217 uint16_t txFifoDepth[4] =
218 {
219 USB_FS_TX_FIFO_0_SIZE, USB_FS_TX_FIFO_1_SIZE,
220 USB_FS_TX_FIFO_2_SIZE, USB_FS_TX_FIFO_3_SIZE
221 };
222
223 txFifoAddr[0] = USB_FS_RX_FIFO_SIZE;
224 txFifoAddr[1] = txFifoAddr[0] + USB_FS_TX_FIFO_0_SIZE;
225 txFifoAddr[2] = txFifoAddr[1] + USB_FS_TX_FIFO_1_SIZE;
226 txFifoAddr[3] = txFifoAddr[2] + USB_FS_TX_FIFO_2_SIZE;
227
228 /* TxFIFO Configuration */
229 for (uint8_t ep = 0; ep < USB_IN_EP_MAX_NUM; ep++)
230 {
231 USB_OTG_ConfigDeviceTxFifo(ep, txFifoAddr[ep], txFifoDepth[ep]);
232 }
233#else /* HS */
234 /* Device Speed */
235 USB_OTG_ConfigDeviceSpeed(USB_DEV_SPEED_HS);
236
237 /* RxFIFO Configuration */
238 USB_OTG_ConfigRxFifoDepth(USB_HS_RX_FIFO_SIZE);
239
240 uint16_t txFifoAddr[8];
241 uint16_t txFifoDepth[8] =
242 {
243 USB_HS_TX_FIFO_0_SIZE, USB_HS_TX_FIFO_1_SIZE,
244 USB_HS_TX_FIFO_2_SIZE, USB_HS_TX_FIFO_3_SIZE,
245 USB_HS_TX_FIFO_4_SIZE, USB_HS_TX_FIFO_5_SIZE,
246 USB_HS_TX_FIFO_6_SIZE, USB_HS_TX_FIFO_7_SIZE
247 };
248
249 txFifoAddr[0] = USB_HS_RX_FIFO_SIZE;
250 txFifoAddr[1] = txFifoAddr[0] + USB_HS_TX_FIFO_0_SIZE;
251 txFifoAddr[2] = txFifoAddr[1] + USB_HS_TX_FIFO_1_SIZE;
252 txFifoAddr[3] = txFifoAddr[2] + USB_HS_TX_FIFO_2_SIZE;
253 txFifoAddr[4] = txFifoAddr[3] + USB_HS_TX_FIFO_3_SIZE;
254 txFifoAddr[5] = txFifoAddr[4] + USB_HS_TX_FIFO_4_SIZE;
255 txFifoAddr[6] = txFifoAddr[5] + USB_HS_TX_FIFO_5_SIZE;
256 txFifoAddr[7] = txFifoAddr[6] + USB_HS_TX_FIFO_6_SIZE;
257
258 /* TxFIFO Configuration */
259 for (uint8_t ep = 0; ep < 8; ep++)
260 {
261 USB_OTG_ConfigDeviceTxFifo(ep, txFifoAddr[ep], txFifoDepth[ep]);
262 }
263#endif
264 /* Flush TxFIFO */
265 USB_OTG_FlushTxFIFO(0x10);
266
267 /* Flush RxFIFO */
268 USB_OTG_FlushRxFIFO();
269
270 USB_OTG_DisableOutEpInterrupt(USB_INT_EP_OUT_ALL);
271 USB_OTG_DisableInEpInterrupt(USB_INT_EP_IN_ALL);
272
273 USB_OTG_DisableOutEpInterruptMask(USB_EP_0);
274 USB_OTG_D->DAEPIMASK_B.AOUTM = 0;
275
276 uint8_t i;
277
278 /* Init endpoint */
279 for (i = 0; i < USB_MAX(USB_OUT_EP_MAX_NUM, USB_IN_EP_MAX_NUM); i++)
280 {
281 /* Reset OUT endpoint */
282 if (USB_OTG_D->EP_OUT[i].DOEPCTRL_B.EPEN)
283 {
284 USB_OTG_D->EP_OUT[i].DOEPCTRL_B.EPDIS = BIT_SET;
285 USB_OTG_D->EP_OUT[i].DOEPCTRL_B.NAKSET = BIT_SET;
286 }
287 else
288 {
289 USB_OTG_D->EP_OUT[i].DOEPCTRL = 0;
290 }
291
292 /* Reset IN endpoint */
293 if (USB_OTG_D->EP_IN[i].DIEPCTRL_B.EPEN)
294 {
295 USB_OTG_D->EP_IN[i].DIEPCTRL_B.EPDIS = BIT_SET;
296 USB_OTG_D->EP_IN[i].DIEPCTRL_B.NAKSET = BIT_SET;
297 }
298 else
299 {
300 USB_OTG_D->EP_IN[i].DIEPCTRL = 0;
301 }
302
303 /* Reset transfer size */
304 USB_OTG_D->EP_OUT[i].DOEPTRS = 0;
305 USB_OTG_D->EP_IN[i].DIEPTRS = 0;
306
307 /* Clear endpoint Flag */
308 USB_OTG_D->EP_OUT[i].DOEPINT = 0xFF;
309 USB_OTG_D->EP_IN[i].DIEPINT = 0xFF;
310 }

320void USB_CoreSoftReset(void)
321{
322 __IO uint32_t timeout = 0;
323
324 /* Wait for AHB master idle */
325 while (USB_OTG_ReadAHBMasterIdleFlag() != BIT_SET)
326 {
327 if (++timeout > 0x30000)
328 {
329 return;
330 }
331 }
332
333 timeout = 0;
334
335 USB_OTG_EnableCoreSoftReset();
336
337 /* Wait for Reset ok */
338 while (USB_OTG_ReadCoreSoftResetFlag() != BIT_RESET)
339 {
340 if (++timeout > 0x30000)
341 {
342 break;
343 }
344 }
345
346 USB_Delay_us(3);
347}
