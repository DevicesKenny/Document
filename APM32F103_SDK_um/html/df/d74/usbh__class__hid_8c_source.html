468static void HID_PollHandler(void)
469{
470 USBH_URB_STATUS_T urbStatus;
471 USBH_CTRL_STATE_T ctrlState;
472
473
474 if (((USB_OTG_H_ReadFrameNum() - hidTimer) >= HID_POLL_TIME) || (USB_OTG_H_ReadFrameNum() < hidTimer))
475 {
476 s_classState = HID_GET_DATA;
477 }
478 else
479 {
480 urbStatus = g_usbHost.xferPipe[2].urbStatus;
481
482 switch (urbStatus)
483 {
484 case USBH_URB_OK:
485 /* IN data OK */
486 /* Parse HID data */
487 if (flagParseHID == 1)
488 {
489 flagParseHID = 0;
490 ParseMouseData(dat);
491 }
492 break;
493 case USBH_URB_STALL:
494 /* Config Standard Clear Feature request */
495 if (g_usbHost.control.xferState == USBH_XFER_START)
496 {
497 USBH_StandardClearFeature(USBH_RECIPIENT_ENDPOINT, USBH_FEATURE_ENDPOINT_HALT, USB_OTG_H_ReadEpNum(2), 2);
498 g_usbHost.control.xferState = USBH_XFER_WAITING;
499 }
500 /* Do the transaction */
501 else if (g_usbHost.control.xferState == USBH_XFER_WAITING)
502 {
503 ctrlState = USBH_ControlXferHandler(&g_usbHost.control.reqData, NULL);
504
505 switch (ctrlState)
506 {
507 case USBH_CTRL_COMPLETE:
508 g_usbHost.control.xferState = USBH_XFER_OK;
509 break;
510 case USBH_CTRL_STALL :
511 case USBH_CTRL_ERROR :
512 g_usbHost.control.xferState = USBH_XFER_START;
513 break;
514 default :
515 break;
516 }
517 }
518 /* If Get Descriptor OK */
519 else if (g_usbHost.control.xferState == USBH_XFER_OK)
520 {
521 g_usbHost.control.xferState = USBH_XFER_START;
522 s_classState = HID_GET_DATA;
523 }
524 break;
525
526 default :
527 break;
528 }

539void ParseMouseData(uint8_t* rBuf)
540{
541 int i;
542
543 mouseData.button = rBuf[0];
544 mouseData.x = rBuf[1];
545 mouseData.y = rBuf[2];
546
547 for (i = 0; i < 20; i++)
548 {
549 printMouse[i] = '.';
550 }
551 printMouse[i] = '\n';
552
553 if ((mouseData.button & 0x01) == 0x01)
554 {
555 printMouse[0] = 'o';
556 }
557
558 if ((mouseData.button & 0x02) == 0x02)
559 {
560 printMouse[19] = 'o';
561 }
562
563 if (((mouseData.x & 0x80) == 0x80) && (mouseData.x > 0))
564 {
565 if (mouseSeat > 0)
566 {
567 mouseSeat --;
568 }
569 else
570 {
571 mouseSeat = 0;
572 }
573 }
574 else if (((mouseData.x & 0x80) == 0x00) && (mouseData.x > 0))
575 {
576 if (mouseSeat < 17)
577 {
578 mouseSeat ++;
579 }
580 else
581 {
582 mouseSeat = 17;
583 }
584 }
585
586 printMouse[mouseSeat + 1] = '|';
587 printf("%s", printMouse);
588}
