112static USBH_BOT_STATUS_T MSC_BOT_DecodeCSW(void)
113{
114 USBH_BOT_STATUS_T status = USBH_BOT_FAIL;
115
116 /* Check validity of the CSW Signature and CSWStatus */
117 if(g_BOTInfo.cmdPack.CSW.field.dSignature == MSC_BOT_CSW_SIGNATURE)
118 {
119 if(g_BOTInfo.cmdPack.CSW.field.dTag == g_BOTInfo.cmdPack.CBW.field.dTag)
120 {
121 if(g_BOTInfo.cmdPack.CSW.field.bStatus == USBH_BOT_OK)
122 {
123 status = USBH_BOT_OK;
124 }
125 else if(g_BOTInfo.cmdPack.CSW.field.bStatus == USBH_BOT_FAIL)
126 {
127 status = USBH_BOT_FAIL;
128 }
129 else if(g_BOTInfo.cmdPack.CSW.field.bStatus == USBH_BOT_PHASE_ERROR)
130 {
131 status = USBH_BOT_PHASE_ERROR;
132 }
133 }
134 }
135 else
136 {
137 /* If the CSW Signature is not valid, We Sall return the Phase Error to
138 Upper Layers for Reset Recovery */
139 status = USBH_BOT_PHASE_ERROR;
140 }
141
142 g_BOTInfo.BOTXferStatus = status;
143
144 return status;
145}

188static void MSC_BOT_SentCBWHandler(void)
189{
190 g_UrbStatus = g_usbHost.xferPipe[g_BOTInfo.outChNum].urbStatus;
191 if (g_UrbStatus != USBH_URB_IDLE)
192 {
193 if (test == 2)
194 {
195 __NOP();
196 }
197 }
198 switch(g_UrbStatus)
199 {
200 case USBH_URB_OK:
201 g_errorCntBOTStall = 0;
202 /* Data Stage */
203 if(g_BOTInfo.cmdPack.CBW.field.dDataXferLen != 0)
204 {
205 g_tempDataXferLen = g_BOTInfo.cmdPack.CBW.field.dDataXferLen;
206 g_pdata = g_BOTInfo.pXferBuf;
207 g_pdataPrev = g_pdata;
208
209 /* If there is Data Transfer Stage */
210 if(g_BOTInfo.cmdPack.CBW.field.bmFlags.bit.dir == USBH_REQ_DIR_IN)
211 {
212 USBH_BOT_ConfigBOTState(USBH_BOT_DATAIN);
213 }
214 else
215 {
216 USBH_BOT_ConfigBOTState(USBH_BOT_DATAOUT);
217 }
218 }
219 else
220 {
221 /* If there is NO Data Transfer Stage */
222 USBH_BOT_ConfigBOTState(USBH_BOT_RECEIVE_CSW);
223 }
224 g_UrbStatus = USBH_URB_IDLE;
225 break;
226
227 case USBH_URB_NOREADY:
228 g_BOTInfo.curBOTState = g_BOTInfo.preBOTState;
229 break;
230
231 case USBH_URB_STALL:
232 g_UrbStatus = USBH_URB_IDLE;
233 USBH_BOT_ConfigBOTState(USBH_BOT_ERROR_OUT);
234 break;
235
236 case USBH_URB_ERROR:
237 g_errorTimeout++;
238
239 if(g_errorTimeout >= 3)
240 {
241 g_UrbStatus = USBH_URB_IDLE;
242 USBH_BOT_ConfigBOTState(USBH_BOT_ERROR_OUT);
243 }
244 else
245 {
246 g_BOTInfo.curBOTState = g_BOTInfo.preBOTState;
247 }
248 break;
249 case USBH_URB_PING:
250 USB_OTG_EnableChannelPing(g_BOTInfo.outChNum);
251 break;
252 default:
253 break;
254 }

263static void MSC_BOT_DataInHandler(void)
264{
265 g_UrbStatus = g_usbHost.xferPipe[g_BOTInfo.inChNum].urbStatus;
266
267 if((g_UrbStatus == USBH_URB_OK) || (g_BOTInfo.preBOTState != USBH_BOT_DATAIN))
268 {
269 g_errorCntBOTStall = 0;
270 g_BOTInfo.preBOTState = USBH_BOT_DATAIN;
271
272 if(g_tempDataXferLen > g_BOTInfo.bulkInEpSize)
273 {
274 USBH_BulkInData(g_BOTInfo.inChNum, g_pdata, g_BOTInfo.bulkInEpSize);
275
276 g_tempDataXferLen -= g_BOTInfo.bulkInEpSize;
277 g_pdata = g_pdata + g_BOTInfo.bulkInEpSize;
278 }
279 else if(g_tempDataXferLen == 0)
280 {
281 if (g_MSCInfo.preMSCState == USBH_MSC_READ_CAPACITY10)
282 {
283
284 g_MSCInfo.storageInfo.capacity = g_pdata[0] << 24| \
285 g_pdata[1] << 16| \
286 g_pdata[2] << 8| \
287 g_pdata[3];
288
289 g_MSCInfo.storageInfo.blockSize = g_pdata[4] << 24| \
290 g_pdata[5] << 16| \
291 g_pdata[6] << 8| \
292 g_pdata[7];
293
294 g_MSCInfo.storageInfo.sectorSize = g_MSCInfo.storageInfo.blockSize;
295 }
296 /* If value was 0, and successful transfer, then change the state */
297 g_BOTInfo.curBOTState = USBH_BOT_RECEIVE_CSW;
298 }
299 else
300 {
301 USBH_BulkInData(g_BOTInfo.inChNum, g_pdata, g_tempDataXferLen);
302
303 /* Reset this value and keep in same state */
304 g_tempDataXferLen = 0;
305 }
306 }
307 else if(g_UrbStatus == USBH_URB_STALL)
308 {
309 g_MSCInfo.preMSCState = g_MSCInfo.curMSCState;
310 g_BOTInfo.preBOTState = g_BOTInfo.curBOTState;
311
312 g_BOTInfo.curBOTState = USBH_BOT_ERROR_IN;
313 }
314 else if(g_UrbStatus == USBH_URB_ERROR)
315 {
316 g_errorTimeout++;
317
318 if(g_errorTimeout >= 3)
319 {
320 g_MSCInfo.preMSCState = g_MSCInfo.curMSCState;
321 g_BOTInfo.curBOTState = USBH_BOT_ERROR_IN;
322 }
323 else
324 {
325 g_BOTInfo.curBOTState = g_BOTInfo.preBOTState;
326 }

495static void MSC_BOT_ErrorInHandler(void)
496{
497 USBH_CTRL_STATE_T xferCtrlState;
498
499 /* Config Request */
500 if(g_usbHost.control.xferState == USBH_XFER_START)
501 {
502 USBH_StandardClearFeature(USBH_RECIPIENT_ENDPOINT,
503 USBH_FEATURE_ENDPOINT_HALT,
504 g_BOTInfo.bulkInEpAddr,
505 g_BOTInfo.inChNum);
506
507 g_usbHost.control.xferState = USBH_XFER_WAITING;
508 }
509 /* Do the transaction */
510 else if(g_usbHost.control.xferState == USBH_XFER_WAITING)
511 {
512 /* Polling transaction handler */
513 xferCtrlState = USBH_ControlXferHandler(&g_usbHost.control.reqData, NULL);
514
515 switch (xferCtrlState)
516 {
517 case USBH_CTRL_COMPLETE:
518 g_usbHost.control.xferState = USBH_XFER_OK;
519 break;
520 case USBH_CTRL_STALL :
521 case USBH_CTRL_ERROR :
522 g_usbHost.control.xferState = USBH_XFER_START;
523 break;
524 default :
525 break;
526 }
527
528 }
529 /* If transfer OK */
530 else if (g_usbHost.control.xferState == USBH_XFER_OK)
531 {
532 g_usbHost.control.xferState = USBH_XFER_START;
533
534 g_MSCInfo.curMSCState = USBH_MSC_TEST_UNIT_READY;
535 g_MSCInfo.SCSIStates = MSC_SCSI_SEND;
536 }
