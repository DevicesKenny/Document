87static void USBD_HID_GetReportDescriptor(void)
88{
89 USBD_DevReqData_T *reqData = &g_usbDev.reqData;
90 uint8_t len;
91
92 if((reqData->domain.bRequest == USBD_GET_DESCRIPTOR) &&
93 (reqData->domain.bmRequestType.bit.recipient == USBD_RECIPIENT_INTERFACE) &&
94 (reqData->domain.bmRequestType.bit.type == USBD_REQ_TYPE_STANDARD))
95 {
96 if(reqData->domain.wValue[1] == 0x21)
97 {
98 len = USB_MIN(reqData->domain.wLength[0], 9);
99 USBD_CtrlInData((uint8_t *)&g_configDescriptor.pDesc[0x12], len);
100 }
101 else if(reqData->domain.wValue[1] == 0x22)
102 {
103 len = USB_MIN(reqData->domain.wLength[0], sizeof(g_hidMouseReportDescriptor));
104 USBD_CtrlInData((uint8_t *)g_hidMouseReportDescriptor, len);
105 }
106 }
107 else
108 {
109 USB_OTG_SetStall(USB_EP_0);
110 USB_OTG_ReceiveSetupPacket(3);
111 }
112 if (reqData->domain.wLength[0] == 0)
113 {
114 USBD_CtrlTxStatus();
115 }
116}

140void USBD_HID_Write(uint8_t key)
141{
142 int8_t x = 0;
143 int8_t y = 0;
144 uint8_t g_buffer[4] = {0, 0, 0, 0};
145
146 switch (key)
147 {
148 case HID_MOUSE_KEY_LEFT:
149 x -= 10;
150 break;
151
152 case HID_MOUSE_KEY_RIGHT:
153 x += 10;
154 break;
155
156 case HID_MOUSE_KEY_UP:
157 y -= 10;
158 break;
159
160 case HID_MOUSE_KEY_DOWN:
161 y += 10;
162 break;
163
164 default:
165 return;
166 }
167
168 g_buffer[1] = x;
169 g_buffer[2] = y;
170
171 s_statusEP = 0;
172
173 USBD_TxData(USB_EP_1, g_buffer, 4);
174}
182uint8_t USBD_HID_ReadKey(void)

195
196 return HID_MOUSE_KEY_NULL;

206void USBD_HID_Proc(void)
207{
208 uint8_t key;
209
210 key = HID_MOUSE_KEY_NULL;
211
212 /* Check the usb device configured state */
213 if(g_usbDev.devState != USBD_DEVICE_STATE_CONFIGURED)
214 {
215 return;
216 }
217
218 key = USBD_HID_ReadKey();
219
220 if(key != HID_MOUSE_KEY_NULL)
221 {
222 if(s_statusEP)
223 {
224 USBD_HID_Write(key);
225 }
226 }
227}
