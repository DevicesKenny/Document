129static void
130low_level_init(struct netif *netif)
131{
132 int i;
133 /* set MAC hardware address length */
134 netif->hwaddr_len = ETHARP_HWADDR_LEN;
135
136 /* set MAC hardware address */
137 netif->hwaddr[0] = SetMACaddr[0];
138 netif->hwaddr[1] = SetMACaddr[1];
139 netif->hwaddr[2] = SetMACaddr[2];
140 netif->hwaddr[3] = SetMACaddr[3];
141 netif->hwaddr[4] = SetMACaddr[4];
142 netif->hwaddr[5] = SetMACaddr[5];
143
144 /* Config MAC Address */
145 ETH_ConfigMACAddress(ETH_MAC_ADDRESS0, SetMACaddr);
146
147 /* maximum transfer unit */
148 netif->mtu = 1500;
149
150 /* device capabilities */
151 /* don't set NETIF_FLAG_ETHARP if this device is not an ethernet one */
152 netif->flags = NETIF_FLAG_BROADCAST | NETIF_FLAG_ETHARP | NETIF_FLAG_LINK_UP;
153
154 /* Initialize Tx Descriptors list: Chain Mode */
155 ETH_ConfigDMATxDescChain(DMATxDscrTab, &Tx_Buff[0][0], ETH_TXBUFNB);
156 /* Initialize Rx Descriptors list: Chain Mode */
157 ETH_ConfigDMARxDescChain(DMARxDscrTab, &Rx_Buff[0][0], ETH_RXBUFNB);
158
159 /* Enable Ethernet Rx interrrupt */
160 for(i=0; i<ETH_RXBUFNB; i++)
161 {
162 ETH_EnableDMARxDescReceiveInterrupt(&DMARxDscrTab[i]);
163 }
164
165#ifdef HARDWARE_CHECKSUM
166 /* Enable the checksum insertion for the Tx frames */
167 for(i=0; i<ETH_TXBUFNB; i++)
168 {
169 ETH_ConfigDMATxDescChecksumInsertion(&DMATxDscrTab[i], ETH_DMATXDESC_CHECKSUMTCPUDPICMPFULL);
170 }
171#endif
172
173 eth_rx_sem = xSemaphoreCreateCounting(300, 0);
174
175 /* create the task that handles the ETH_MAC */
176 sys_thread_new("eth_rx",
177 ethernetif_input,
178 netif,
179 1024,
180 TCPIP_THREAD_PRIO);
181
182 /* Enable MAC and DMA transmission and reception */
183 ETH_Start();
184}

201static err_t
202low_level_output(struct netif *netif, struct pbuf *p)
203{
204 struct pbuf *q;
205 int l = 0;
206 u8 *buffer = (u8 *)(DMATxDescToSet->Buffer1Addr);
207
208 for(q = p; q != NULL; q = q->next)
209 {
210 memcpy((u8_t*)&buffer[l], q->payload, q->len);
211 l = l + q->len;
212 }
213
214 ETH_TxPacket(l);
215
216 return ERR_OK;
217}

227static struct pbuf *
228low_level_input(struct netif *netif)
229{
230 struct pbuf *p, *q;
231 u16_t len;
232 int l =0;
233 ETH_Frame_T frame;
234 u8 *buffer;
235 __IO ETH_DMADescConfig_T *DMARxNextDesc;
236 int i = 0;
237
238 p = NULL;
239
240 frame = ETH_ReadReceivedFrameInterrupt();
241 if(frame.length == 0)
242 {
243 return NULL;
244 }
245
246 /* Obtain the size of the packet and put it into the "len"
247 variable. */
248 len = frame.length;
249 buffer = (u8 *)frame.buffer;
250
251 /* check that frame has no error */
252 if ((frame.descriptor->Status & ETH_DMARXDESC_ERRS) == (uint32_t)RESET)
253 {
254 /* We allocate a pbuf chain of pbufs from the pool. */
255 p = pbuf_alloc(PBUF_RAW, len, PBUF_POOL);
256
257 if (p != NULL)
258 {
259 for (q = p; q != NULL; q = q->next)
260 {
261 memcpy((u8_t*)q->payload, (u8_t*)&buffer[l], q->len);
262 l = l + q->len;
263 }
264 }
265 }
266
267 /* Release descriptors to DMA */
268 /* Check if frame with multiple DMA buffer segments */
269 if (DMARxFraminfos->segCount > 1)
270 {
271 DMARxNextDesc = DMARxFraminfos->FS_RxDesc;
272 }
273 else
274 {
275 DMARxNextDesc = frame.descriptor;
276 }
277
278 /* Set Own bit in Rx descriptors: gives the buffers back to DMA */
279 for (i=0; i<DMARxFraminfos->segCount; i++)
280 {
281 DMARxNextDesc->Status = ETH_DMARXDESC_OWN;
282 DMARxNextDesc = (ETH_DMADescConfig_T *)(DMARxNextDesc->Buffer2NextDescAddr);
283 }
284
285 /* Clear Segment_Count */
286 DMARxFraminfos->segCount =0;
287
288 /* When Rx Buffer unavailable flag is set: clear it and resume reception */
289 if ((ETH->DMASTS & BIT7) != (u32)RESET)
290 {
291 /* Clear RBUS ETHERNET DMA flag */
292 ETH->DMASTS = BIT7;
293 /* Resume DMA reception */
294 ETH->DMARXPD = 0;
295 }
296
297 return p;
298}

309void ethernetif_input(void *pParams)
310{
311 struct pbuf *p = NULL;
312 struct netif *netif = (struct netif*) pParams;
313
314 LWIP_DEBUGF(NETIF_DEBUG, ("ethernetif_input: IP input error\n"));
315
316 while(1)
317 {
318 if(xSemaphoreTake(eth_rx_sem, portMAX_DELAY) == pdTRUE)
319 {
320 while(1)
321 {
322 /* move received packet into a new pbuf */
323 p = low_level_input(netif);
324 if(p == NULL)
325 {
326 break;
327 }
328
329 /* full packet send to tcpip_thread to process */
330 if (netif->input(p, netif) != ERR_OK)
331 {
332 LWIP_DEBUGF(NETIF_DEBUG, ("ethernetif_input: IP input error\n"));
333 pbuf_free(p);
334 p = NULL;
335 }
336 }
337 }
338 }

353err_t
354ethernetif_init(struct netif *netif)
355{
356 struct ethernetif *ethernetif;
357
358 LWIP_ASSERT("netif != NULL", (netif != NULL));
359
360 ethernetif = mem_malloc(sizeof(struct ethernetif));
361 if (ethernetif == NULL)
362 {
363 LWIP_DEBUGF(NETIF_DEBUG, ("ethernetif_init: out of memory\n"));
364 return ERR_MEM;
365 }
366
367#if LWIP_NETIF_HOSTNAME
368 /* Initialize interface hostname */
369 netif->hostname = "lwip";
370#endif /* LWIP_NETIF_HOSTNAME */
371
372 /*
373 * Initialize the snmp variables and counters inside the struct netif.
374 * The last argument should be replaced with your link speed, in units
375 * of bits per second.
376 */
377 NETIF_INIT_SNMP(netif, snmp_ifType_ethernet_csmacd, 100000000);
378
379 netif->state = ethernetif;
380 netif->name[0] = IFNAME0;
381 netif->name[1] = IFNAME1;
382 /* We directly use etharp_output() here to save a function call.
383 * You can instead declare your own function an call etharp_output()
384 * from it if you have to do some checks before sending (e.g. if link
385 * is available...) */
386 netif->output = etharp_output;
387 netif->linkoutput = low_level_output;
388
389 ethernetif->ethaddr = (struct eth_addr *)&(netif->hwaddr[0]);
390
391 /* initialize the hardware */
392 low_level_init(netif);
393
394 return ERR_OK;
395}

406ETH_Frame_T ETH_RxPacket(void)
407{
408 u32 ethframelength = 0;
409 ETH_Frame_T ethframe = {0,0};
410
411 /* Check if the descriptor is owned by the ETHERNET DMA (when set) or CPU (when reset) */
412 if((DMARxDescToGet->Status & ETH_DMARXDESC_OWN) != (u32)RESET)
413 {
414 ethframe.length = ETH_ERROR;
415
416 if ((ETH->DMASTS & BIT7) != (u32)RESET)
417 {
418 /* Clear RBUS ETHERNET DMA flag */
419 ETH->DMASTS = BIT7;
420 /* Resume DMA reception */
421 ETH->DMARXPD = 0;
422 }
423
424 /* Return error: OWN bit set */
425 return ethframe;
426 }
427
428 if(((DMARxDescToGet->Status & ETH_DMARXDESC_ERRS) == (u32)RESET) &&
429 ((DMARxDescToGet->Status & ETH_DMARXDESC_LDES) != (u32)RESET) &&
430 ((DMARxDescToGet->Status & ETH_DMARXDESC_FDES) != (u32)RESET))
431 {
432 /* Get the Frame Length of the received packet: substruct 4 bytes of the CRC */
433 ethframelength = ((DMARxDescToGet->Status & ETH_DMARXDESC_FL) >> 16) - 4;
434
435 /* Get the addrees of the actual buffer */
436 ethframe.buffer = DMARxDescToGet->Buffer1Addr;
437 }
438 else
439 {
440 /* Return ERROR */
441 ethframelength = ETH_ERROR;
442 }
443
444 ethframe.length = ethframelength;
445 ethframe.descriptor = DMARxDescToGet;
446
447 /* Selects the next DMA Rx descriptor list for next buffer to read */
448 DMARxDescToGet = (ETH_DMADescConfig_T*) (DMARxDescToGet->Buffer2NextDescAddr);
449
450 /* Return Frame */
451 return ethframe;
452}

464int ETH_TxPacket(u16 Length)
465{
467 if((DMATxDescToSet->Status & ETH_DMATXDESC_OWN) != (u32)RESET)
468 {
470 return ETH_ERROR;
471 }
472
474 DMATxDescToSet->ControlBufferSize = (Length & ETH_DMATXDESC_TXBS1);
475
477 DMATxDescToSet->Status |= ETH_DMATXDESC_LS | ETH_DMATXDESC_FS;
478
480 DMATxDescToSet->Status |= ETH_DMATXDESC_OWN;
481
483 if ((ETH->DMASTS & BIT2) != (u32)RESET)
484 {
486 ETH->DMASTS = BIT2;
488 ETH->DMATXPD = 0;
489 }
490
494 DMATxDescToSet = (ETH_DMADescConfig_T*) (DMATxDescToSet->Buffer2NextDescAddr);
495
497 return ETH_SUCCESS;
498}
