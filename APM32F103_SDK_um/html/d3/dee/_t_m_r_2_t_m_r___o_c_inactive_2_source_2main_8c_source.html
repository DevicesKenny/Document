96 OCcongigStruct.pulse = 200;
97 TMR_ConfigOC4(TMR2, &OCcongigStruct);
98 TMR_ConfigOC4Preload(TMR2, TMR_OC_PRELOAD_DISABLE);
99
100 TMR_DisableAutoReload(TMR2);
101
102 /* generate raising edge */
103 GPIO_WriteBitValue(GPIOA, GPIO_PIN_0, 1);
104 GPIO_WriteBitValue(GPIOA, GPIO_PIN_1, 1);
105 GPIO_WriteBitValue(GPIOA, GPIO_PIN_2, 1);

107
108 APM_MINI_LEDOn(LED2);
109
110 NVIC_EnableIRQRequest(TMR2_IRQn, 0, 0);
111
112 /* start generate four different signals */
113 TMR_Enable(TMR2);
114
115 TMR_EnableInterrupt(TMR2, TMR_INT_CC1 | TMR_INT_CC2 | TMR_INT_CC3 | TMR_INT_CC4);
116 while (1)
117 {
118 }
119}
127void TMR2_Isr(void)
128{
129 if (TMR_ReadIntFlag(TMR2, TMR_INT_CC1))
130 {
131 /* Clear the TMR2 CC1 pending bit */
132 TMR_ClearIntFlag(TMR2, TMR_INT_CC1);
133
134 /* delay 800us generate falling edge */
135 GPIO_WriteBitValue(GPIOA, GPIO_PIN_0, 0);
136 }
137
138 if (TMR_ReadIntFlag(TMR2, TMR_INT_CC2))
139 {
140 /* Clear the TMR2 CC2 pending bit */
141 TMR_ClearIntFlag(TMR2, TMR_INT_CC2);
142
143 /* delay 600us generate falling edge */
144 GPIO_WriteBitValue(GPIOA, GPIO_PIN_1, 0);
145 }
146
147 if (TMR_ReadIntFlag(TMR2, TMR_INT_CC3))
148 {
149 /* Clear the TMR2 CC3 pending bit */
150 TMR_ClearIntFlag(TMR2, TMR_INT_CC3);
151
152 /* delay 400us generate falling edge */
153 GPIO_WriteBitValue(GPIOA, GPIO_PIN_2, 0);
154 }
155
156 if (TMR_ReadIntFlag(TMR2, TMR_INT_CC4))
157 {

159 TMR_ClearIntFlag(TMR2, TMR_INT_CC4);
160
161 /* delay 200us generate falling edge */
162 GPIO_WriteBitValue(GPIOA, GPIO_PIN_3, 0);
163 }
