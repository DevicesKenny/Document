125void USBD_CtrlInProcess(void)
126{
127 uint32_t len;
128 uint16_t maxPackSize = g_usbDev.inBuf[USB_EP_0].maxPackSize;
129
130 if (g_usbDev.ctrlState == USBD_CTRL_STATE_IN_DATA)

194void USBD_CtrlInData(uint8_t* buf, uint32_t len)
195{
196 uint16_t maxPackSize = g_usbDev.inBuf[USB_EP_0].maxPackSize;
197 uint16_t reqLen = ((uint16_t)g_usbDev.reqData.domain.wLength[1] << 8) | \
198 (g_usbDev.reqData.domain.wLength[0]);
199
200 /* Init Buffer to Send */
201 g_usbDev.inBuf[USB_EP_0].pBuf = buf;
202 g_usbDev.inBuf[USB_EP_0].bufLen = len;
203 g_usbDev.inBuf[USB_EP_0].packNum = (len + (maxPackSize - 1)) / maxPackSize;
204
205 if ((len < reqLen) && ((len % maxPackSize) == 0))
206 {
207 g_usbDev.inBuf[USB_EP_0].zeroPackFill = 1;
208 }
209
210 if (len)
211 {
212 len = USB_MIN(g_usbDev.inBuf[USB_EP_0].bufLen, maxPackSize);
213 /* Enable IN Transfer */
214 USB_OTG_EnableInEpTransfer(USB_EP_0, 1, len);
215
216 /* Enable TxFIFO empty interrupt */
217 USB_OTG_EnableInEpEmptyInterrupt(USB_EP_0);
218
219 g_usbDev.ctrlState = USBD_CTRL_STATE_IN_DATA;
220 }
221 /* Zero packet */
222 else
223 {
224 USBD_CtrlTxStatus();
225 }

323void USBD_PushDataToTxFIFO(uint8_t ep)
324{
325 uint32_t txFifoSpace;
326 uint32_t len;
327 uint16_t maxPackSize = g_usbDev.inBuf[ep].maxPackSize;
328
329 if ((!g_usbDev.inBuf[ep].bufLen) || (g_usbDev.inBuf[ep].pBuf == NULL))
330 {
331 return;
332 }
333
334 len = USB_MIN(g_usbDev.inBuf[ep].bufLen, maxPackSize);
335
336 /* Check the Available space of TxFifo */
337 txFifoSpace = 4 * USB_OTG_ReadTxFifoAvailable(ep);
338
339 /* Push data to TxFIFO */
340 if (txFifoSpace >= len)
341 {
342 USB_FIFO_WriteFifoPacket(ep, g_usbDev.inBuf[ep].pBuf, len);
343
344 g_usbDev.inBuf[ep].pBuf += len;
345 g_usbDev.inBuf[ep].bufLen -= len;
346 g_usbDev.inBuf[ep].packNum--;
347
348 USB_OTG_DisableInEpEmptyInterrupt(ep);
349 }
