89int main(void)
90{
91 USART_Config_T USART_ConfigStruct;
92
93 /* USART config */
94 USART_ConfigStruct.baudRate = 115200;
95 USART_ConfigStruct.hardwareFlow = USART_HARDWARE_FLOW_NONE;
96 USART_ConfigStruct.mode = USART_MODE_TX_RX;
97 USART_ConfigStruct.parity = USART_PARITY_NONE;
98 USART_ConfigStruct.stopBits = USART_STOP_BIT_1;
99 USART_ConfigStruct.wordLength = USART_WORD_LEN_8B;
100
101 APM_MINI_COMInit(COM1, &USART_ConfigStruct);
102 APM_MINI_COMInit(COM2, &USART_ConfigStruct);
103
104 APM_MINI_LEDInit(LED2);
105 APM_MINI_LEDInit(LED3);
106 APM_MINI_LEDOff(LED2);
107 APM_MINI_LEDOff(LED3);
108
109 /* USART1 RX and USART2 TX DMA Init */
110 USART1_RX_DMA_Init();
111 USART2_TX_DMA_Init();
112
113 /* Init TxBuf */
114 for (int i = 0; i < BufSize; i++)
115 {
116 DMA_USART2_TxBuf[i] = i;
117 }
118
119 /* Enable USART1 RX and USART2 TX DMA */
120 USART_EnableDMA(USART1, USART_DMA_RX);
121 USART_EnableDMA(USART2, USART_DMA_TX);
122
123 /* Wait until USART2 TX DMA1 Channel Transfer Complete */
124 while (DMA_ReadStatusFlag(DMA1_FLAG_TC7) == RESET);
125
126 /* Wait until USART1 RX DMA1 Channel Transfer Complete */
127 while (DMA_ReadStatusFlag(DMA1_FLAG_TC5) == RESET);
128
129 /* Data is ok then turn off LED2 */
130 if (BufferCompare(DMA_USART2_TxBuf, DMA_USART1_RxBuf, BufSize) == TRUE)
131 {
132 APM_MINI_LEDOn(LED2);
133 }
134 else
135 {
136 APM_MINI_LEDOn(LED3);
137 }
138
139 while(1)
140 {
141 }
142}

151void USART1_RX_DMA_Init(void)
152{
153 DMA_Config_T dmaConfig;
154
155 /* Enable DMA Clock */
156 RCM_EnableAHBPeriphClock(RCM_AHB_PERIPH_DMA1);
157
158 /* DMA config */
159 dmaConfig.peripheralBaseAddr = USART1_DR_ADDRESS;
160 dmaConfig.memoryBaseAddr = (uint32_t)DMA_USART1_RxBuf;
161 dmaConfig.dir = DMA_DIR_PERIPHERAL_SRC;
162 dmaConfig.bufferSize = BufSize;
163 dmaConfig.peripheralInc = DMA_PERIPHERAL_INC_DISABLE;
164 dmaConfig.memoryInc = DMA_MEMORY_INC_ENABLE;
165 dmaConfig.peripheralDataSize = DMA_PERIPHERAL_DATA_SIZE_BYTE;
166 dmaConfig.memoryDataSize = DMA_MEMORY_DATA_SIZE_BYTE;
167 dmaConfig.loopMode = DMA_MODE_NORMAL;
168 dmaConfig.priority = DMA_PRIORITY_MEDIUM;
169 dmaConfig.M2M = DMA_M2MEN_DISABLE;
170
171 /* Enable DMA channel */
172 DMA_Config(DMA1_Channel5, &dmaConfig);
173
174 /* Enable DMA */
175 DMA_Enable(DMA1_Channel5);
176}
