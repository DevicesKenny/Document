143static void USBD_MSC_BOT_DecodeCBW(void)
144{
145 uint32_t xferCnt = g_usbDev.outBuf[MSC_OUT_EP & 0x7f].xferCnt;
146
147 g_BOTInfo.CSW.dTag = g_BOTInfo.CBW.dTag;
148 g_BOTInfo.CSW.dDataResidue = g_BOTInfo.CBW.dDataXferLen;
149
150 if ((xferCnt != MSC_BOT_CBW_LENGTH) || \
151 (g_BOTInfo.CBW.dSignature != MSC_BOT_CBW_SIGNATURE) || \
152 (g_BOTInfo.CBW.bLUN > 1) || (g_BOTInfo.CBW.bCBLen < 1) || \
153 (g_BOTInfo.CBW.bCBLen > 16))
154 {
155 SCSI_PutSenseCode(g_BOTInfo.CBW.bLUN, SCSI_SKEY_ILLEGAL_REQUEST,
156 SCSI_ASC_INVALID_CDB, 0);
157
158 g_BOTInfo.status = BOT_STATUS_ERROR;
159 }
160 else
161 {
162 if (SCSI_CmdHandler(g_BOTInfo.CBW.bLUN, g_BOTInfo.CBW.CB) != SCSI_OK)
163 {
164 USBD_MSC_BOT_Stall();
165 }
166 else if ((g_BOTInfo.state == BOT_STATE_IDLE) || \
167 (g_BOTInfo.state == BOT_STATE_SEND_DATA))
168 {
169 if (g_BOTInfo.dataLen)
170 {
171 USBD_MSC_BOT_TxData(g_BOTInfo.data, g_BOTInfo.dataLen);
172 }
173 else
174 {
175 USBD_MSC_BOT_TxCSW(BOT_CSW_STATUS_CMD_OK);
176 }
177 }
178 }

190static void USBD_MSC_BOT_TxData(uint8_t *txBuf, uint16_t len)
191{
192 len = USB_MIN(len, g_BOTInfo.CBW.dDataXferLen);
193
194 g_BOTInfo.CSW.dDataResidue -= len;
195 g_BOTInfo.CSW.bStatus = BOT_CSW_STATUS_CMD_OK;
196 g_BOTInfo.state = BOT_STATE_SEND_DATA;
197
198 USBD_TxData(MSC_IN_EP & 0x7f, txBuf, len);
199}

208void USBD_MSC_BOT_TxCSW(uint8_t cswStatus)
209{
210 g_BOTInfo.CSW.dSignature = MSC_BOT_CSW_SIGNATURE;
211 g_BOTInfo.CSW.bStatus = cswStatus;
212 g_BOTInfo.state = BOT_STATE_IDLE;
213

215 MSC_BOT_CSW_LENGTH);

218 MSC_BOT_CBW_LENGTH);

228void USBD_MSV_BOT_ClearFeatureHandler(void)
229{

231 {
232 USB_OTG_ClearInEpStall(MSC_IN_EP & 0x7f);
233 g_BOTInfo.status = BOT_STATUS_NORMAL;

238 USBD_MSC_BOT_TxCSW(BOT_CSW_STATUS_CMD_FAIL);

249static void USBD_MSC_BOT_Stall(void)
250{
251 if ((g_BOTInfo.CBW.bmFlags == 0) && (g_BOTInfo.CBW.dDataXferLen != 0) && \
252 (g_BOTInfo.status == BOT_STATUS_NORMAL))
253 {
254 USB_OTG_SetOutEpStall(MSC_OUT_EP & 0x7f);
255 }
256
257 USB_OTG_SetInEpStall(MSC_IN_EP & 0x7f);
258
259 if (g_BOTInfo.status == BOT_STATUS_ERROR)
260 {
261 USBD_RxData(MSC_OUT_EP & 0x7f, (uint8_t *)&g_BOTInfo.CBW,
262 MSC_BOT_CBW_LENGTH);
263 }
