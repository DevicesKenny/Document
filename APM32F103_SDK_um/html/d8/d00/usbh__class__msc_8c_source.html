105void USBH_MSC_ClassInitHandler(void)
106{
107 uint8_t itfNum, epNum;
108
109 itfNum = USBH_ReadConfigurationItfNum(&g_usbHost);
110
111 while (itfNum--)
112 {
113 if ((USBH_ReadInterfaceClass(&g_usbHost, itfNum) != USBH_CLASS_MSC) || \
114 (USBH_ReadInterfaceProtocol(&g_usbHost, itfNum) != USBH_MSC_PROTOCOL_BBB))
115 {
116 g_userCallback.deviceNotSupportedHandler();
117 continue;
118 }
119
120 epNum = USBH_ReadInterfaceEpNum(&g_usbHost, itfNum);
121 while (epNum--)
122 {
123 /* Get endpoint and size */
124 if (USBH_ReadEndpointAddress(&g_usbHost, itfNum, epNum) & 0x80)
125 {
126 g_BOTInfo.bulkInEpAddr = USBH_ReadEndpointAddress(&g_usbHost, itfNum, epNum);
127 g_BOTInfo.bulkInEpSize = USBH_ReadEndpointMPS(&g_usbHost, itfNum, epNum);
128 }
129 else
130 {
131 g_BOTInfo.bulkOutEpAddr = USBH_ReadEndpointAddress(&g_usbHost, itfNum, epNum);
132 g_BOTInfo.bulkOutEpSize = USBH_ReadEndpointMPS(&g_usbHost, itfNum, epNum);
133 }
134 }
135 }
136
137 /* Out channels */
138 g_BOTInfo.outChNum = USBH_CH_AllocChannel(g_BOTInfo.bulkOutEpAddr);
139 /* In channels */
140 g_BOTInfo.inChNum = USBH_CH_AllocChannel(g_BOTInfo.bulkInEpAddr);
141
142 /* Open the new channels */
143 USBH_CH_OpenChannel(g_BOTInfo.outChNum,
144 g_usbHost.devAddress,
145 USB_EP_TYPE_BULK,
146 g_BOTInfo.bulkOutEpSize);
147
148 /* Open the new channels */
149 USBH_CH_OpenChannel(g_BOTInfo.inChNum,
150 g_usbHost.devAddress,
151 USB_EP_TYPE_BULK,
152 g_BOTInfo.bulkInEpSize);
153
154 g_MSCInfo.storageInfo.maxLun = 0;
155 g_MSCInfo.storageInfo.sectorSize = 512;
156 g_MSCInfo.storageInfo.blockSize = 512;
157 g_MSCInfo.stallCnt = 0;
158
159 /* Specify the next state of the USB host state machine */
160 USBH_ConfigHostState(USBH_HOST_CLASS);
161 USBH_ConfigClassCoreState(USBH_MSC_RESET_BOT);
162}

234static void MSC_GetMaxLunHandler(void)
235{
236 USBH_CTRL_STATE_T xferCtrlState;
237
238 /* Config Request */
239 if (g_usbHost.control.xferState == USBH_XFER_START)
240 {
241 g_usbHost.control.reqData.domain.bmRequestType.bit.dir = USBH_REQ_DIR_IN;
242 g_usbHost.control.reqData.domain.bmRequestType.bit.recipient = USBH_RECIPIENT_INTERFACE;
243 g_usbHost.control.reqData.domain.bmRequestType.bit.type = USBH_REQ_TYPE_CLASS;
244
245 g_usbHost.control.reqData.domain.bRequest = USBH_CLASS_GET_MAX_LUN;
246
247 g_usbHost.control.reqData.domain.wValue[0] = 0x00;
248 g_usbHost.control.reqData.domain.wValue[1] = 0x00;
249
250 g_usbHost.control.reqData.domain.wIndex[0] = 0x00;
251 g_usbHost.control.reqData.domain.wIndex[1] = 0x00;
252
253 g_usbHost.control.reqData.domain.wLength[0] = 0x01;
254 g_usbHost.control.reqData.domain.wLength[1] = 0x00;
255
256 g_usbHost.control.ctrlState = USBH_CTRL_SETUP;
257
258 g_usbHost.control.xferState = USBH_XFER_WAITING;
259
260 }
261 /* Do the transaction */
262 else if (g_usbHost.control.xferState == USBH_XFER_WAITING)
263 {
264 /* Polling transaction handler */
265 xferCtrlState = USBH_ControlXferHandler(&g_usbHost.control.reqData,
266 (uint8_t*)&g_MSCInfo.storageInfo.maxLun);
267
268 switch (xferCtrlState)
269 {
270 case USBH_CTRL_COMPLETE:
271 g_MSCInfo.stallCnt = 0;
272 g_usbHost.control.xferState = USBH_XFER_START;
273 /* Get device max LUN */
274 if (g_MSCInfo.storageInfo.maxLun == 0)
275 {
276 /* Go to next state */
277 USBH_ConfigClassCoreState(USBH_MSC_TEST_UNIT_READY);
278 }
279 else
280 {
281 g_usbHost.errorState = USBH_ERROR_NOT_SUPPORT;
282 }
283 break;
284 case USBH_CTRL_STALL :
285 if (g_MSCInfo.stallCnt++ < USBH_MSC_SATLL_RETRY_CNT)
286 {
287 g_MSCInfo.preClassReqState = USBH_MSC_REQ_GET_MAX_LUN;
288 g_MSCInfo.curClassReqState = USBH_MSC_REQ_CTRL_ERROR;
289 }
290 else
291 {
292 g_MSCInfo.stallCnt = 0;
293 USBH_ConfigClassCoreState(USBH_MSC_TEST_UNIT_READY);
294 }
295
296 g_usbHost.control.xferState = USBH_XFER_START;
297 g_MSCInfo.storageInfo.maxLun = 0;
298 case USBH_CTRL_ERROR :
299 g_usbHost.control.xferState = USBH_XFER_START;
300 break;
301 default :
302 break;
303 }
304 }

315static void MSC_BotResetHandler(void)
316{

318
319 /* Config Request */
320 if (g_usbHost.control.xferState == USBH_XFER_START)

366 g_usbHost.control.xferState = USBH_XFER_START;
367 break;
368 default :
369 break;
370 }
371 }
372 /* If transfer OK */

374 {
375 g_usbHost.control.xferState = USBH_XFER_START;

377 g_MSCInfo.curClassReqState = USBH_MSC_REQ_GET_MAX_LUN;

390static void MSC_CtrlErrorStateHandler(void)
391{
392 USBH_CTRL_STATE_T xferCtrlState;
393
394 /* Config Request */
395 if (g_usbHost.control.xferState == USBH_XFER_START)
396 {
397 USBH_StandardClearFeature(USBH_RECIPIENT_ENDPOINT,
398 USBH_FEATURE_ENDPOINT_HALT,
399 0,
400 g_usbHost.control.outChannelNum);
401
402 g_usbHost.control.xferState = USBH_XFER_WAITING;
403 }
404 /* Do the transaction */
405 else if (g_usbHost.control.xferState == USBH_XFER_WAITING)
406 {
407 /* Polling transaction handler */
408 xferCtrlState = USBH_ControlXferHandler(&g_usbHost.control.reqData, NULL);
409
410 switch (xferCtrlState)
411 {
412 case USBH_CTRL_COMPLETE:
413 g_usbHost.control.xferState = USBH_XFER_OK;
414 break;
415 case USBH_CTRL_STALL :
416 case USBH_CTRL_ERROR :
417 g_usbHost.control.xferState = USBH_XFER_START;
418 break;
419 default :
420 break;
421 }
422
423 }
424 /* If transfer OK */
425 else if (g_usbHost.control.xferState == USBH_XFER_OK)
426 {
427 g_usbHost.control.xferState = USBH_XFER_START;
428
429 g_MSCInfo.storageInfo.maxLun = 0;
430
431 /* Go to backup state */
432 g_MSCInfo.curClassReqState = g_MSCInfo.preClassReqState;
433 }
