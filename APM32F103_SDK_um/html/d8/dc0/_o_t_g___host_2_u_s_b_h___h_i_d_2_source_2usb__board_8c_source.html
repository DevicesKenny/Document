286void ConfigDelayTimer(void)
287{
288#if ((DELAY_SOURCE == USE_TIMER))
289 TMR_BaseConfig_T TMR_BaseConfigStruct;
290 uint32_t pClk2 = 0;
291
292 RCM_ReadPCLKFreq(&tmr2Clk, &pClk2);
293
294 /* Enable TMR2 Periph Clock */
295 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_TMR2);
296
297 TMR_BaseConfigStruct.division = 3;
298 TMR_BaseConfigStruct.clockDivision = TMR_CLOCK_DIV_1;
299 TMR_BaseConfigStruct.countMode = TMR_COUNTER_MODE_UP;
300
301 /* T = (1 + period)*(1/((tmr2Clk*2)/(1+division))) */
302 TMR_BaseConfigStruct.period = (tmr2Clk*2)/4/1000 - 1;
303
304 /* Config TMR2 */
305 TMR_ConfigTimeBase(TMR2, &TMR_BaseConfigStruct);
306 TMR_EnableAutoReload(TMR2);
307
308 /* Enable TMR2 Interrupt */
309 NVIC_EnableIRQRequest(TMR2_IRQn, 0, 0);
310 TMR_EnableInterrupt(TMR2, TMR_INT_UPDATE);
311 TMR_Enable(TMR2);
312#endif
313}

329static void TMR2_Delay(uint32_t delayTime, uint8_t timeUnit)
330{
331 TMR_BaseConfig_T TMR_BaseConfigStruct;
332
333 TMR_Disable(TMR2);
334 TMR_DisableInterrupt(TMR2, TMR_INT_UPDATE);
335 TMR_ClearStatusFlag(TMR2, TMR_INT_UPDATE);
336
337 /* set delay value */
338 TMR_delayCnt = delayTime;
339
340 TMR_BaseConfigStruct.division = 3;
341 TMR_BaseConfigStruct.clockDivision = TMR_CLOCK_DIV_1;
342 TMR_BaseConfigStruct.countMode = TMR_COUNTER_MODE_UP;
343
344 /* T = (1 + period)*(1/((tmr2Clk*2)/(1+division))) */
345 switch (timeUnit)
346 {
347 case 0:
348 TMR_BaseConfigStruct.period = (tmr2Clk*2)/4/1000000 - 1;
349 break;
350 case 1 :
351 TMR_BaseConfigStruct.period = (tmr2Clk*2)/4/1000 - 1;
352 break;
353 default :
354 TMR_BaseConfigStruct.period = (tmr2Clk*2)/4/1000 - 1;
355 break;
356 }
357 TMR_ConfigTimeBase(TMR2, &TMR_BaseConfigStruct);
358
359 /* delay time start*/
360 TMR_EnableInterrupt(TMR2, TMR_INT_UPDATE);
361 TMR_Enable(TMR2);
362 while (TMR_delayCnt > 0);
363
364 /* delay time end*/
365 TMR_Disable(TMR2);
366}
367#endif
