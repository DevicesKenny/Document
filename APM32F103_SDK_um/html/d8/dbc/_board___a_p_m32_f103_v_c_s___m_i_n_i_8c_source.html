168void APM_MINI_PBInit(Button_TypeDef Button, ButtonMode_TypeDef Button_Mode)
169{
170 GPIO_Config_T GPIO_configStruct;
171 EINT_Config_T EINT_configStruct;
172
173 /* Enable the BUTTON Clock */
174 RCM_EnableAPB2PeriphClock(BUTTON_CLK[Button] | RCM_APB2_PERIPH_AFIO);
175
176 /* Configure Button pin as input floating */
177 GPIO_configStruct.mode = GPIO_MODE_IN_PU;
178 GPIO_configStruct.pin = BUTTON_PIN[Button];
179 GPIO_Config(BUTTON_PORT[Button], &GPIO_configStruct);
180
181 if (Button_Mode == BUTTON_MODE_EINT)
182 {
183 /* Connect Button EINT Line to Button GPIO Pin */
184 GPIO_ConfigEINTLine(BUTTON_PORT_SOURCE[Button], BUTTON_PIN_SOURCE[Button]);
185
186 /* Configure Button EINT line */
187 EINT_configStruct.line = BUTTON_EINT_LINE[Button];
188 EINT_configStruct.mode = EINT_MODE_INTERRUPT;
189 EINT_configStruct.trigger = EINT_TRIGGER_FALLING;
190 EINT_configStruct.lineCmd = ENABLE;
191 EINT_Config(&EINT_configStruct);
192
193 /* Enable and set Button EINT Interrupt to the lowest priority */
194 NVIC_EnableIRQRequest(BUTTON_IRQn[Button], 0x0f, 0x0f);
195 }
196}

223void APM_MINI_COMInit(COM_TypeDef COM, USART_Config_T* configStruct)
224{
225 GPIO_Config_T GPIO_configStruct;
226
227 /* Enable GPIO clock */
228 RCM_EnableAPB2PeriphClock(COM_TX_PORT_CLK[COM] | COM_RX_PORT_CLK[COM]);
229
230 if (COM == COM1)
231 {
232 RCM_EnableAPB2PeriphClock(COM_USART_CLK[COM]);
233 }
234 else
235 {
236 RCM_EnableAPB1PeriphClock(COM_USART_CLK[COM]);
237 }
238
239 /* Configure USART Tx as alternate function push-pull */
240 GPIO_configStruct.mode = GPIO_MODE_AF_PP;
241 GPIO_configStruct.pin = COM_TX_PIN[COM];
242 GPIO_configStruct.speed = GPIO_SPEED_50MHz;
243 GPIO_Config(COM_TX_PORT[COM], &GPIO_configStruct);
244
245 /* Configure USART Rx as input floating */
246 GPIO_configStruct.mode = GPIO_MODE_IN_FLOATING;
247 GPIO_configStruct.pin = COM_RX_PIN[COM];
248 GPIO_Config(COM_RX_PORT[COM], &GPIO_configStruct);
249
250 /* USART configuration */
251 USART_Config(COM_USART[COM], configStruct);
252
253 /* Enable USART */
254 USART_Enable(COM_USART[COM]);
255}
