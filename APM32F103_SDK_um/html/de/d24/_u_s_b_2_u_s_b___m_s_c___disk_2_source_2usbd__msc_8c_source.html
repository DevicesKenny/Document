132void MSC_Reset(void)
133{
134 USBD_EPConfig_T epConfig;
135
136 /* Endpoint 1 IN */
137 epConfig.epNum = USBD_EP_1;
138 epConfig.epType = USBD_EP_TYPE_BULK;
139 epConfig.epBufAddr = USB_EP1_TX_ADDR;
140 epConfig.maxPackSize = 64;
141 epConfig.epStatus = USBD_EP_STATUS_NAK;
142 USBD_OpenInEP(&epConfig);
143
144 epConfig.epBufAddr = USB_EP1_RX_ADDR;
145 USBD_OpenOutEP(&epConfig);
146}
