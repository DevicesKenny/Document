239void USB_CoreSoftReset(void)
240{
241 __IO uint32_t timeout = 0;
242
243 /* Wait for AHB master idle */
244 while (USB_OTG_G->GRSTCTRL_B.AHBMIDL != BIT_SET)
245 {
246 if (++timeout > 0x30000)
247 {
248 return;
249 }
250 }
251
252 timeout = 0;
253
254 USB_OTG_G->GRSTCTRL_B.CSRST = BIT_SET;
255
256 /* Wait for Reset ok */
257 while (USB_OTG_G->GRSTCTRL_B.CSRST != BIT_RESET)
258 {
259 if (++timeout > 0x30000)
260 {
261 break;
262 }
263 }
264
265 USB_Delay_us(3);
266}
