120static void USBD_LowPriorityProc(void)
121{
122 USBD_EP_T ep;
123
124 while (USBD_ReadIntFlag(USBD_INT_CTR))
125 {
126 ep = (USBD_EP_T)USBD_ReadEP();
127
129 if (ep == 0)
130 {
131 USBD_SetEPTxRxStatus(USBD_EP_0, USBD_EP_STATUS_NAK, USBD_EP_STATUS_NAK);
132
134 if (USBD_ReadDir() == 0)
135 {
136 USBD_ResetEPTxFlag(USBD_EP_0);
137 USBD_CtrlInProcess();
138 }
139 else
140 {
142 if (USBD_ReadEPSetup(USBD_EP_0) == SET)
143 {
144 USBD_ResetEPRxFlag(USBD_EP_0);
145 USBD_SetupProcess();
146 }
148 else
149 {
150 USBD_ResetEPRxFlag(USBD_EP_0);
151 USBD_CtrlOutProcess();
152 }
153 }
154 }
156 else
157 {
158 if (USBD_ReadEPRxFlag(ep))
159 {
160 USBD_ResetEPRxFlag(ep);
161 USBD_DataOutProcess(ep);
162 }
163
164 if (USBD_ReadEPTxFlag(ep))
165 {
166 USBD_ResetEPTxFlag(ep);
167 USBD_DataInProcess(ep);
168 }
169 }
170 }
171}

234static void USBD_SuspendIsrHandler(void)
235{
236 uint8_t i;
237 uint16_t bakEP[8];
238#if USB_LOW_POWER_SWITCH
239 uint32_t bakPwrCR;
240 uint32_t tmp;
241#endif
242
243 for (i = 0; i < 8; i++)
244 {
245 bakEP[i] = (uint16_t)USBD->EP[i].EP;
246 }
247
248 USBD_EnableInterrupt(USBD_INT_RST);
249
250 USBD_SetForceReset();
251 USBD_ResetForceReset();
252
253 while (USBD_ReadIntFlag(USBD_INT_RST) == RESET);
254
255 for (i = 0; i < 8; i++)
256 {
257 USBD->EP[i].EP = bakEP[i];
258 }
259
260 USBD_SetForceSuspend();
261
262#if USB_LOW_POWER_SWITCH
263 USBD_SetLowerPowerMode();
264
265 bakPwrCR = PMU->CTRL;
266 tmp = PMU->CTRL;
267 tmp &= (uint32_t)0xfffffffc;
268 tmp |= PMU_REGULATOR_LOWPOWER;
269 PMU->CTRL = tmp;
270
271 SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;
272
273 if (USBD_ReadIntFlag(USBD_INT_WKUP) == RESET)
274 {
275 __WFI();
276 SCB->SCR &= (uint32_t)~((uint32_t)SCB_SCR_SLEEPDEEP_Msk);
277 }
278 else
279 {
280 USBD_ClearIntFlag(USBD_INT_WKUP);
281 USBD_ResetForceSuspend();
282 PMU->CTRL = bakPwrCR;
283 SCB->SCR &= (uint32_t)~((uint32_t)SCB_SCR_SLEEPDEEP_Msk);
284 }
285#endif
286}
