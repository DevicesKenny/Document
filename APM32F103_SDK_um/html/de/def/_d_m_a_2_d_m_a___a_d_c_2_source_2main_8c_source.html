131void DMA_Init(uint32_t *Buf)
132{
133 DMA_Config_T dmaconfig;
134 RCM_EnableAHBPeriphClock(RCM_AHB_PERIPH_DMA1);
135
136 dmaconfig.bufferSize = 1;
137 dmaconfig.dir = DMA_DIR_PERIPHERAL_SRC;
138 dmaconfig.loopMode = DMA_MODE_CIRCULAR;
139 dmaconfig.memoryBaseAddr = (uint32_t)Buf;
140 dmaconfig.memoryDataSize = DMA_MEMORY_DATA_SIZE_HALFWORD;
141 dmaconfig.peripheralBaseAddr = (uint32_t)&ADC1->REGDATA;
142 dmaconfig.peripheralDataSize = DMA_PERIPHERAL_DATA_SIZE_HALFWORD;
143 dmaconfig.memoryInc = DMA_MEMORY_INC_DISABLE;
144 dmaconfig.peripheralInc = DMA_PERIPHERAL_INC_DISABLE;
145 dmaconfig.priority = DMA_PRIORITY_HIGH;
146 dmaconfig.M2M = DMA_M2MEN_DISABLE;
147
148 DMA_Config(DMA1_Channel1, &dmaconfig);
149
150 DMA_EnableInterrupt(DMA1_Channel1, DMA_INT_TC);
151 DMA_Enable(DMA1_Channel1);
152}

161void ADC_Init(void)
162{
163 GPIO_Config_T gpioConfig;
164 ADC_Config_T adcConfig;
165
166 /* Enable GPIOA clock */
167 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA);
168
169 /* ADC channel 0 configuration */
170 GPIO_ConfigStructInit(&gpioConfig);
171 gpioConfig.mode = GPIO_MODE_ANALOG;
172 gpioConfig.speed = GPIO_SPEED_50MHz;
173 gpioConfig.pin = GPIO_PIN_0;
174 GPIO_Config(GPIOA, &gpioConfig);
175
176 /* Enable ADC clock */
177 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_ADC1);
178
179 /* ADC configuration */
180 ADC_Reset(ADC1);
181 ADC_ConfigStructInit(&adcConfig);
182 adcConfig.continuosConvMode = ENABLE;
183 adcConfig.dataAlign = ADC_DATA_ALIGN_RIGHT;
184 adcConfig.externalTrigConv = ADC_EXT_TRIG_CONV_None;
185 adcConfig.scanConvMode = DISABLE;
186 ADC_Config(ADC1, &adcConfig);
187
188 /* ADC channel 10 Convert configuration */
189 ADC_ConfigRegularChannel(ADC1, ADC_CHANNEL_0, 1, ADC_SAMPLETIME_239CYCLES5);
190
191 ADC_EnableDMA(ADC1);
192
193 /* Enable ADC */
194 ADC_Enable(ADC1);
195
196 /* ADC start conversion */
197 ADC_EnableSoftwareStartConv(ADC1);
198}
