165void RCM_Configuration(void)
166{
167 RCM_Reset();
168
169 RCM_ConfigHSE(RCM_HSE_OPEN);
170 if(RCM_WaitHSEReady() == SUCCESS)
171 {
172 FMC_EnablePrefetchBuffer();
173 FMC_ConfigLatency(FMC_LATENCY_2);
174
175 RCM_ConfigAHB(RCM_AHB_DIV_1);
176 RCM_ConfigAPB2(RCM_APB_DIV_1);
177 RCM_ConfigAPB1(RCM_APB_DIV_2);
178 RCM_ConfigADCCLK(RCM_PCLK2_DIV_4);
179 RCM_ConfigPLL(RCM_PLLSEL_HSE, RCM_PLLMF_9);
180
181 RCM_EnablePLL();
182 while(RCM_ReadStatusFlag(RCM_FLAG_PLLRDY) == RESET);
183
184 RCM_ConfigSYSCLK(RCM_SYSCLK_SEL_PLL);
185 while(RCM_ReadSYSCLKSource() != RCM_SYSCLK_SEL_PLL);
186 }
187
188 RCM_EnableAPB2PeriphClock((RCM_APB2_PERIPH_T)(RCM_APB2_PERIPH_GPIOA | RCM_APB2_PERIPH_GPIOB | RCM_APB2_PERIPH_AFIO));
189 RCM_EnableAPB1PeriphClock((RCM_APB1_PERIPH_T)(RCM_APB1_PERIPH_SPI3 | RCM_APB1_PERIPH_SPI2));
190}

244uint8_t Buffercmp(uint16_t* pBuffer1, uint16_t* pBuffer2, uint16_t BufferLength)
245{
246 uint16_t i;
247
248 for(i = 0; i < BufferLength; i++)
249 {
250 if(pBuffer1[i] != pBuffer2[i])
251 {
252 return FAILED;
253 }
254 }
255
256 return PASSED;
257}

270uint8_t Buffercmp24bits(uint16_t* pBuffer1, uint16_t* pBuffer2, uint16_t BufferLength)
271{
272 uint16_t i;
273
274 for(i = 0; i < BufferLength; i++)
275 {
276 if(pBuffer1[i] != pBuffer2[i])
277 {
278 if(pBuffer1[i] != (pBuffer2[i] & 0XFF00))
279 {
280 return FAILED;
281 }
282 }
283 }
284
285 return PASSED;
286}
