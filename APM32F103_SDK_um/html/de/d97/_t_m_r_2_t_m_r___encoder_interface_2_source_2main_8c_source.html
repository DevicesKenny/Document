67int main(void)
68{
69 GPIO_Config_T gpioConfig;
70 USART_Config_T usartConfig;
71 TMR_BaseConfig_T baseConfig;
72 TMR_ICConfig_T ICConfig;
73
74 APM_MINI_LEDInit(LED2);
75 APM_MINI_LEDOff(LED2);
76
77 /* USART2 Tx: PA2 , Rx: PA3 */
78 usartConfig.baudRate = 115200;
79 usartConfig.mode = USART_MODE_TX_RX;
80 usartConfig.hardwareFlow = USART_HARDWARE_FLOW_NONE;
81 usartConfig.parity = USART_PARITY_NONE;
82 usartConfig.stopBits = USART_STOP_BIT_1;
83 usartConfig.wordLength = USART_WORD_LEN_8B;
84 APM_MINI_COMInit(COM2,&usartConfig);
85
86 /* Enable GPIOA Periph Clock */
87 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA);
88 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_AFIO);
89
90 /* Config GPIO */
91 gpioConfig.pin = GPIO_PIN_8 | GPIO_PIN_9;
92 gpioConfig.mode = GPIO_MODE_IN_PU;
93 gpioConfig.speed = GPIO_SPEED_50MHz;
94 GPIO_Config(GPIOA,&gpioConfig);
95
96 /* Config GPIOA Alernate Function for TMR1 */
97 GPIO_ConfigPinRemap(GPIO_PARTIAL_REMAP_TMR1);
98
99 /* Enable TMR1,TMR2 Periph Clock */
100 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_TMR1);
101 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_TMR2);
102
103 /* Config TMR1 */
104 baseConfig.clockDivision = TMR_CLOCK_DIV_1;
105 baseConfig.countMode = TMR_COUNTER_MODE_UP;
106 baseConfig.division = 71;
107 baseConfig.period = 0x0008;
108 baseConfig.repetitionCounter = 0;
109 TMR_ConfigTimeBase(TMR1, &baseConfig);
110
111 /* Config TMR2 */
112 baseConfig.clockDivision = TMR_CLOCK_DIV_1;
113 baseConfig.countMode = TMR_COUNTER_MODE_UP;
114 baseConfig.division = 71;
115 baseConfig.period = 999;
116 baseConfig.repetitionCounter = 0;
117 TMR_ConfigTimeBase(TMR2, &baseConfig);
118
119 /* Encoder */
120 TMR_ConfigEncodeInterface(TMR1,TMR_ENCODER_MODE_TI1,TMR_IC_POLARITY_RISING,TMR_IC_POLARITY_RISING);
121
122 /* Input Capture */
123 ICConfig.channel = TMR_CHANNEL_1;
124 ICConfig.polarity = TMR_IC_POLARITY_RISING;
125 ICConfig.selection = TMR_IC_SELECTION_DIRECT_TI;
126 ICConfig.prescaler = TMR_IC_PSC_1;
127 ICConfig.filter = 0;
128 TMR_ConfigIC(TMR1,&ICConfig);
129
130 ICConfig.channel = TMR_CHANNEL_2;
131 TMR_ConfigIC(TMR1,&ICConfig);
132
133 /* Counter */
134 TMR_ConfigCounter(TMR1,0x0004);
135 TMR_ClearStatusFlag(TMR1,TMR_FLAG_UPDATE);
136 TMR_ClearStatusFlag(TMR2,TMR_FLAG_UPDATE);
137
138 /* Enable TMR1,TMR2 Interrupt */
139 TMR_EnableInterrupt(TMR1, TMR_INT_UPDATE);
140 TMR_EnableInterrupt(TMR2, TMR_INT_UPDATE);
141 NVIC_EnableIRQRequest(TMR1_UP_IRQn, 0, 0);
142 NVIC_EnableIRQRequest(TMR2_IRQn, 0, 0);
143
144 /* Enable TMR1,TMR2 */
145 TMR_Enable(TMR1);
146 TMR_Enable(TMR2);
147
148 while(1)
149 {
150 APM_MINI_LEDToggle(LED2);
151 Delay();
152 }
153}

185int fputc(int ch, FILE* f)
186{
187 /* send a byte of data to the serial port */
188 USART_TxData(DEBUG_USART, (uint8_t)ch);
189
190 /* wait for the data to be send */
191 while (USART_ReadStatusFlag(DEBUG_USART, USART_FLAG_TXBE) == RESET);
192
193 return (ch);
194}
