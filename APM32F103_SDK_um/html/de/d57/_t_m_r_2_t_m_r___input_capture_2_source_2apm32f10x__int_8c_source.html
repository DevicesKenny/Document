190void TMR5_IRQHandler(void)
191{
192 uint32_t tempValue = 0;
193
194 /* Clear TMR5 Capture compare 2 interrupt pending bit */
195 TMR_ClearIntFlag(TMR5,TMR_INT_CC2);
196
197 /* Get the Input Capture value */
198 IC2Value[irqFlag] = TMR_ReadCaputer2(TMR5);
199
200 if(irqFlag != 0)
201 {
202 irqFlag = 0;
203
204 /* Capture computation */
205 if(IC2Value[1] > IC2Value[0])
206 {
207 tempValue = IC2Value[1] - IC2Value[0];
208 }
209 else if(IC2Value[1] < IC2Value[0])
210 {
211 tempValue = 0xFFFFFFFF - IC2Value[0] + IC2Value[1];
212 }
213 else
214 {
215 tempValue = 0;
216 }
217
218 /* Frequency computation */
219 Frequency = (RCM_ReadHCLKFreq()/2) / tempValue;
220 }
221 else
222 {
223 irqFlag++;
224 }
225}
