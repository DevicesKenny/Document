109void USBH_PollingProcess(void)
110{
111 if (!(g_usbHost.connectedFlag || g_usbHost.portEnable) && \
112 (USBH_ReadCurHostState() != USBH_HOST_IDLE))
113 {
114 USBH_ConfigHostState(USBH_HOST_DEVICE_DETACHED);
115 }
116
117 USBH_CoreHandler[g_usbHost.curHostState]();

127static void USBH_IdleHandler(void)
128{
129 /* Check attached state */
130 if (g_usbHost.connectedFlag)
131 {
132 g_usbHost.curHostState = USBH_HOST_DEVICE_ATTACHED;
133
134 g_userCallback.devAttachedHandler();
135
136 USB_Delay_us(100 * 1000);
137 }

147static void USBH_DevAttachedHandler(void)
148{
149 /* Get free channel */
150 g_usbHost.control.outChannelNum = USBH_CH_AllocChannel(0x00);
151 g_usbHost.control.inChannelNum = USBH_CH_AllocChannel(0x80);
152
153 /* Reset Port */
154 USB_OTG_PortReset();
155
156 g_userCallback.resetDevHandler();
157
158 g_usbHost.errorState = USBH_ERROR_NO;
159
160 g_usbHost.speed = USB_OTG_ReadPortSpeed();
161
162 g_userCallback.devSpeedDetectedHandler(g_usbHost.speed);
163
164 /* Open and Config out channel */
165 USBH_CH_OpenChannel(g_usbHost.control.outChannelNum,
166 0,
167 USB_EP_TYPE_CONTROL,
168 g_usbHost.control.maxPackSize);
169
170 /* Open and Config in channel */
171 USBH_CH_OpenChannel(g_usbHost.control.inChannelNum,
172 0,
173 USB_EP_TYPE_CONTROL,
174 g_usbHost.control.maxPackSize);
175
176 /* Prepare to handle enum */
177 g_usbHost.curHostState = USBH_HOST_ENUM;
178 g_usbHost.control.enumState = USBH_ENUM_IDLE;
179}

188static void USBH_DevDetachedHandler(void)
189{
190 g_userCallback.devDetachedHandler();
191
192 /* Config Global Interrupt */
193 USB_OTG_G->GINTMASK = 0;
194 USB_OTG_G->GINT = 0xFFFFFFFF;
195 USB_OTG_G->GCINT = 0xFFFFFFFF;
197 /* Get free channel */
198 USBH_CH_FreeAllChannel();
199
200 /* stop host */
201 USB_OTG_StopHost();
202
203 /* clear relevant State */
204 USBH_ConfigHostState(USBH_HOST_IDLE);
205 USBH_ConfigDeviceAddr(USBH_DEVICE_DEFAULT_ADDRESS);
206
207 g_usbHost.control.maxPackSize = USB_EP0_PACKET_SIZE;
208 g_usbHost.control.ctrlState = USBH_CTRL_SETUP;
209 g_usbHost.control.enumState = USBH_ENUM_IDLE;
210 g_usbHost.control.xferState = USBH_XFER_START;
211
212 USBH_DeInit();
213
214 USB_OTG_DisableAllGlobalInterrupt();
215
216 /* USB Global Init */
217 USB_GlobalInit();
218
219 /* USB Host Init */
220 USB_HostInit();
221
222 /* Enable USB Global interrupt */
223 USB_OTG_EnableAllGlobalInterrupt();
224}

342static void ErrorManage(USBH_ERROR_STATE_T error)
343{
344 switch (error)
345 {
346 case USBH_ERROR_NOT_SUPPORT:
347 g_userCallback.deviceNotSupportedHandler();
348 break;
349 case USBH_ERROR_UNRECOVERED:
350 g_userCallback.unrecoveredErrHandler();
351 break;
352 case USBH_APP_RESET:
353 break;
354 default :
355 break;
356 }
357
358 USBH_ConfigHostState(USBH_HOST_ERROR);
359}
