103uint8_t SCSI_CmdHandler(uint8_t lun, uint8_t* command)
104{
105 uint8_t ret = SCSI_OK;
106
107 switch (command[0])
108 {
109 case SCSI_CMD_TEST_UNIT_READY:
110 ret = SCSI_TestUnitReady(lun);
111 break;
112
113 case SCSI_CMD_INQUIRY:
114 ret = SCSI_Inquiry(lun, command);
115 break;
116
117 case SCSI_CMD_REQUEST_SENSE:
118 ret = SCSI_RequestSense(lun, command);
119 break;
120
121 case SCSI_CMD_READ_FORMAT_CAPACITIES:
122 ret = SCSI_ReadFormatCapacity(lun, command);
123 break;
124
125 case SCSI_CMD_READ_CAPACITY_10:
126 ret = SCSI_ReadCapacity10(lun, command);
127 break;
128
129 case SCSI_CMD_READ_10:
130 ret = SCSI_Read10(lun, command);
131 break;
132
133 case SCSI_CMD_WRITE10:
134 ret = SCSI_Write10(lun, command);
135 break;
136
137 case SCSI_CMD_VERIFY_10:
138 ret = SCSI_Verify10(lun, command);
139 break;
140 case SCSI_CMD_ALLOW_MEDIUM_REMOVAL:
141 case SCSI_CMD_START_STOP_UNIT:
142 ret = SCSI_StartStopUnit();
143 break;
144
145 case SCSI_CMD_MODE_SENSE_6:
146 ret = SCSI_ModeSense6(lun, command);
147 break;
148

151 break;
152
153 default:
154 SCSI_PutSenseCode(lun, SCSI_SKEY_ILLEGAL_REQUEST,
155 SCSI_ASC_INVALID_CDB, 0);
156 ret = SCSI_FAIL;

609static uint8_t SCSI_Read(uint8_t lun)
610{
611 uint32_t len = USB_MIN(MSC_MEDIA_PACKET, s_blkLen);
612
613 if (g_storageCallBack.ReadData(len, g_BOTInfo.data, (s_blkAddr / s_blkSize),
614 (len / s_blkSize)) != SCSI_OK)
615 {
616 SCSI_PutSenseCode(lun, SCSI_SKEY_HARDWARE_ERROR,
617 SCSI_ASC_UNRECOVERED_READ_ERROR, 0);
618
619 return SCSI_FAIL;
620 }
621
622 USBD_TxData(MSC_IN_EP & 0x7F, g_BOTInfo.data, len);
623
624 s_blkAddr += len;
625 s_blkLen -= len;
626
627 g_BOTInfo.CSW.dDataResidue -= len;
628
629 if (s_blkLen == 0)
630 {
631 g_BOTInfo.state = BOT_STATE_LAST_DATA_IN;
632 }
633
634 return SCSI_OK;
