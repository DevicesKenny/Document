108void DMA_Init(void)

142void ADC_Init(void)
143{
144 GPIO_Config_T gpioConfig;
145 ADC_Config_T adcConfig;
146
147 /* Enable GPIOA clock */
148 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA);
149
150 /* ADC channel 0 configuration */
151 GPIO_ConfigStructInit(&gpioConfig);
152 gpioConfig.mode = GPIO_MODE_ANALOG;
153 gpioConfig.pin = GPIO_PIN_0 | GPIO_PIN_1 | GPIO_PIN_2;
154 GPIO_Config(GPIOA, &gpioConfig);
155
156 /* Enable ADC clock */
157 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_ADC1);
158
159 /* ADC configuration */
160 ADC_Reset(ADC1);
161
162 ADC_ConfigStructInit(&adcConfig);
163 adcConfig.mode = ADC_MODE_INDEPENDENT;
164 adcConfig.scanConvMode = ENABLE;
165 adcConfig.continuosConvMode = ENABLE;
166 adcConfig.externalTrigConv = ADC_EXT_TRIG_CONV_None;
167 adcConfig.dataAlign = ADC_DATA_ALIGN_RIGHT;
168 /* channel number */
169 adcConfig.nbrOfChannel = ADC_CH_SIZE;
170 ADC_Config(ADC1, &adcConfig);
171
172 /* ADCCLK = PCLK2/4 */
173 RCM_ConfigADCCLK(RCM_PCLK2_DIV_4);
174
175 /* ADC channel Convert configuration */
176 ADC_ConfigRegularChannel(ADC1, ADC_CHANNEL_0, 1, ADC_SAMPLETIME_13CYCLES5);
177 ADC_ConfigRegularChannel(ADC1, ADC_CHANNEL_1, 2, ADC_SAMPLETIME_13CYCLES5);
178 ADC_ConfigRegularChannel(ADC1, ADC_CHANNEL_2, 3, ADC_SAMPLETIME_13CYCLES5);
179
180 /* Config DMA */
181 DMA_Init();
182
183 /* Enable ADC DMA */
184 ADC_EnableDMA(ADC1);
185
186 /* Enable ADC */
187 ADC_Enable(ADC1);
188
189 /* Enable ADC1 reset calibration register */
190 ADC_ResetCalibration(ADC1);
191 /* Check the end of ADC1 reset calibration register */
192 while(ADC_ReadResetCalibrationStatus(ADC1));
193
194 /* Start ADC1 calibration */
195 ADC_StartCalibration(ADC1);
196 /* Check the end of ADC1 calibration */
197 while(ADC_ReadCalibrationStartFlag(ADC1));
198
199 /* Start ADC1 Software Conversion */
200 ADC_EnableSoftwareStartConv(ADC1);
201}
