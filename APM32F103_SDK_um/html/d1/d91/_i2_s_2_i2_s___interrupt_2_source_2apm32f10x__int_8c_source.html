174void SPI3_IRQHandler(void)
175{
176 /* Check the interrupt source */
177 if (SPI_I2S_ReadIntFlag(SPI3, SPI_I2S_INT_TXBE) == SET)
178 {
179 /* Send a data from I2S3 */
180 SPI_I2S_TxData(SPI3, I2S3_Buffer_Tx[txCnt++]);
181 }
182
183 /* Check the end of buffer transfer */
184 if (rxCnt == 32)
185 {
186 /* Disable the I2S3 TXE interrupt to end the communication */
187 SPI_I2S_DisableInterrupt(SPI3, SPI_I2S_INT_TXBE);
188 }
189}

199void SPI2_IRQHandler(void)
200{
201 /* Check the interrupt source */
202 if (SPI_I2S_ReadIntFlag(SPI2, SPI_I2S_INT_RXBNE) == SET)
203 {
204 /* Store the I2S2 received data in the relative data table */
205 I2S2_Buffer_Rx[rxCnt++] = SPI_I2S_RxData(SPI2);
206 }
