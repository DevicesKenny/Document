71int main(void)
72{
73 GPIO_Config_T gpioConfig;
74 USART_Config_T usartConfig;
75 TMR_ICConfig_T ICConfig;
76 TMR_BaseConfig_T baseConfig;
77
78 /* Init LED */
79 APM_MINI_LEDInit(LED2);
80
81 /* USART1 Tx:PA9 Rx:PA10 */
82 usartConfig.baudRate = 115200;
83 usartConfig.mode = USART_MODE_TX_RX;
84 usartConfig.hardwareFlow = USART_HARDWARE_FLOW_NONE;
85 usartConfig.parity = USART_PARITY_NONE;
86 usartConfig.stopBits = USART_STOP_BIT_1;
87 usartConfig.wordLength = USART_WORD_LEN_8B;
88 APM_MINI_COMInit(COM1,&usartConfig);
89
90 /* Enable Periph Clock */
91 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_TMR1);
92 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_TMR5);
93 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA);
94
95 /* TMR5 chennel2 for PA1 */
96 gpioConfig.pin = GPIO_PIN_1;
97 gpioConfig.mode = GPIO_MODE_AF_PP;
98 gpioConfig.speed = GPIO_SPEED_50MHz;
99 GPIO_Config(GPIOA, &gpioConfig);
100
101 /* TMR5 PWM Input mode */
102 ICConfig.channel = TMR_CHANNEL_2;
103 ICConfig.polarity = TMR_IC_POLARITY_RISING;
104 ICConfig.selection = TMR_IC_SELECTION_DIRECT_TI;
105 ICConfig.prescaler = TMR_IC_PSC_1;
106 ICConfig.filter = 0x00;
107 TMR_ConfigPWM(TMR5, &ICConfig);
108
109 /* TMR5 Slave Mode */
110 TMR_SelectInputTrigger(TMR5,TMR_TRIGGER_SOURCE_TI2FP2);
111 TMR_SelectSlaveMode(TMR5,TMR_SLAVE_MODE_RESET);
112 TMR_EnableMasterSlaveMode(TMR5);
113
114 /* Enable TMR5 Interrupt */
115 TMR_Enable(TMR5);
116 NVIC_EnableIRQRequest(TMR5_IRQn, 0, 0);
117 TMR_EnableInterrupt(TMR5, TMR_INT_CC2);
118
119 /* Configure TMR1 */
120 baseConfig.clockDivision = TMR_CLOCK_DIV_1;
121 baseConfig.countMode = TMR_COUNTER_MODE_UP;
122 baseConfig.division = 71;
123 baseConfig.period = 999;
124 baseConfig.repetitionCounter = 0;
125 TMR_ConfigTimeBase(TMR1, &baseConfig);
126
127 /* Enable TMR1 Interrupt */
128 TMR_Enable(TMR1);
129 TMR_EnableInterrupt(TMR1, TMR_INT_UPDATE);
130 NVIC_EnableIRQRequest(TMR1_UP_IRQn, 0, 0);
131
132 while(1)
133 {
134 if(irqFlag > 1)
135 {
136 printf("DutyCycle: %d %%, Frequency: %d Hz\r\n",DutyCycle, Frequency);
137 APM_MINI_LEDToggle(LED2);
138 Delay();
139 }
140 }
141}

174{
