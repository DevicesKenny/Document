257void USBD_ConfigSerialString(uint8_t *string)
258{
259 uint32_t serialNum1 = 0;
260 uint16_t serialNum2 = 0;
261 uint8_t index, movCnt = 0;
262 serialNum1 = (*(uint32_t *)UID_ADDRESS1) + (*(uint32_t *)UID_ADDRESS2);
263 serialNum2 = (uint16_t)((*(uint32_t *)UID_ADDRESS1) + (*(uint32_t *)UID_ADDRESS3));
264 for (index = 2; index <= 16; index += 2)
265 {
266 string[index] = USBD_NumberToChar((uint8_t)(serialNum1 >> movCnt) & 0x0F);
267 if (movCnt < 16)
268 {
269 string[index + 16] = USBD_NumberToChar(((uint8_t)(serialNum2 >> movCnt) & 0x0F));
270 }
271 movCnt += 4;
272 }
273}
