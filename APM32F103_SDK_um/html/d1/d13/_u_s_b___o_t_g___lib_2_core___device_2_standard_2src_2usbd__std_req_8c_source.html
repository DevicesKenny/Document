139static uint8_t USBD_StandardGetConfiguration(void)
140{
141 /* Only for Device */
142 if (g_usbDev.reqData.domain.bmRequestType.bit.recipient == USBD_RECIPIENT_DEVICE)
143 {
144 /* Configured State */
145 if (g_usbDev.devState == USBD_DEVICE_STATE_CONFIGURED)
146 {
147 USBD_CtrlInData(&g_usbDev.curConfiguration, 1);
148 }
149 /* Address State */
150 else if (g_usbDev.devState == USBD_DEVICE_STATE_ADDRESS)
151 {
152 g_usbDev.curConfiguration = 0;
153 USBD_CtrlInData(&g_usbDev.curConfiguration, 1);
154 }
155 else
156 {
157 USB_OTG_SetStall(USB_EP_0);
158 return ERROR;
159 }
160
161 /* User Handler */
162 if (g_usbDev.pStdReqCallback->getConfigurationHandler != NULL)
163 {
164 g_usbDev.pStdReqCallback->getConfigurationHandler();
165 }
166
167 return SUCCESS;
168 }
169
170 USB_OTG_SetStall(USB_EP_0);
171
172 return ERROR;

182static uint8_t USBD_StandardGetDescriptor(void)
183{
184 uint32_t len = 0;
185 uint8_t wValue0 = g_usbDev.reqData.domain.wValue[0];
186 uint8_t wValue1 = g_usbDev.reqData.domain.wValue[1];
187 uint16_t reqLength = (g_usbDev.reqData.domain.wLength[1] << 8) | \
188 (g_usbDev.reqData.domain.wLength[0]);;
189
190 /* Device Descriptor */
191 if (wValue1 == USBD_DESC_DEVICE)
192 {
193 len = USB_MIN(reqLength, g_usbDev.pDeviceDesc->size);
194 USBD_CtrlInData((uint8_t*)g_usbDev.pDeviceDesc->pDesc, len);
195 }
196 /* Configuration Descriptor */
197 else if (wValue1 == USBD_DESC_CONFIGURATION)
198 {
199 len = USB_MIN(reqLength, g_usbDev.pConfigurationDesc->size);
200 USBD_CtrlInData((uint8_t*)g_usbDev.pConfigurationDesc->pDesc, len);
201 }
202 /* String Descriptor */
203 else if ((wValue1 == USBD_DESC_STRING) && (g_usbDev.pStringDesc[wValue0].pDesc != NULL))
204 {
205 len = USB_MIN(reqLength, g_usbDev.pStringDesc[wValue0].size);
206 USBD_CtrlInData((uint8_t*)g_usbDev.pStringDesc[wValue0].pDesc, len);
207 }
208 /* Qualifier Descriptor */
209 else if (wValue1 == USBD_DESC_DEVICE_QUALIFIER)
210 {
211#ifdef USB_OTG_FS
212 USB_OTG_SetStall(USB_EP_0);
213
214 return ERROR;
215#else
216 len = USB_MIN(reqLength, g_usbDev.pQualifierDesc->size);
217 USBD_CtrlInData((uint8_t*)g_usbDev.pQualifierDesc->pDesc, len);
218#endif
219 }
220 /* Other Descriptor */
221 else
222 {
223 /* User Handler */
224 if (g_usbDev.pStdReqCallback->getDescriptorHandler != NULL)
225 {
226 g_usbDev.pStdReqCallback->getDescriptorHandler();
227 }
228 else
229 {
230 USB_OTG_SetStall(USB_EP_0);
231
232 return ERROR;
233 }
234 }
235
236 return SUCCESS;
237}

277static uint8_t USBD_StandardGetStatus(void)
278{
279 uint8_t ret = 1;
280 USBD_DevReqData_T* reqData = &g_usbDev.reqData;
281 uint8_t recipient = reqData->domain.bmRequestType.bit.recipient;
282
283 uint8_t status[2] = {0, 0};
284
285 /* User Handler */
286 if (g_usbDev.pStdReqCallback->getStatusHandler != NULL)
287 {
288 g_usbDev.pStdReqCallback->getStatusHandler();
289 }
290
291 /* Device */
292 if (recipient == USBD_RECIPIENT_DEVICE)
293 {
294 /* Remote Wakeup */
295 if (g_usbDev.curFeature & (1 << 5))
296 {
297 status[0] |= 0x02;
298 }
299 /* Self power */
300 if (g_usbDev.curFeature & (1 << 6))
301 {
302 status[0] |= 0x01;
303 }
304
305 USBD_CtrlInData(status, 2);
306 }
307 /* Interface */
308 else if (recipient == USBD_RECIPIENT_INTERFACE)
309 {
310 USBD_CtrlInData(status, 2);
311 }
312 /* Endpoint */
313 else if (recipient == USBD_RECIPIENT_ENDPOINT)
314 {
315 /* IN Endpoint */
316 if (reqData->domain.wIndex[0] & 0x80)
317 {
318 status[0] = USB_OTG_ReadInEpStallStatus(reqData->domain.wIndex[0] & 0x7F);
319 }
320 /* OUT Endpoint */
321 else
322 {
323 status[0] = USB_OTG_ReadOutEpStallStatus(reqData->domain.wIndex[0] & 0x7F);
324 }
325
326 USBD_CtrlInData(status, 2);
327 }
328 else
329 {
330 USB_OTG_SetStall(USB_EP_0);
331
332 ret = 0;
333 }
334
335 return ret;
336}

455static uint8_t USBD_StandardSetFeature(void)
456{
457 uint8_t ret = 1;
458 uint8_t ep;
459
460 if ((g_usbDev.reqData.domain.wValue[0] == USBD_FEATURE_REMOTE_WAKEUP) &&
461 (g_usbDev.reqData.domain.bmRequestType.bit.recipient) == USBD_DESC_DEVICE)
462 {
463 g_usbDev.curFeature |= (1 << 5);
464 }
465 else if ((g_usbDev.reqData.domain.wValue[0] == USBD_FEATURE_ENDPOINT_HALT) &&
466 (g_usbDev.reqData.domain.bmRequestType.bit.recipient) == USBD_DESC_ENDPOINT)
467 {
468 ep = g_usbDev.reqData.domain.wIndex[0] & 0x7f;
469
470 if ((ep < USB_MAX(USB_OUT_EP_MAX_NUM, USB_IN_EP_MAX_NUM)) && (!g_usbDev.curConfiguration))
471 {
472 if (g_usbDev.reqData.domain.wIndex[0] & 0x80)
473 {
474 USB_OTG_SetInEpStall(ep);
475 }
476 else
477 {
478 USB_OTG_SetOutEpStall(ep);
479 }
480 }
481 }
482 else
483 {
484 ret = 0;
485 }
486
487 /* User Handler */
488 if (g_usbDev.pStdReqCallback->setFeatureHandler != NULL)
489 {
490 g_usbDev.pStdReqCallback->setFeatureHandler();

492
