74int main(void)
75{
76 /* ADC convert to voltage */
77 float voltage = 0;
78
79 USART_Config_T USART_ConfigStruct;
80
81 /* USART config */
82 USART_ConfigStruct.baudRate = 115200;
83 USART_ConfigStruct.hardwareFlow = USART_HARDWARE_FLOW_NONE;
84 USART_ConfigStruct.mode = USART_MODE_TX;
85 USART_ConfigStruct.parity = USART_PARITY_NONE;
86 USART_ConfigStruct.stopBits = USART_STOP_BIT_1;
87 USART_ConfigStruct.wordLength = USART_WORD_LEN_8B;
88
89 APM_MINI_COMInit(COM1, &USART_ConfigStruct);
90
91 /* DMA initialization */
92 DMA_Init();
93
94 /* ADC1 initialization */
95 ADC_Init();
96
97 while(1)
98 {
99 if(DMA_ReadStatusFlag(DMA1_FLAG_TC1) == SET)
100 {
101 voltage = (float)DMA_ADCConvertedValue / 4095 * 3.3;
102
103 printf("\r\n");
104 printf("ADC register data = 0x%04X \r\n", DMA_ADCConvertedValue);
105 printf("voltage = %.03f V \r\n", voltage);
106
107 Delay(200);
108 DMA_ClearStatusFlag(DMA1_FLAG_TC1);
109 }
110 }
111}

120void DMA_Init(void)
121{
122 DMA_Config_T dmaConfig;
123
124 /* Enable DMA Clock */
125 RCM_EnableAHBPeriphClock(RCM_AHB_PERIPH_DMA1);
126
127 /* DMA config */
128 dmaConfig.peripheralBaseAddr = ADC_DR_ADDR;
129 dmaConfig.memoryBaseAddr = (uint32_t)&DMA_ADCConvertedValue;
130 dmaConfig.dir = DMA_DIR_PERIPHERAL_SRC;
131 dmaConfig.bufferSize = 1;
132 dmaConfig.peripheralInc = DMA_PERIPHERAL_INC_DISABLE;
133 dmaConfig.memoryInc = DMA_MEMORY_INC_DISABLE;
134 dmaConfig.peripheralDataSize = DMA_PERIPHERAL_DATA_SIZE_HALFWORD;
135 dmaConfig.memoryDataSize = DMA_MEMORY_DATA_SIZE_HALFWORD;
136 dmaConfig.loopMode = DMA_MODE_CIRCULAR;
137 dmaConfig.priority = DMA_PRIORITY_HIGH;
138 dmaConfig.M2M = DMA_M2MEN_DISABLE;
139
140 /* Enable DMA channel */
141 DMA_Config(DMA1_Channel1, &dmaConfig);
142
143 /* Enable DMA */
144 DMA_Enable(DMA1_Channel1);
145}

154void ADC_Init(void)
155{
156 GPIO_Config_T gpioConfig;
157 ADC_Config_T adcConfig;
158
159 /* Enable GPIOA clock */
160 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA);
161
162 /* Configure PC0 (ADC Channel0) as analog input */
163 GPIO_ConfigStructInit(&gpioConfig);
164 gpioConfig.mode = GPIO_MODE_ANALOG;
165 gpioConfig.pin = GPIO_PIN_0;
166 GPIO_Config(GPIOA, &gpioConfig);
167
168 /* ADCCLK = PCLK2/4 */
169 RCM_ConfigADCCLK(RCM_PCLK2_DIV_4);
170
171 /* Enable ADC clock */
172 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_ADC1);
173
174 /* ADC configuration */
175 ADC_Reset(ADC1);
176
177 ADC_ConfigStructInit(&adcConfig);
178 adcConfig.mode = ADC_MODE_INDEPENDENT;
179 adcConfig.scanConvMode = DISABLE;
180 adcConfig.continuosConvMode = ENABLE;
181 adcConfig.externalTrigConv = ADC_EXT_TRIG_CONV_None;
182 adcConfig.dataAlign = ADC_DATA_ALIGN_RIGHT;
183 /* channel number */
184 adcConfig.nbrOfChannel = 1;
185 ADC_Config(ADC1, &adcConfig);
186
187 /* ADC channel Convert configuration */
188 ADC_ConfigRegularChannel(ADC1, ADC_CHANNEL_0, 1, ADC_SAMPLETIME_13CYCLES5);
189
190 /* Enable ADC DMA */
191 ADC_EnableDMA(ADC1);
192
193 /* Enable ADC */
194 ADC_Enable(ADC1);
195
196 /* Enable ADC1 reset calibration register */
197 ADC_ResetCalibration(ADC1);
198 /* Check the end of ADC1 reset calibration register */
199 while(ADC_ReadResetCalibrationStatus(ADC1));
200
201 /* Start ADC1 calibration */
202 ADC_StartCalibration(ADC1);
203 /* Check the end of ADC1 calibration */
204 while(ADC_ReadCalibrationStartFlag(ADC1));
205
206 /* Start ADC1 Software Conversion */
207 ADC_EnableSoftwareStartConv(ADC1);
208}
