69void USBH_OTG_IsrHandler(void)
70{
71 /* Handle RxFIFO no empty interrupt */
72 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_RXFNONE))
73 {
74 USBH_RxFifoNoEmptyIsrHandler();
75 }
76 /* Handle USB Non-periodic TxFIFO empty interrupt */
77 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_NPTXFEM))
78 {
79 USBH_NonPerTxFifoEmptyIsrHandler();
80 }
81
82 /* Handle USB Periodic TxFIFO empty interrupt */
83 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_PTXFE))
84 {
85 USBH_PerTxFifoEmptyIsrHandler();
86 }
87
88 /* Handle Start of Frame interrupt */
89 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_SOF))
90 {
91 USBH_FrameStartIsrHandler();
92 }
93
94 /* Handle Host channels interrupt */
95 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_HCHAN))
96 {
97 USBH_HostChannelsIsrHandler();
98 }
99
100 /* Handle Incomplete isochronous OUT transfer interrupt */
101 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_IP_OUTTX))
102 {
103 USBH_ISOOutTXIsrHandler();
104 }
105
106 /* Handle Host port interrupt */
107 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_HPORT))
108 {
109 USBH_HostPortIsrHandler();
110 }
111
112 /* Handle Disconnect detected interrupt */
113 if (USB_OTG_ReadGlobalIntFlag(USB_INT_G_DEDIS))
114 {
115 USBH_DisconnectIsrHandler();
116 }
117
118 /* Hadler the rest of global interrupt */
119 if (g_usbHost.intHandler)
120 {
121 g_usbHost.intHandler();
122 }
123}

132static void USBH_RxFifoNoEmptyIsrHandler(void)
133{
134 USBH_FIFO_Status_T fifoStatus;
135 uint16_t channelNum;
136 uint8_t packetStatus;
137
138 /* Disable RxFIFO no empty Interrup */
139 USB_OTG_DisableGlobalInterrupt(USB_INT_G_RXFNONE);
140
141 /* Read and pop the RxFIFO status data */
142 fifoStatus.word = USB_OTG_PopRxFifoStatus();
143 channelNum = fifoStatus.bit.chNum;
144 packetStatus = fifoStatus.bit.status;
145
146 if (packetStatus == PKTSTS_IN)
147 {
148 /* Read the data into the host buffer */
149 if ((fifoStatus.bit.byteCount > 0) && (g_usbHost.xferPipe[channelNum].buf != NULL))
150 {
151 USB_FIFO_ReadRxFifoPacket(g_usbHost.xferPipe[channelNum].buf, fifoStatus.bit.byteCount);
152 /* manage multiple Xfer */
153 g_usbHost.xferPipe[channelNum].buf += fifoStatus.bit.byteCount;
154 g_usbHost.xferPipe[channelNum].bufCount += fifoStatus.bit.byteCount;
155
156 if (USB_OTG_H_ReadPacketCount(channelNum) > 0)
157 {
158 /* re-activate the channel when more packets are expected */
159 USB_OTG_H_EnableChannel(channelNum);
160 }
161 }
162 }
163 else if (packetStatus == PKTSTS_IN_XFER_COMPLETE)
164 {
165 }
166 else if (packetStatus == PKTSTS_DATA_ISO_ERR)
167 {
168 }
169 else if (packetStatus == PKTSTS_CHANNEL_HALT)
170 {
171 }
172 else
173 {
174 }
175
176 /* Enable RxFIFO no empty Interrupt */
177 USB_OTG_EnableGlobalInterrupt(USB_INT_G_RXFNONE);
178}

187static void USBH_NonPerTxFifoEmptyIsrHandler(void)
188{
189 uint32_t txFifoSpace;
190 uint16_t len;
191 uint8_t channelNum;
192
193 channelNum = ((USB_OTG_G->GNPTXFQSTS_B.NPTXRQ >> 3) & 0x0F);
194 len = g_usbHost.xferPipe[channelNum].bufLen;
195 txFifoSpace = USB_OTG_G->GNPTXFQSTS_B.NPTXFSA * 4;
196
197 while ((txFifoSpace > len) && (g_usbHost.xferPipe[channelNum].bufLen != 0))
198 {
199 if (txFifoSpace > g_usbHost.xferPipe[channelNum].bufLen)
200 {
201 /* Last packet */
202 len = g_usbHost.xferPipe[channelNum].bufLen;
203
204 USB_OTG_DisableGlobalInterrupt(USB_INT_G_NPTXFEM);
205 }
206 else
207 {
208 len = txFifoSpace;
209 }
210
211 USB_FIFO_WriteFifoPacket(channelNum, g_usbHost.xferPipe[channelNum].buf, len);
212
213 g_usbHost.xferPipe[channelNum].buf += len;
214 g_usbHost.xferPipe[channelNum].bufLen -= len;
215 g_usbHost.xferPipe[channelNum].bufCount += len;
216
217 txFifoSpace = USB_OTG_G->GNPTXFQSTS_B.NPTXFSA * 4;
218 }

327static void USBH_HostPortIsrHandler(void)
328{
329 /* Read Port Connect Flag */
330 if (USB_OTG_H_ReadPortConnectIntTriFlag() == SET)
331 {
332 USB_OTG_H_ClearPortConnectIntTriFlag();
333 /* USBH Connected Handler */
334 g_usbHost.connectedFlag = 1;
335 }
336
337 /* Port Enable is Changed */
338 if (USB_OTG_H_PenBitIsChange() == SET)
339 {
340 /* Port is Enable */
341 if (USB_OTG_H_PortIsEnable() == SET)
342 {
343 if (USB_OTG_H_ReadPortSpeedType() == SPEED_LS)
344 {
345 USB_OTG_H_SetFrameInterval(6000);
346 if (USB_OTG_H_ReadHostClk() != PHYCLK_6_MHZ)
347 {
348#ifdef OTG_HS_PHY_ULPI
349 USB_OTG_H_SetFrameInterval(48000);
350 if (USB_OTG_H_ReadHostClk() != PHYCLK_48_MHZ)
351 {
352 USB_OTG_H_ConfigHostClk(PHYCLK_48_MHZ);
353 }
354#else
355 USB_OTG_H_ConfigHostClk(PHYCLK_6_MHZ);
356#endif
357 USB_OTG_PortReset();
358 }
359 }
360 else if (USB_OTG_H_ReadPortSpeedType() == SPEED_FS)
361 {
362 USB_OTG_H_SetFrameInterval(48000);
363 if(USB_OTG_H_ReadHostClk() != PHYCLK_48_MHZ)
364 {
365 USB_OTG_H_ConfigHostClk(PHYCLK_48_MHZ);
366 USB_OTG_PortReset();
367 }
368 }
369 else
370 {
371 USB_OTG_PortReset();
372 }
373
374 /* USBH Port Enabled Handler */
375 g_usbHost.portEnable = 1;
376
377 /* Enable Disconnect detected interrupt */
378 USB_OTG_EnableGlobalInterrupt(USB_INT_G_DEDIS);
379 }
380 else
381 {
382 /* USBH Port Disabled Handler */
383 g_usbHost.portEnable = 0;
384 }
385 }
386
387 /* Port overcurrent is Changed */
388 if (USB_OTG_H_PovcBitIsChange() == SET)
389 {
390 }
391 /* Clear Port Interrupts */
392 USB_OTG_H_DisablePort();
393}

421static void USBH_IN_HostChannelIsrHandler(uint16_t channelNum)
422{
423 uint8_t intNak;
424 intNak = USB_OTG_H_ReadChannelIntFlag(channelNum, CHANNEL_INT_DTOG);
425
426 if (USB_OTG_H_ReadChannelIntFlag(channelNum, CHANNEL_INT_AHBERR))
427 {
428 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_AHBERR);
429 USB_OTG_H_EnableChannelInt(channelNum, CHANNEL_INT_TSFCMPAN);
430 }
431
432 else if (USB_OTG_H_ReadChannelIntFlag(channelNum, CHANNEL_INT_RXTXACK))
433 {
434 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_RXTXACK);
435 }
436
437 else if (USB_OTG_H_ReadChannelIntFlag(channelNum, CHANNEL_INT_RXSTALL))
438 {
439 USB_OTG_H_EnableChannelInt(channelNum, CHANNEL_INT_TSFCMPAN);
440 g_usbHost.xferPipe[channelNum].pipeState = PIPE_STALL;
441 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_RXNAK);
442 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_RXSTALL);
443 intNak = 0;
444
445 USB_OTG_HaltChannel(channelNum);
446 }
447
448 else if (USB_OTG_H_ReadChannelIntFlag(channelNum, CHANNEL_INT_DTOG))
449 {
450 USB_OTG_H_EnableChannelInt(channelNum, CHANNEL_INT_TSFCMPAN);
451 USB_OTG_HaltChannel(channelNum);
452 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_RXNAK);
453 g_usbHost.xferPipe[channelNum].pipeState = PIPE_DATATGLERR;
454 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_DTOG);
455 }
456
457
458 if (USB_OTG_H_ReadChannelIntFlag(channelNum, CHANNEL_INT_FOVR))
459 {
460 USB_OTG_H_EnableChannelInt(channelNum, CHANNEL_INT_TSFCMPAN);
461 USB_OTG_HaltChannel(channelNum);
462 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_FOVR);
463 }
464
465 else if (USB_OTG_H_ReadChannelIntFlag(channelNum, CHANNEL_INT_TSFCMPN))
466 {
467 g_usbHost.xferPipe[channelNum].pipeState = PIPE_OK;
468 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_TSFCMPN);
469
470 if ((USB_OTG_H_ReadEpType(channelNum) == EP_TYPE_CONTROL)
471 || (USB_OTG_H_ReadEpType(channelNum) == EP_TYPE_BULK))
472 {
473 USB_OTG_H_EnableChannelInt(channelNum, CHANNEL_INT_TSFCMPAN);
474 USB_OTG_HaltChannel(channelNum);
475 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_RXNAK);
476 g_usbHost.xferPipe[channelNum].pidToggle ^= 1;
477 }
478
479 else if (USB_OTG_H_ReadEpType(channelNum) == EP_TYPE_INTERRUPT)
480 {
481 USB_OTG_H_TxUseOddFrame(channelNum);
482 g_usbHost.xferPipe[channelNum].urbStatus = USBH_URB_OK;
483 }
484 }
485 else if (USB_OTG_H_ReadChannelIntFlag(channelNum, CHANNEL_INT_TSFCMPAN))
486 {
487 USB_OTG_H_DisableChannelInt(channelNum, CHANNEL_INT_TSFCMPAN);
488
489 if (g_usbHost.xferPipe[channelNum].pipeState == PIPE_OK)
490 {
491 g_usbHost.xferPipe[channelNum].urbStatus = USBH_URB_OK;
492 }
493
494 else if (g_usbHost.xferPipe[channelNum].pipeState == PIPE_STALL)
495 {
496 g_usbHost.xferPipe[channelNum].urbStatus = USBH_URB_STALL;
497 }
498
499 else if ((g_usbHost.xferPipe[channelNum].pipeState == PIPE_XACTERR)
500 || (g_usbHost.xferPipe[channelNum].pipeState == PIPE_DATATGLERR))
501 {
502 g_usbHost.xferPipe[channelNum].urbStatus = USBH_URB_ERROR;
503 }
504 if (USB_OTG_H_ReadEpType(channelNum) == EP_TYPE_INTERRUPT)
505 {
506 g_usbHost.xferPipe[channelNum].pidToggle ^= 1;
507 }
508
509 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_TSFCMPAN);
510
511 }
512 else if (USB_OTG_H_ReadChannelIntFlag(channelNum, CHANNEL_INT_TERR))
513 {
514 USB_OTG_H_EnableChannelInt(channelNum, CHANNEL_INT_TSFCMPAN);
515 g_usbHost.xferPipe[channelNum].pipeState = PIPE_XACTERR;
516 USB_OTG_HaltChannel(channelNum);
517 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_TERR);
518 }
519 else if ((USB_OTG_H_ReadChannelIntFlag(channelNum, CHANNEL_INT_RXNAK)) || (intNak == 1))
520 {
521 if (USB_OTG_H_ReadEpType(channelNum) == EP_TYPE_INTERRUPT)
522 {
523 USB_OTG_H_EnableChannelInt(channelNum, CHANNEL_INT_TSFCMPAN);
524
525 USB_OTG_HaltChannel(channelNum);
526 }
527 else if ((USB_OTG_H_ReadEpType(channelNum) == EP_TYPE_CONTROL)
528 || (USB_OTG_H_ReadEpType(channelNum) == EP_TYPE_BULK))
529 {
530 /* Enable the channel */
531 USB_OTG_H_EnableChannel(channelNum);
532 }
533 g_usbHost.xferPipe[channelNum].pipeState = PIPE_NAK;
534 USB_OTG_H_ClearChannelIntFlag(channelNum, CHANNEL_INT_RXNAK);
535 }
