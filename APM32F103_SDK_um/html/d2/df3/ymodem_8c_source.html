116uint16_t CRC16_Update(uint16_t value, uint8_t b)
117{
118 uint32_t value_crc = value;
119 uint32_t value_int = b | 0x100;
120 uint16_t returnTemp;
121
122 do
123 {
124 value_crc <<= 1;
125 value_int <<= 1;
126
127 if(value_int & 0x100)
128 {
129 ++ value_crc;
130 }
131
132 if(value_crc & 0x10000)
133 {
134 value_crc ^= 0x1021;
135 }
136 }
137 while(!(value_int & 0x10000));
138
139 returnTemp = (uint16_t)(value_crc & 0xffff);
140
141 return returnTemp;
142}

155uint16_t CRC16_Sum(const uint8_t *data, uint32_t len)
156{
157 uint32_t crc_sum = 0;
158 const uint8_t *data_len = data + len;
159
160 do
161 {
162 crc_sum = CRC16_Update(crc_sum,*data++);
163 }
164 while(data < data_len);
165
166 crc_sum = CRC16_Update(CRC16_Update(crc_sum,0),0);
167
168 crc_sum &= (uint16_t)0xffff;
169
170 return crc_sum;
171}

219static int32_t Receive_YmodePacket(uint8_t *data, int32_t *len, uint32_t timeout)
220{
221 uint16_t i, pack_len, pack_crc;
222 uint8_t character;
223 *len = 0;
224
225 if(Receive_Byte(&character, timeout) != SUCCESS)
226 {
227 return -1;
228 }
229
230 if(character == FRAME_SOH)
231 {
232 pack_len = PACKET_SIZE;
233 }
234 else if(character == FRAME_STX)
235 {
236 pack_len = PACKET_1K_SIZE;
237 }
238 else if(character == FRAME_EOT)
239 {
240 return 0;
241 }
242 else if(character == FRAME_CAN)
243 {
244 if((Receive_Byte(&character, timeout) == SUCCESS) && (character == FRAME_CAN))
245 {
246 *len = -1;
247 return 0;
248 }
249 else
250 {
251 return -1;
252 }
253 }
254 else if((character == ABORT1)|(character == ABORT2))
255 {
256 return 1;
257 }
258 else
259 {
260 return -1;
261 }
262
263 *data = character;
264
265 for(i = 1; i < (pack_len + PACK_FRAME_OVERHEAD); i ++)
266 {
267 if(Receive_Byte(data + i, timeout) != SUCCESS)
268 {
269 return -1;
270 }
271 }
272
273 if(data[PACK_FRAMENO_INDEX] != ((data[PACK_FRAMENO_COMP_INDEX] ^ 0xff) & 0xff))
274 {
275 return -1;
276 }
277
278 /* Calculate the CRC16 */
279 pack_crc = CRC16_Sum(&data[PACK_FRAME_HEADER], (uint32_t)pack_len);
280
281 if(pack_crc != (uint16_t)((data[pack_len + 3] << 8) | data[pack_len + 4]))
282 {
283 /* CRC16 error */
284 return -1;
285 }
286
287 *len = pack_len;
288
289 return 0;
290}
