113void USB_OTG_EnableOutChannelTransfer(uint16_t chNum,
114 uint8_t* transferBuff,
115 uint32_t transferLen,
116 uint16_t maxPackSize,
117 uint8_t dataPid,
118 uint8_t epType)
119{
120 uint16_t transferWordLen = 0;
121 uint16_t packetNum;
122 /* the number of packets to transfer */
123 if (transferLen > 0)
124 {
125 packetNum = (transferLen + maxPackSize - 1) / maxPackSize;
126
127 if (packetNum > 256)
128 {
129 packetNum = 256;
130 transferLen = packetNum * maxPackSize;
131 }
132 }
133 else
134 {
135 packetNum = 1;
136 }
137
138 /* Config the HCHTSIZE register */
139 USB_OTG_H_ConfigTransferSize(chNum, transferLen);
140 USB_OTG_H_ConfigPacketCount(chNum, packetNum);
141 USB_OTG_H_ConfigDataPID(chNum, dataPid);
142 USB_OTG_H_DisableDoPing(chNum);
143
144 if ((USB_OTG_H_ReadFrameNum() % 2) == 1)
145 {
146 USB_OTG_H_TxUseEvenFrame(chNum);
147 }
148 else
149 {
150 USB_OTG_H_TxUseOddFrame(chNum);
151 }
152 /* enable channel */
153 USB_OTG_H_EnableChannel(chNum);
154
155 if (transferLen > 0)
156 {
157 if ((epType == USB_EP_TYPE_CONTROL) || \
158 (epType == USB_EP_TYPE_BULK))
159 {
160 transferWordLen = (transferLen + 3) / 4;
161
162 /* check FIFO space */
163 if (transferWordLen > USB_OTG_G->GNPTXFQSTS_B.NPTXFSA)
164 {
165 /* enable Non-periodic TxFIFO empty interrupt */
166 USB_OTG_EnableGlobalInterrupt(USB_INT_G_NPTXFEM);
167 }
168 else
169 {
170 /* Write packet into FIFO */
171 USB_FIFO_WriteFifoPacket(chNum, transferBuff, transferLen);
172 }
173 }
174 else if ((epType == USB_EP_TYPE_ISO) || \
175 (epType == USB_EP_TYPE_INTERRUPT))
176 {
177 transferWordLen = (transferLen + 3) / 4;
178 /* check FIFO space */
179 if (transferWordLen > USB_OTG_H->HPTXSTS_B.FSPACE)
180 {
181 /* enable Periodic TxFIFO empty interrupt */
182 USB_OTG_EnableGlobalInterrupt(USB_INT_G_PTXFE);
183 }
184 else
185 {
186 /* Write packet into FIFO */
187 USB_FIFO_WriteFifoPacket(chNum, transferBuff, transferLen);
188 }
189 }
190
191
192 }

274void USB_OTG_HaltChannel(uint16_t chNum)
275{
276 /* Read Remain space */
277 if ((USB_OTG_H_ReadEpType(chNum) == EP_TYPE_CONTROL)
278 || (USB_OTG_H_ReadEpType(chNum) == EP_TYPE_BULK))
279 {
280 if (USB_OTG_G->GNPTXFQSTS_B.NPTXRSA == 0)
281 {
282 USB_OTG_H_ClearQueue(chNum);

306void USB_OTG_InitChannel(USB_ChannelConfig_T* channelConfig)
307{
308 /* Clear interrupt conditions */
309 USB_OTG_H_ClearChannelIntFlag(channelConfig->chNum, CHANNEL_INT_ALL);
310
311 /* Enable channel interrupts required for this transfer. */
312 USB_OTG_H_DisableChannelInt(channelConfig->chNum, CHANNEL_INT_ALL);
313
314 if ((channelConfig->epType == USB_EP_TYPE_CONTROL)
315 || (channelConfig->epType == USB_EP_TYPE_BULK))
316 {
317 USB_OTG_H_EnableChannelInt(channelConfig->chNum,
318 (CHANNEL_INT_TSFCMPN |
319 CHANNEL_INT_RXSTALL |
320 CHANNEL_INT_TERR |
321 CHANNEL_INT_DTOG |
322 CHANNEL_INT_RXNAK));
323
324 if (channelConfig->direction == EP_DIR_IN)
325 {
326 USB_OTG_H_EnableChannelInt(channelConfig->chNum, CHANNEL_INT_BABBLE);
327 }
328 else
329 {
330 USB_OTG_H_EnableChannelInt(channelConfig->chNum, CHANNEL_INT_RXNYET);
331 if (channelConfig->enbalePing == 1)
332 {
333 USB_OTG_H_EnableChannelInt(channelConfig->chNum, CHANNEL_INT_RXTXACK);
334 }
335 }
336 }
337 else if (channelConfig->epType == USB_EP_TYPE_ISO)
338 {
339 USB_OTG_H_EnableChannelInt(channelConfig->chNum,
340 (CHANNEL_INT_TSFCMPN |
341 CHANNEL_INT_FOVR |
342 CHANNEL_INT_RXTXACK));
343
344
345 if (channelConfig->direction == EP_DIR_IN)
346 {
347 USB_OTG_H_EnableChannelInt(channelConfig->chNum,
348 (CHANNEL_INT_TERR |
349 CHANNEL_INT_BABBLE));
350 }
351 }
352 else if (channelConfig->epType == USB_EP_TYPE_INTERRUPT)
353 {
354 USB_OTG_H_EnableChannelInt(channelConfig->chNum,
355 (CHANNEL_INT_TSFCMPN |
356 CHANNEL_INT_RXNAK |
357 CHANNEL_INT_RXSTALL |
358 CHANNEL_INT_TERR |
359 CHANNEL_INT_DTOG |
360 CHANNEL_INT_FOVR));
361
362 if (channelConfig->direction == EP_DIR_IN)
363 {
364 USB_OTG_H_EnableChannelInt(channelConfig->chNum, CHANNEL_INT_BABBLE);
365 }
366 }
367
368 /* Enable the top level host channel interrupt. */
369 USB_OTG_H_EnableAllChannelsInt(1 << channelConfig->chNum);
370
371 /* Enabled global host channel interrupts */
372 USB_OTG_EnableGlobalInterrupt(USB_INT_G_HCHAN);
373
374 /* Config the OTG_HS1_HCHX register */
375 USB_OTG_H_ConfigDeviceAddr(channelConfig->chNum, channelConfig->devAddr);
376
377 USB_OTG_H_ConfigEpNum(channelConfig->chNum, channelConfig->epNum);
378
379 USB_OTG_H_ConfigEpDirection(channelConfig->chNum, channelConfig->direction);
380
381 if (channelConfig->devSpeed == SPEED_LS)
382 {
383 USB_OTG_H_EnableLowSpeedDevice(channelConfig->chNum);
384 }
385 else
386 {
387 USB_OTG_H_DisableLowSpeedDevice(channelConfig->chNum);
388 }
389
390 USB_OTG_H_ConfigEpType(channelConfig->chNum, channelConfig->epType);
391
392 USB_OTG_H_ConfigEpMaxPSize(channelConfig->chNum, channelConfig->maxPackSize);
393
394 if (channelConfig->epType == USB_EP_TYPE_INTERRUPT)
395 {
396 USB_OTG_H_TxUseOddFrame(channelConfig->chNum);
397 }
398 else
399 {
400 USB_OTG_H_TxUseEvenFrame(channelConfig->chNum);
401 }
