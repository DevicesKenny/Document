105void USBH_CDC_ClassInitHandler(void)
106{
107 uint8_t itfNum, epNum;
108
109 /* Initialize cdc buff */
110 CDC_Tx.cdcState = USBH_CDC_IDIE;
111 CDC_Tx.dataLen = 0;
112 CDC_Tx.buff = txBuf;
113
114 CDC_Rx.cdcState = USBH_CDC_IDIE;
115 CDC_Rx.dataLen = 0;
116 CDC_Rx.buff = rxBuf;
117 CDC_Rx.pFillBuff = CDC_Rx.buff;
118 CDC_Rx.pEmptyBuff = CDC_Rx.buff;
119 CDC_Rx.buffLen = CDC_BUFFER_SIZE;
120
121 /* Prepare to Handler class request */
122 g_usbHost.preHostState = USBH_HOST_CLASS_REQ;
123
124 itfNum = USBH_ReadConfigurationItfNum(&g_usbHost);
125
126 while (itfNum--)
127 {
128 if (USBH_ReadInterfaceClass(&g_usbHost, itfNum) != USBH_CLASS_CDCC)
129 {
130 if (USBH_ReadInterfaceClass(&g_usbHost, itfNum) != USBH_CLASS_CDCD)
131 {
132 g_userCallback.deviceNotSupportedHandler();
133 continue;
134 }
135 }
136
137 epNum = USBH_ReadInterfaceEpNum(&g_usbHost, itfNum);
138 while (epNum--)
139 {
140 if (USBH_ReadEndpointType(&g_usbHost, itfNum, epNum) == USB_EP_TYPE_BULK)
141 {
142 /* Get endpoint and size */
143 if (USBH_ReadEndpointAddress(&g_usbHost, itfNum, epNum) & 0x80)
144 {
145 CDC_Rx.chNum = USBH_CH_AllocChannel(USBH_ReadEndpointAddress(&g_usbHost, itfNum, epNum));
146 CDC_Rx.maxPackSize = USBH_ReadEndpointMPS(&g_usbHost, itfNum, epNum);
147
148 /* Open and Config data in channel */
149 USBH_CH_OpenChannel(CDC_Rx.chNum,
150 USBH_DEVICE_CONFIGURED_ADDRESS,
151 USB_EP_TYPE_BULK,
152 CDC_Rx.maxPackSize);
153 }
154 else
155 {
156 CDC_Tx.chNum = USBH_CH_AllocChannel(USBH_ReadEndpointAddress(&g_usbHost, itfNum, epNum));
157 CDC_Tx.maxPackSize = USBH_ReadEndpointMPS(&g_usbHost, itfNum, epNum);
158
159 /* Open and Config data in channel */
160 USBH_CH_OpenChannel(CDC_Tx.chNum,
161 USBH_DEVICE_CONFIGURED_ADDRESS,
162 USB_EP_TYPE_BULK,
163 CDC_Tx.maxPackSize);
164 }
165 }
166 }
167 }
168
169 /* Go to Class request handler */
170 g_usbHost.curHostState = USBH_HOST_CLASS_REQ;
171
172 CDC_EnableGet();
173}

418uint8_t USBH_CDCProcessGet(void)
419{
420 USBH_URB_STATUS_T urbState;
421
422 if (flagEnableGet == 1)
423 {
424 if ((CDC_Rx.cdcState == USBH_CDC_COMPLETE) || (CDC_Rx.cdcState == USBH_CDC_IDIE))
425 {
426 CDC_Rx.cdcState = USBH_CDC_GET_DATA;
427 }
428 }
429 else
430 {
431 CDC_DisableGet();
432 CDC_Rx.cdcState = USBH_CDC_IDIE;
433 CDC_Rx.cdcState = CDC_Rx.cdcState;
434 return CDC_Rx.cdcState;
435 }
436
437 switch (CDC_Rx.cdcState)
438 {
439 case USBH_CDC_IDIE:
440 break;
441
442 case USBH_CDC_GET_DATA:
443 /* Sent OUT packet */
444 if (CDC_Rx.dataLen < (CDC_Rx.buffLen - g_usbHost.xferPipe[CDC_Rx.chNum].maxPackSize))
445 {
446 USBH_BulkInData(CDC_Rx.chNum, CDC_Rx.pFillBuff, g_usbHost.xferPipe[CDC_Rx.chNum].maxPackSize);
447 CDC_Rx.cdcState = USBH_CDC_GET_WAIT;
448 }
449 break;
450 /* Wait for OUT stage complete */
451 case USBH_CDC_GET_WAIT:
452 urbState = g_usbHost.xferPipe[CDC_Rx.chNum].urbStatus;
453 switch (urbState)
454 {
455 case USBH_URB_OK:
456 CDC_Rx.dataLen += g_usbHost.xferPipe[CDC_Rx.chNum].bufCount;
457 CDC_Rx.pFillBuff += g_usbHost.xferPipe[CDC_Rx.chNum].bufCount;
458 CDC_ReceiveData();
459 CDC_Rx.cdcState = USBH_CDC_COMPLETE;
460 break;
461 case USBH_URB_ERROR:
462 case USBH_URB_STALL:
463 CDC_Rx.cdcState = USBH_CDC_ERROR;
464 break;
465 case USBH_URB_NOREADY:
466 CDC_Rx.cdcState = USBH_CDC_GET_DATA;;
467 break;
468 default :
469 break;
470 }
471 break;
472 default :
473 break;
474 }
475
476 if (CDC_Rx.cdcState == USBH_CDC_ERROR)
477 {
478 CDC_Rx.cdcState = USBH_CDC_GET_DATA;
479 }
480 else if (CDC_Rx.cdcState == USBH_CDC_COMPLETE)
481 {
482 }
483
484 return CDC_Rx.cdcState;

494uint8_t USBH_CDCProcessSend(void)
495{
496 USBH_URB_STATUS_T urbState;
497 static uint16_t len;
498
499 switch (CDC_Tx.cdcState)
500 {
501 case USBH_CDC_IDIE:
502 break;
503
504 case USBH_CDC_SEND_DATA:
505 /* Sent OUT packet */
506 if (CDC_Tx.dataLen > g_usbHost.xferPipe[CDC_Tx.chNum].maxPackSize)
507 {
508 len = g_usbHost.xferPipe[CDC_Tx.chNum].maxPackSize;
509 }
510 else
511 {
512 len = CDC_Tx.dataLen;
513 }
514 USBH_BulkOutData(CDC_Tx.chNum, CDC_Tx.buff, len);
515 CDC_Tx.cdcState = USBH_CDC_SEND_WAIT;
516 break;
517 /* Wait for OUT stage complete */
518 case USBH_CDC_SEND_WAIT:
519 urbState = g_usbHost.xferPipe[CDC_Tx.chNum].urbStatus;
520 switch (urbState)
521 {
522 case USBH_URB_OK:
523 CDC_Tx.buff += len ;
524 CDC_Tx.dataLen -= len;
525
526 if (CDC_Tx.dataLen == 0)
527 {
528 CDC_Tx.cdcState = USBH_CDC_COMPLETE;
529 }
530 else
531 {
532 CDC_Tx.cdcState = USBH_CDC_SEND_DATA;
533 }
534 break;
535 case USBH_URB_ERROR:
536 case USBH_URB_STALL:
537 CDC_Tx.cdcState = USBH_CDC_ERROR;
538 break;
539 case USBH_URB_NOREADY:
540 CDC_Tx.cdcState = USBH_CDC_SEND_DATA;;
541 break;
542 default :
543 break;
544 }
545 break;
546 default :
547 break;
548 }
549
550 if (CDC_Tx.cdcState == USBH_CDC_ERROR)
551 {
552 CDC_Tx.cdcState = USBH_CDC_SEND_DATA;
553 }
554 else if (CDC_Tx.cdcState == USBH_CDC_COMPLETE)
555 {
556
557 }
558
559 return CDC_Tx.cdcState;
