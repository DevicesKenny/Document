203uint8_t I2C_Write(char * pBuffer)
204{
205 uint16_t I2CTimeout = I2CT_LONG_TIMEOUT;
206
207 while(I2C_ReadStatusFlag(I2C1, I2C_FLAG_BUSBSY))
208 {
209 I2C_Init();
210 if((I2CTimeout--) == 0) return I2C_TIMEOUT_UserCallback(4);
211 }
212
213 I2C_DisableInterrupt(I2C1, I2C_INT_EVT);
214 /* Send START condition */
215 I2C_EnableGenerateStart(I2C1);
216
217 I2CTimeout = I2CT_FLAG_TIMEOUT;
218 /* EV5 */
219 while(!I2C_ReadEventStatus(I2C1, I2C_EVENT_MASTER_MODE_SELECT))
220 {
221 if((I2CTimeout--) == 0) return I2C_TIMEOUT_UserCallback(5);
222 }
223
224 /* Send address for write */
225 I2C_Tx7BitAddress(I2C1, 0xB0, I2C_DIRECTION_TX);
226
227 I2CTimeout = I2CT_FLAG_TIMEOUT;
228 /* EV6 */
229 while(!I2C_ReadEventStatus(I2C1, I2C_EVENT_MASTER_TRANSMITTER_MODE_SELECTED))
230 {
231 if((I2CTimeout--) == 0) return I2C_TIMEOUT_UserCallback(6);
232 }
233
234 /* While there is data to be written */
235 while(*pBuffer != '\0')
236 {
237 /* Send the current byte */
238 I2C_TxData(I2C1, *pBuffer);
239
240 /* Point to the next byte to be written */
241 pBuffer++;
242
243 I2CTimeout = I2CT_LONG_TIMEOUT;
244 /* EV8 */
245 while (!I2C_ReadEventStatus(I2C1, I2C_EVENT_MASTER_BYTE_TRANSMITTING))
246 {
247 if((I2CTimeout--) == 0)
248 {
249 return I2C_TIMEOUT_UserCallback(8);
250 }
251 }
252 }
253
254 return 1;
255}
