146void HidMouse_Write(uint8_t key)
147{
148 int8_t x = 0;
149 int8_t y = 0;
150 uint8_t buffer[4] = {0, 0, 0, 0};
151
152 switch (key)
153 {
154 case HID_MOUSE_KEY_LEFT:
155 x -= 10;
156 break;
157
158 case HID_MOUSE_KEY_RIGHT:
159 x += 10;
160 break;
161
162 case HID_MOUSE_KEY_UP:
163 y -= 10;
164 break;
165
166 case HID_MOUSE_KEY_DOWN:
167 y += 10;
168 break;
169
170 default:
171 return;
172 }
173
174 buffer[1] = x;
175 buffer[2] = y;
176
177 s_statusEP = 0;
178
179 USBD_TxData(USBD_EP_1, buffer, sizeof(buffer));
180}

189void HidMouse_ReportDescriptor(USBD_DevReqData_T *reqData)
190{
191 uint8_t len;
192
193 if((reqData->byte.bRequest == USBD_GET_DESCRIPTOR) &&
194 (reqData->byte.bmRequestType.bit.recipient == USBD_RECIPIENT_INTERFACE) &&
195 (reqData->byte.bmRequestType.bit.type == USBD_REQ_TYPE_STANDARD))
196 {
197 if(reqData->byte.wValue[1] == 0x21)
198 {
199 len = USB_MIN(reqData->byte.wLength[0], 9);
200 USBD_CtrlInData((uint8_t *)&g_configDescriptor.pDesc[0x12], len);
201 }
202 else if(reqData->byte.wValue[1] == 0x22)
203 {
204 len = USB_MIN(reqData->byte.wLength[0], g_ReportDescriptor.size);
205 USBD_CtrlInData((uint8_t *)g_ReportDescriptor.pDesc, len);
206 }
207 }
208 else
209 {
210 USBD_SetEPTxRxStatus(USBD_EP_0, USBD_EP_STATUS_STALL, USBD_EP_STATUS_STALL);
211 }
212}

221uint8_t HidMouse_ReadKey(void)
222{
223 /* Right key */
224 if(!GPIO_ReadInputBit(GPIOA, GPIO_PIN_0))
225 {
226 return HID_MOUSE_KEY_RIGHT;
227 }
228
229 /* Left key */
230 if(!GPIO_ReadInputBit(GPIOA, GPIO_PIN_1))
231 {
232 return HID_MOUSE_KEY_LEFT;
233 }
234
235 return HID_MOUSE_KEY_NULL;
236}

245void HidMouse_Proc(void)
246{
247 uint8_t key = HID_MOUSE_KEY_NULL;
248
249 if(!s_usbConfigStatus)
250 {
251 return;
252 }
253
254 key = HidMouse_ReadKey();
255 if(key != HID_MOUSE_KEY_NULL)
256 {
257 if(s_statusEP)
258 {
259 HidMouse_Write(key);
260 }
261 }
262}
