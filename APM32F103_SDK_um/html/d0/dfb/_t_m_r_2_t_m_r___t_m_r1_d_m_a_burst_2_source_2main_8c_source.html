150void DMA_Init(void)
151{
152 DMA_Config_T dmaConfig;
153
154 /* Enable DMA Clock */
155 RCM_EnableAHBPeriphClock(RCM_AHB_PERIPH_DMA1);
156
157 /* DMA config */
158 dmaConfig.peripheralBaseAddr = (uint32_t)&TMR1->DMADDR;
159 dmaConfig.memoryBaseAddr = (uint32_t)ConfigBuf;
160 dmaConfig.dir = DMA_DIR_PERIPHERAL_DST;
161 dmaConfig.bufferSize = 5;
162 dmaConfig.peripheralInc = DMA_PERIPHERAL_INC_DISABLE;
163 dmaConfig.memoryInc = DMA_MEMORY_INC_ENABLE;
164 dmaConfig.peripheralDataSize = DMA_PERIPHERAL_DATA_SIZE_HALFWORD;
165 dmaConfig.memoryDataSize = DMA_MEMORY_DATA_SIZE_HALFWORD;
166 dmaConfig.loopMode = DMA_MODE_NORMAL;
167 dmaConfig.priority = DMA_PRIORITY_HIGH;
168 dmaConfig.M2M = DMA_M2MEN_DISABLE;
169
170 /* Config TMR1 UP DMA channel */
171 DMA_Config(DMA1_Channel5, &dmaConfig);
172
173 /* Enable DMA */
174 DMA_Enable(DMA1_Channel5);
175}
