73static int uart_init(void)
74{
75//#error "TODO 2: Enable the hardware uart and config baudrate."
76 GPIO_Config_T GPIO_ConfigStruct;
77 USART_Config_T USART_ConfigStruct;
78
79 /* Enable GPIO peripheral Clock */
80 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOC);
81
82 /* Enable USART peripheral Clock */
83 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_UART4);
84
85 /* Configure USART Tx as alternate function push-pull */
86 GPIO_ConfigStruct.mode = GPIO_MODE_AF_PP;
87 GPIO_ConfigStruct.pin = GPIO_PIN_10;
88 GPIO_ConfigStruct.speed = GPIO_SPEED_50MHz;
89 GPIO_Config(GPIOC, &GPIO_ConfigStruct);
90
91 /* Configure USART Rx as input floating */
92 GPIO_ConfigStruct.mode = GPIO_MODE_IN_FLOATING;
93 GPIO_ConfigStruct.pin = GPIO_PIN_11;
94 GPIO_Config(GPIOC, &GPIO_ConfigStruct);
95
96 /* USART configuration */
97 USART_ConfigStruct.baudRate = 115200;
98 USART_ConfigStruct.hardwareFlow = USART_HARDWARE_FLOW_NONE;
99 USART_ConfigStruct.mode = USART_MODE_TX_RX;
100 USART_ConfigStruct.parity = USART_PARITY_NONE;
101 USART_ConfigStruct.stopBits = USART_STOP_BIT_1;
102 USART_ConfigStruct.wordLength = USART_WORD_LEN_8B;
103 USART_Config(UART4, &USART_ConfigStruct);
104
105 /* Enable USART */
106 USART_Enable(UART4);
107
108 return 0;
109}

112void rt_hw_console_output(const char *str)
113{
114//#error "TODO 3: Output the string 'str' through the uart."
115 rt_enter_critical();
116
117 while (*str != '\0')
118 {
119 if (*str == '\n')
120 {
121 while (USART_ReadStatusFlag(UART4, USART_FLAG_TXBE) == RESET);
122
123 USART_TxData(UART4, '\r');
124 }
125
126 while (USART_ReadStatusFlag(UART4, USART_FLAG_TXBE) == RESET);
127
128 USART_TxData(UART4, *str++ );
129 }
130
131 rt_exit_critical();
132}

134char rt_hw_console_getchar(void)
135{
136 int ch = -1;
137
138 if (USART_ReadStatusFlag(UART4, USART_FLAG_RXBNE) != RESET)
139 {
140 ch = (int)USART_RxData(UART4);
141 USART_ClearStatusFlag(UART4, USART_FLAG_RXBNE);
142 }
143 else
144 {
145 if (USART_ReadStatusFlag(UART4, USART_FLAG_OVRE) != RESET)
146 {
147 USART_ClearStatusFlag(UART4, USART_FLAG_OVRE);
148 }
149
150 rt_thread_mdelay(10);
151 }
152
153 return ch;
154}
