121static void USBD_MSC_BOT_DecodeCBW(void)
122{
123 uint32_t xferCnt = g_usbDev.outBuf[MSC_OUT_EP & 0x7f].xferCnt;
124
125 g_BOTInfo.CSW.dTag = g_BOTInfo.CBW.dTag;
126 g_BOTInfo.CSW.dDataResidue = g_BOTInfo.CBW.dDataXferLen;
127
128 if ((xferCnt != MSC_BOT_CBW_LENGTH) || \
129 (g_BOTInfo.CBW.dSignature != MSC_BOT_CBW_SIGNATURE) || \
130 (g_BOTInfo.CBW.bLUN > 1) || (g_BOTInfo.CBW.bCBLen < 1) || \
131 (g_BOTInfo.CBW.bCBLen > 16))
132 {
133 SCSI_PutSenseCode(g_BOTInfo.CBW.bLUN, SCSI_SKEY_ILLEGAL_REQUEST,
134 SCSI_ASC_INVALID_CDB, 0);
135
136 g_BOTInfo.status = BOT_STATUS_ERROR;
137 }
138 else
139 {
140 if (SCSI_CmdHandler(g_BOTInfo.CBW.bLUN, g_BOTInfo.CBW.CB) != SCSI_OK)
141 {
142 USBD_MSC_BOT_Stall();
143 }
144 else if ((g_BOTInfo.state == BOT_STATE_IDLE) || \
145 (g_BOTInfo.state == BOT_STATE_SEND_DATA))
146 {
147 if (g_BOTInfo.dataLen)
148 {
149 USBD_MSC_BOT_TxData(g_BOTInfo.data, g_BOTInfo.dataLen);
150 }
151 else
152 {
153 USBD_MSC_BOT_TxCSW(BOT_CSW_STATUS_CMD_OK);
154 }
155 }
156 }

168static void USBD_MSC_BOT_TxData(uint8_t* txBuf, uint16_t len)
169{
170 len = USB_MIN(len, g_BOTInfo.CBW.dDataXferLen);
171
172 g_BOTInfo.CSW.dDataResidue -= len;
173 g_BOTInfo.CSW.bStatus = BOT_CSW_STATUS_CMD_OK;
174 g_BOTInfo.state = BOT_STATE_SEND_DATA;
175
176 USBD_TxData(MSC_IN_EP & 0x7f, txBuf, len);
177}
