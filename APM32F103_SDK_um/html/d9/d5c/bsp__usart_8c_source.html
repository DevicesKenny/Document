133void USART_Init(uint32_t baudRate)
134{
135 GPIO_Config_T gpioConfig;
136 USART_Config_T usartConfigStruct;
137
138 /* Enable GPIO clock */
139 RCM_EnableAPB2PeriphClock(DEBUG_USART_TX_CLK);
140 RCM_EnableAPB2PeriphClock(DEBUG_USART_RX_CLK);
141
142 /* Enable USART clock */
143 RCM_EnableAPB2PeriphClock(DEBUG_USART_CLK);
144
145 /* Configure USART Tx as alternate function push-pull */
146 gpioConfig.mode = GPIO_MODE_AF_PP;
147 gpioConfig.pin = DEBUG_USART_TX_PIN;
148 gpioConfig.speed = GPIO_SPEED_50MHz;
149 GPIO_Config(DEBUG_USART_TX_PORT, &gpioConfig);
150
151 /* Configure USART Rx as input floating */
152 gpioConfig.mode = GPIO_MODE_IN_FLOATING;
153 gpioConfig.pin = DEBUG_USART_RX_PIN;
154 GPIO_Config(DEBUG_USART_RX_PORT, &gpioConfig);
155
156 /*
157 MINI_USARTs configured as follow:
158 BaudRate = baudRate
159 Word Length = 8 Bits
160 One Stop Bit
161 No parity
162 Hardware flow control disabled (RTS and CTS signals)
163 Receive and transmit enabled
164 */
165 usartConfigStruct.baudRate = baudRate;
166 usartConfigStruct.mode = USART_MODE_TX_RX;
167 usartConfigStruct.hardwareFlow = USART_HARDWARE_FLOW_NONE;
168 usartConfigStruct.parity = USART_PARITY_NONE;
169 usartConfigStruct.stopBits = USART_STOP_BIT_1;
170 usartConfigStruct.wordLength = USART_WORD_LEN_8B;
171 USART_Config(DEBUG_USART, &usartConfigStruct);
172
173 /* Enable USART */
174 USART_Enable(DEBUG_USART);
175}
