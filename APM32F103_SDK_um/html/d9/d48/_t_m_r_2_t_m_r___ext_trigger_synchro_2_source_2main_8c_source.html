52 GPIO_Config_T GPIO_ConfigStruct;

57 /* Enable peripheral clock */

59 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_TMR2 | RCM_APB1_PERIPH_TMR3);
60 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA);
61
62 /* Configure TMR1, TMR2, TMR3 channel 1 GPIO */
63 GPIO_ConfigStruct.pin = GPIO_PIN_0 | GPIO_PIN_6 | GPIO_PIN_8 | GPIO_PIN_9;
64 GPIO_ConfigStruct.mode = GPIO_MODE_AF_PP;
65 GPIO_ConfigStruct.speed = GPIO_SPEED_50MHz;
66 GPIO_Config(GPIOA, &GPIO_ConfigStruct);
67
68 /* TMR1 Input Capture mode */
69 ICConfig.channel = TMR_CHANNEL_2;
70 ICConfig.polarity = TMR_IC_POLARITY_RISING;
71 ICConfig.selection = TMR_IC_SELECTION_DIRECT_TI;
72 ICConfig.prescaler = TMR_IC_PSC_1;
73 ICConfig.filter = 0x00;
74 TMR_ConfigIC(TMR1, &ICConfig);
75
76 /* config TMR1,TMR2,TMR3 */
77 TMR_TimeBaseStruct.clockDivision = TMR_CLOCK_DIV_1;
78 TMR_TimeBaseStruct.countMode = TMR_COUNTER_MODE_UP;
79 TMR_TimeBaseStruct.division = 0;
80 TMR_TimeBaseStruct.repetitionCounter = 0;
81
82 /* TMR1 frequency = 72MHz / (71 + 1) = 1MHz */
83 TMR_TimeBaseStruct.period = 71;
84 TMR_ConfigTimeBase(TMR1, &TMR_TimeBaseStruct);
85
86 /* TMR2 frequency = 48MHz / (95 + 1) = 500KHz */
87 TMR_TimeBaseStruct.period = 95;
88 TMR_ConfigTimeBase(TMR2, &TMR_TimeBaseStruct);
89
90 /* TMR3 frequency = 48MHz / (191 + 1) = 250KHz */
91 TMR_TimeBaseStruct.period = 191;
92 TMR_ConfigTimeBase(TMR3, &TMR_TimeBaseStruct);
93
94 TMR_ConfigOCStructInit(&OCcongigStruct);
95
96 /* Configure TMR1, TMR2, TMR3 channel 1 */
97 OCcongigStruct.mode = TMR_OC_MODE_TOGGEL;
98 OCcongigStruct.outputState = TMR_OC_STATE_ENABLE;
99 OCcongigStruct.polarity = TMR_OC_POLARITY_HIGH;
100 OCcongigStruct.idleState = TMR_OC_IDLE_STATE_SET;
101 OCcongigStruct.outputNState = TMR_OC_NSTATE_DISABLE;
102 OCcongigStruct.nPolarity = TMR_OC_NPOLARITY_HIGH;
103 OCcongigStruct.nIdleState = TMR_OC_NIDLE_STATE_RESET;
104
105 TMR_ConfigOC1(TMR1, &OCcongigStruct);
106 TMR_ConfigOC1(TMR2, &OCcongigStruct);
107 TMR_ConfigOC1(TMR3, &OCcongigStruct);
108
109 /* Selects the Slave Mode and Input Trigger source */
110 TMR_SelectSlaveMode(TMR1, TMR_SLAVE_MODE_GATED);
111 TMR_SelectInputTrigger(TMR1, TMR_TRIGGER_SOURCE_TI2FP2);
112
113 /* Enable TMR1 the Master Slave Mode */
114 TMR_EnableMasterSlaveMode(TMR1);
115 TMR_SelectOutputTrigger(TMR1, TMR_TRGO_SOURCE_ENABLE);
116
117 /* Selects the Slave Mode and Input Trigger source */
118 TMR_SelectSlaveMode(TMR2, TMR_SLAVE_MODE_GATED);
119 TMR_SelectInputTrigger(TMR2, TMR_TRIGGER_SOURCE_ITR0);
120
121 /* Enable TMR2 the Master Slave Mode */
122 TMR_EnableMasterSlaveMode(TMR2);
123 TMR_SelectOutputTrigger(TMR2, TMR_TRGO_SOURCE_ENABLE);
124
125 /* Selects the Slave Mode and Input Trigger source */
126 TMR_SelectSlaveMode(TMR3, TMR_SLAVE_MODE_GATED);
127 TMR_SelectInputTrigger(TMR3, TMR_TRIGGER_SOURCE_ITR1);
128
129 /* Enable the specified TMR peripheral */
130 TMR_Enable(TMR1);
131 TMR_Enable(TMR2);
132 TMR_Enable(TMR3);
133
134 TMR_EnablePWMOutputs(TMR1);
135
136 while(1)
137 {
138 }
139}
