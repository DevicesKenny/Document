130void TMR2_IRQHandler(void)
131{
132 uint32_t capValue = 0;
133
134 /* TMR2_CH1 toggling with frequency = 2.5KHz */
135 if (TMR_ReadIntFlag(TMR2, TMR_INT_CC1))
136 {
137 /* Clear the TMR2 CC1 pending bit */
138 TMR_ClearIntFlag(TMR2, TMR_INT_CC1);
139 capValue = TMR_ReadCaputer1(TMR2);
140 TMR_ConfigCompare1(TMR2, capValue + CC1_Val);
141 }
142
143 /* TMR2_CH2 toggling with frequency = 5KHz */
144 if (TMR_ReadIntFlag(TMR2, TMR_INT_CC2))
145 {
146 /* Clear the TMR2 CC2 pending bit */
147 TMR_ClearIntFlag(TMR2, TMR_INT_CC2);
148 capValue = TMR_ReadCaputer2(TMR2);
149 TMR_ConfigCompare2(TMR2, capValue + CC2_Val);
150 }
151
152 /* TMR2_CH3 toggling with frequency = 25KHz */
153 if (TMR_ReadIntFlag(TMR2, TMR_INT_CC3))
154 {
155 /* Clear the TMR2 CC3 pending bit */
156 TMR_ClearIntFlag(TMR2, TMR_INT_CC3);
157 capValue = TMR_ReadCaputer3(TMR2);
158 TMR_ConfigCompare3(TMR2, capValue + CC3_Val);
159 }
160
161 /* TMR2_CH4 toggling with frequency = 50KHz */
162 if (TMR_ReadIntFlag(TMR2, TMR_INT_CC4))
163 {
164 /* Clear the TMR2 CC4 pending bit */
165 TMR_ClearIntFlag(TMR2, TMR_INT_CC4);
166 capValue = TMR_ReadCaputer4(TMR2);
167 TMR_ConfigCompare4(TMR2, capValue + CC4_Val);
168 }
169}
