175void EINT1_IRQHandler(void)
176{
177 if(EINT_ReadIntFlag(KEY1_BUTTON_EINT_LINE) != RESET)
178 {
179 EINT_ClearIntFlag(KEY1_BUTTON_EINT_LINE);
180 APM_MINI_LEDOn(LED2);
181
182 RTC_ClearStatusFlag(RTC_FLAG_SEC);
183 while(RTC_ReadStatusFlag(RTC_FLAG_SEC) == RESET);
184
185 RTC_ConfigAlarm(RTC_ReadCounter()+ 3);
186 RTC_WaitForLastTask();
187
188 PMU_EnterSTANDBYMode();
189 }
190}
