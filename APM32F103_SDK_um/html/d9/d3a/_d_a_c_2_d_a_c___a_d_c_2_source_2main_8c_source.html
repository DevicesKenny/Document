147void ADC_Init(void)
148{
149 GPIO_Config_T gpioConfig;
150 ADC_Config_T adcConfig;
151
152 /* Enable GPIOA clock */
153 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA);
154
155 /* ADC channel 0 configuration */
156 GPIO_ConfigStructInit(&gpioConfig);
157 gpioConfig.mode = GPIO_MODE_ANALOG;
158 gpioConfig.speed = GPIO_SPEED_50MHz;
159 gpioConfig.pin = GPIO_PIN_0;
160 GPIO_Config(GPIOA, &gpioConfig);
161
162 /* Enable ADC clock */
163 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_ADC1);
164
165 /* ADC configuration */
166 ADC_Reset(ADC1);
167 ADC_ConfigStructInit(&adcConfig);
168 adcConfig.continuosConvMode = ENABLE;
169 adcConfig.dataAlign = ADC_DATA_ALIGN_RIGHT;
170 adcConfig.externalTrigConv = ADC_EXT_TRIG_CONV_None;
171 adcConfig.scanConvMode = DISABLE;
172 ADC_Config(ADC1, &adcConfig);
173
174 /* ADC channel 10 Convert configuration */
175 ADC_ConfigRegularChannel(ADC1, ADC_CHANNEL_0, 1, ADC_SAMPLETIME_239CYCLES5);
176
177 /* Enable complete conversion interupt */
178 ADC_EnableInterrupt(ADC1, ADC_INT_EOC);
179
180 /* NVIC configuration */
181 NVIC_EnableIRQRequest(ADC1_2_IRQn, 1, 1);
182
183 /* Enable ADC */
184 ADC_Enable(ADC1);
185
186 /* ADC start conversion */
187 ADC_EnableSoftwareStartConv(ADC1);
188}

197void ADC_Isr(void)
198{
199 uint16_t adcData = 0;
200 uint16_t voltage = 0;
201
202 /* generate analog signal*/
203 DAC_ConfigChannel1Data(DAC_ALIGN_12BIT_R, dacData);
204
205 if (dacData >= 0x0FFF)
206 {
207 dacData = 0;
208 }
209
210 dacData += 0x7F;
211
212 if (ADC_ReadStatusFlag(ADC1, ADC_FLAG_EOC))
213 {
214 adcData = ADC_ReadConversionValue(ADC1);
215
216 voltage = (adcData * 3300) / 4095;
217 printf("\r\n voltage : %d mV\r\n", voltage);
218
219 /* voltage greater than 2000mv */
220 if (voltage > 2000)
221 {
222 APM_MINI_LEDOn(LED2);
223 APM_MINI_LEDOff(LED3);
224 }
225 /* voltage less than 800mv */
226 else if (voltage < 800)
227 {
228 APM_MINI_LEDOn(LED3);
229 APM_MINI_LEDOff(LED2);
230 }
231 /* 800mv <= voltage <= 2000 */
232 else
233 {
234 APM_MINI_LEDOn(LED2);
235 APM_MINI_LEDOn(LED3);
236 }
237
238 Delay(0x2FFFFF);
239
240 ADC_ClearStatusFlag(ADC1, ADC_FLAG_EOC);
241 }
242}
