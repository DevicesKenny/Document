79static void USBD_MSC_SetConfigCallBack(void)

101static void USBD_MSC_ClearFeatureHandler(void)
102{
103 uint8_t ep = g_usbDev.reqData.domain.wIndex[0] & 0x7f;
104
105 if (g_usbDev.reqData.domain.wValue[0] == USBD_FEATURE_ENDPOINT_HALT)
106 {
107 if (g_usbDev.reqData.domain.wIndex[0] & 0x80)
108 {
109 USB_OTG_FlushTxFIFO(ep);
110 }
111 else
112 {
113 USB_OTG_FlushRxFIFO();
114 }
115
116 USBD_MSV_BOT_ClearFeatureHandler();
117 }
118}
