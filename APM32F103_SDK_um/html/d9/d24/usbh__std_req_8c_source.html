410void USBH_ParseCfgDesc(uint8_t* parseBuf,
411 USBH_ConfigurationDesc_T* cfgDescBuf,
412 USBH_Interface_T* interface)
413{
414 uint16_t parseIndex = 0;
415 uint8_t itfIndex = 0;
416 uint8_t epIndex = 0;
417 uint16_t totalLen = ((uint16_t)parseBuf[3] << 8) | (uint16_t)parseBuf[2];
418 uint8_t subLen = 0;
419
420
421 if (cfgDescBuf != NULL)
422 {
423 /* Parse Configuration Descriptor */
424 USBH_CopyBuffer((uint8_t*)cfgDescBuf, parseBuf, STD_CFG_DESC_SIZE);
425 }
426
427 parseIndex = STD_CFG_DESC_SIZE;
428
429 while (totalLen > parseIndex)
430 {
431 /* Get descriptor length at first byte */
432 subLen = parseBuf[parseIndex];
433
434 /* Check the descriptor Type at second byte */
435 switch (parseBuf[parseIndex + 1])
436 {
437 case USBH_DESC_INTERFACE:
438 if (itfIndex < INTERFACE_DESC_MAX_NUM)
439 {
440 /* Move data from parse Buffer to Interface descriptor */
441 USBH_CopyBuffer((uint8_t*)&interface[itfIndex].itfDesc,
442 &parseBuf[parseIndex],
443 STD_INTERFACE_DESC_SIZE);
444 itfIndex++;
445 epIndex = 0;
446 }
447 break;
448 case USBH_DESC_ENDPOINT:
449 if ((0 < itfIndex) && (itfIndex < INTERFACE_DESC_MAX_NUM))
450 {
451 if (epIndex < ENDPOINT_DESC_MAX_NUM)
452 {
453 /* Move data from parse Buffer to Endpoint descriptor */
454 USBH_CopyBuffer((uint8_t*)&interface[itfIndex - 1].epDesc[epIndex],
455 &parseBuf[parseIndex],
456 STD_EP_DESC_SIZE);
457
458 epIndex++;
459 }
460
461 }
462 break;
463 default :
464 break;
465 }
466
467 parseIndex += subLen;
468
469 /* To avoid some useless data left */
470 if ((totalLen - parseIndex) < STD_EP_DESC_SIZE)
471 break;
472 }
473}

484void USBH_ParseStrDescriptor(uint8_t* DescBuff, uint8_t* strBuff)
485{
486 uint16_t index;
487
488 switch (DescBuff[1])
489 {
490 case 0X03:
492 for (index = 2; index < DescBuff[0]; index += 2)
493 {
494 /* get string */
495 *strBuff = DescBuff[index];
496 strBuff++;
497 }
498 /* string end mark*/
499 *strBuff = 0x00;
500 break;
501
502 default:
503 break;
504 }
