228static void USER_CfgDescHandler(void *cfgDesc, void *interface)
229{
230 USBH_Interface_T *itf = interface;
231 uint8_t i;
232
233 switch (itf->itfDesc.bInterfaceClass)
234 {
235 case 0x08:
236 printf(">> This is a Mass Storage Class Device.\r\n");
237 break;
238 case 0x03:
239 printf(">> This is a HID Class Device.\r\n");
240 break;
241 case 0x0A:
242 printf(">> This is a Communications Class Device.\r\n");
243 break;
244 default :
245 printf(">> Class Code is : 0x%02X\r\n", itf->itfDesc.bInterfaceClass);
246 break;
247 }
248 printf(">> Use %d endpoint:\r\n", itf->itfDesc.bNumEndpoints);
249 for (i = 0; i < itf->itfDesc.bNumEndpoints; i++)
250 {
251 printf(" Endpoint 0x%02X: max pack size is %d bytes\r\n", \
252 itf->epDesc[i].bEndpointAddress,
253 itf->epDesc[i].wMaxPacketSize[0] | itf->epDesc[i].wMaxPacketSize[1] << 8);
254 }
255}
