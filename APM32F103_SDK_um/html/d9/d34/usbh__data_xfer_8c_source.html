484static USBH_CTRL_STATE_T USBH_CtrlXferWaitUrbState(USBH_HostReqData_T* reqData, uint8_t* buf)
485{
486 uint16_t len = (uint16_t)reqData->domain.wLength[1] << 8 | \
487 ((uint16_t)reqData->domain.wLength[0]);
488 uint8_t chNum = ((s_preXferState == USBH_CTRL_XFER_DATA_IN) || (s_preXferState == USBH_CTRL_XFER_STATUS_IN)) ? \
489 g_usbHost.control.inChannelNum : g_usbHost.control.outChannelNum;
490
491 switch (g_usbHost.xferPipe[chNum].urbStatus)
492 {
493 case USBH_URB_OK:
494 switch (s_preXferState)
495 {
496 case USBH_CTRL_XFER_SETUP:
497 /* Device to Host */
498 if (reqData->domain.bmRequestType.bit.dir == USBH_REQ_DIR_IN)
499 {
500 s_curXferState = len ? USBH_CTRL_XFER_DATA_IN : USBH_CTRL_XFER_STATUS_OUT;
501 }
502 /* Host to Device */
503 else
504 {
505 s_curXferState = len ? USBH_CTRL_XFER_DATA_OUT : USBH_CTRL_XFER_STATUS_IN;
506 }
507 break;
508 case USBH_CTRL_XFER_DATA_OUT:
509 s_curXferState = USBH_CTRL_XFER_STATUS_IN;
510 break;
511 case USBH_CTRL_XFER_DATA_IN:
512 s_curXferState = USBH_CTRL_XFER_STATUS_OUT;
513 break;
514 case USBH_CTRL_XFER_STATUS_OUT:
515 case USBH_CTRL_XFER_STATUS_IN:
516 s_curXferState = USBH_CTRL_XFER_SETUP;
517 g_usbHost.control.ctrlState = USBH_CTRL_COMPLETE;
518 break;
519 default :
520 break;
521 }
522 break;
523 case USBH_URB_ERROR:
524 if (g_usbHost.control.errCnt++ < CTRL_XFER_ERROR_CNT)
525 {
526 s_curXferState = USBH_CTRL_XFER_SETUP;
527 }
528 else
529 {
530 g_usbHost.control.ctrlState = USBH_CTRL_ERROR;
531 }
532
533 case USBH_URB_STALL:
534 g_usbHost.control.ctrlState = USBH_CTRL_STALL;
535 s_curXferState = USBH_CTRL_XFER_SETUP;
536 break;
537 case USBH_URB_NOREADY:
538 /* If OUT data is not received */
539 if (s_preXferState == USBH_CTRL_XFER_DATA_OUT)
540 {
541 s_curXferState = USBH_CTRL_XFER_DATA_OUT;
542 }
543 default :
544 break;
545 }
546
547 return g_usbHost.control.ctrlState;
548}
