56/*
57 * Create a new semaphore
58 *
59 * @return the operation status, ERR_OK on OK; others on error
60 */
61err_t sys_sem_new(sys_sem_t *sem, u8_t count)
62{
63 static unsigned short counter = 0;
64 char tname[RT_NAME_MAX];
65 sys_sem_t tmpsem;
66
67 RT_DEBUG_NOT_IN_INTERRUPT;
68
69 rt_snprintf(tname, RT_NAME_MAX, "%s%d", SYS_LWIP_SEM_NAME, counter);
70 counter ++;
71
72 tmpsem = rt_sem_create(tname, count, RT_IPC_FLAG_FIFO);
73 if (tmpsem == RT_NULL)
74 return ERR_MEM;
75 else
76 {
77 *sem = tmpsem;
78
79 return ERR_OK;
80 }
81}

100/*
101 * Block the thread while waiting for the semaphore to be signaled
102 *
103 * @return If the timeout argument is non-zero, it will return the number of milliseconds
104 * spent waiting for the semaphore to be signaled; If the semaphore isn't signaled
105 * within the specified time, it will return SYS_ARCH_TIMEOUT; If the thread doesn't
106 * wait for the semaphore, it will return zero
107 */
108u32_t sys_arch_sem_wait(sys_sem_t *sem, u32_t timeout)
109{
110 rt_err_t ret;
111 s32_t t;
112 u32_t tick;
113
114 RT_DEBUG_NOT_IN_INTERRUPT;
115
116 /* get the begin tick */
117 tick = rt_tick_get();
118 if (timeout == 0)
119 t = RT_WAITING_FOREVER;
120 else
121 {
122 /* convert msecond to os tick */
123 if (timeout < (1000/RT_TICK_PER_SECOND))
124 t = 1;
125 else
126 t = timeout / (1000/RT_TICK_PER_SECOND);
127 }
128
129 ret = rt_sem_take(*sem, t);
130
131 if (ret == -RT_ETIMEOUT)
132 return SYS_ARCH_TIMEOUT;
133 else
134 {
135 if (ret == RT_EOK)
136 ret = 1;
137 }
138
139 /* get elapse msecond */
140 tick = rt_tick_get() - tick;
141
142 /* convert tick to msecond */
143 tick = tick * (1000 / RT_TICK_PER_SECOND);
144 if (tick == 0)
145 tick = 1;
146
147 return tick;
148}

300/*
301 * Try to post the "msg" to the mailbox
302 *
303 * @return return ERR_OK if the "msg" is posted, ERR_MEM if the mailbox is full
304 */
305err_t sys_mbox_trypost(sys_mbox_t *mbox, void *msg)
306{
307 if (rt_mb_send(*mbox, (rt_uint32_t)msg) == RT_EOK)
308 return ERR_OK;
309
310 return ERR_MEM;

321u32_t sys_arch_mbox_fetch(sys_mbox_t *mbox, void **msg, u32_t timeout)
322{
323 rt_err_t ret;
324 s32_t t;
325 u32_t tick;
326
327 RT_DEBUG_NOT_IN_INTERRUPT;
328
329 /* get the begin tick */
330 tick = rt_tick_get();
331
332 if(timeout == 0)
333 t = RT_WAITING_FOREVER;
334 else
335 {
336 /* convirt msecond to os tick */
337 if (timeout < (1000/RT_TICK_PER_SECOND))
338 t = 1;
339 else
340 t = timeout / (1000/RT_TICK_PER_SECOND);
341 }
342
343 ret = rt_mb_recv(*mbox, (rt_ubase_t *)msg, t);
344 if(ret == -RT_ETIMEOUT)
345 {
346 return SYS_ARCH_TIMEOUT;
347 }
348 else
349 {
350 LWIP_ASSERT("rt_mb_recv returned with error!", ret == RT_EOK);
351 }
352
353 /* get elapse msecond */
354 tick = rt_tick_get() - tick;
355
356 /* convert tick to msecond */
357 tick = tick * (1000 / RT_TICK_PER_SECOND);
358 if (tick == 0)
359 tick = 1;
360
361 return tick;
362}

410/*
411 * Start a new thread named "name" with priority "prio" that will begin
412 * its execution in the function "thread()". The "arg" argument will be
413 * passed as an argument to the thread() function
414 */
415sys_thread_t sys_thread_new(const char *name,
416 lwip_thread_fn thread,
417 void *arg,
418 int stacksize,
419 int prio)
420{
421 rt_thread_t t;
422
423 RT_DEBUG_NOT_IN_INTERRUPT;
424
425 /* create thread */
426 t = rt_thread_create(name, thread, arg, stacksize, prio, 20);
427 RT_ASSERT(t != RT_NULL);
428
429 /* startup thread */
430 rt_thread_startup(t);
431
432 return t;
433}
