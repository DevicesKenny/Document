116void RTC_Init(void)
117{
118 RCM_EnableAPB1PeriphClock((RCM_APB1_PERIPH_T)RCM_APB1_PERIPH_PMU);
119 PMU_EnableBackupAccess();
120
121 /* LSI Enable */
122 RCM_EnableLSI();
123 while (RCM_ReadStatusFlag(RCM_FLAG_LSIRDY) == RESET);
124 RCM_ConfigRTCCLK(RCM_RTCCLK_LSI);
125 RCM_EnableRTCCLK();
126
127 RTC_WaitForSynchro();
128 RTC_WaitForLastTask();
129
130 /* ALARM INTERRUPT Enable */
131 RTC_EnableInterrupt(RTC_INT_ALR);
132 RTC_WaitForLastTask();
133
134 RTC_ConfigPrescaler(32767);
135 RTC_WaitForLastTask();
136
137 /* 5 s ALARM */
138 RTC_ConfigAlarm(5);
139 RTC_WaitForLastTask();
140
141 /* COUNT SET 0 */
142 RTC_ConfigCounter(0);
143 RTC_WaitForLastTask();
144
145 NVIC_EnableIRQRequest(RTC_IRQn, 0, 0);
146}
