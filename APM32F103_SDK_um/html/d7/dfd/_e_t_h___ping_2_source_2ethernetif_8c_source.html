126static void
127low_level_init(struct netif *netif)
128{
129 int i;
130 /* set MAC hardware address length */
131 netif->hwaddr_len = ETHARP_HWADDR_LEN;
132
133 /* set MAC hardware address */
134 netif->hwaddr[0] = SetMACaddr[0];
135 netif->hwaddr[1] = SetMACaddr[1];
136 netif->hwaddr[2] = SetMACaddr[2];
137 netif->hwaddr[3] = SetMACaddr[3];
138 netif->hwaddr[4] = SetMACaddr[4];
139 netif->hwaddr[5] = SetMACaddr[5];
140
141 /* maximum transfer unit */
142 netif->mtu = 1500;
143
144 /* device capabilities */
145 /* don't set NETIF_FLAG_ETHARP if this device is not an ethernet one */
146 netif->flags = NETIF_FLAG_BROADCAST | NETIF_FLAG_ETHARP | NETIF_FLAG_LINK_UP;
147
148 /* Initialize Tx Descriptors list: Chain Mode */
149 ETH_ConfigDMATxDescChain(DMATxDscrTab, &Tx_Buff[0][0], ETH_TXBUFNB);
150 /* Initialize Rx Descriptors list: Chain Mode */
151 ETH_ConfigDMARxDescChain(DMARxDscrTab, &Rx_Buff[0][0], ETH_RXBUFNB);
152
153 /* Enable Ethernet Rx interrrupt */
154 for(i=0; i<ETH_RXBUFNB; i++)
155 {
156 ETH_EnableDMARxDescReceiveInterrupt(&DMARxDscrTab[i]);
157 }
158
159#ifdef HARDWARE_CHECKSUM
160 /* Enable the checksum insertion for the Tx frames */
161 for(i=0; i<ETH_TXBUFNB; i++)
162 {
163 ETH_ConfigDMATxDescChecksumInsertion(&DMATxDscrTab[i], ETH_DMATXDESC_CHECKSUMTCPUDPICMPFULL);
164 }
165#endif
166
167 /* Enable MAC and DMA transmission and reception */
168 ETH_Start();
169
170}

188static err_t
189low_level_output(struct netif *netif, struct pbuf *p)
190{
191 struct pbuf *q;
192 int l = 0;
193 u8 *buffer = (u8 *)(DMATxDescToSet->Buffer1Addr);
194
195 for(q = p; q != NULL; q = q->next)
196 {
197 memcpy((u8_t*)&buffer[l], q->payload, q->len);
198 l = l + q->len;
199 }
200
201 ETH_TxPacket(l);
202
203 return ERR_OK;
204}

214static struct pbuf *
215low_level_input(struct netif *netif)
216{
217 struct pbuf *p, *q;
218 u16_t len;
219 int l =0;
220 ETH_Frame_T frame;
221 u8 *buffer;
222
223 p = NULL;
224 frame = ETH_RxPacket();
225 /* Obtain the size of the packet and put it into the "len"
226 variable. */
227 len = frame.length;
228
229 buffer = (u8 *)frame.buffer;
230
231 /* We allocate a pbuf chain of pbufs from the pool. */
232 p = pbuf_alloc(PBUF_RAW, len, PBUF_POOL);
233
234 if (p != NULL)
235 {
236 for (q = p; q != NULL; q = q->next)
237 {
238 memcpy((u8_t*)q->payload, (u8_t*)&buffer[l], q->len);
239 l = l + q->len;
240 }
241 }
242
244 frame.descriptor->Status = ETH_DMARXDESC_OWN;
245
247 if ((ETH->DMASTS & BIT7) != (u32)RESET)
248 {
250 ETH->DMASTS = BIT7;
252 ETH->DMARXPD = 0;
253 }
254
255 return p;
256}

267err_t
268ethernetif_input(struct netif *netif)
269{
270 err_t err;
271 struct pbuf *p;
272
273 /* move received packet into a new pbuf */
274 p = low_level_input(netif);
275
276 /* no packet could be read, silently ignore this */
277 if (p == NULL) return ERR_MEM;
278
279 err = netif->input(p, netif);
280 if (err != ERR_OK)
281 {
282 LWIP_DEBUGF(NETIF_DEBUG, ("ethernetif_input: IP input error\n"));
283 pbuf_free(p);
284 p = NULL;
285 }
286
287 return err;
288}

302err_t
303ethernetif_init(struct netif *netif)
304{
305 struct ethernetif *ethernetif;
306
307 LWIP_ASSERT("netif != NULL", (netif != NULL));
308
309 ethernetif = mem_malloc(sizeof(struct ethernetif));
310 if (ethernetif == NULL)
311 {
312 LWIP_DEBUGF(NETIF_DEBUG, ("ethernetif_init: out of memory\n"));
313 return ERR_MEM;
314 }
315
316#if LWIP_NETIF_HOSTNAME
317 /* Initialize interface hostname */
318 netif->hostname = "lwip";
319#endif /* LWIP_NETIF_HOSTNAME */
320
321 /*
322 * Initialize the snmp variables and counters inside the struct netif.
323 * The last argument should be replaced with your link speed, in units
324 * of bits per second.
325 */
326 NETIF_INIT_SNMP(netif, snmp_ifType_ethernet_csmacd, 100000000);
327
328 netif->state = ethernetif;
329 netif->name[0] = IFNAME0;
330 netif->name[1] = IFNAME1;
331 /* We directly use etharp_output() here to save a function call.
332 * You can instead declare your own function an call etharp_output()
333 * from it if you have to do some checks before sending (e.g. if link
334 * is available...) */
335 netif->output = etharp_output;
336 netif->linkoutput = low_level_output;
337
338 ethernetif->ethaddr = (struct eth_addr *)&(netif->hwaddr[0]);
339
340 /* initialize the hardware */
341 low_level_init(netif);
342
343 return ERR_OK;
344}

355ETH_Frame_T ETH_RxPacket(void)
356{
357 u32 ethframelength = 0;
358 ETH_Frame_T ethframe = {0,0};
359
360 /* Check if the descriptor is owned by the ETHERNET DMA (when set) or CPU (when reset) */
361 if((DMARxDescToGet->Status & ETH_DMARXDESC_OWN) != (u32)RESET)
362 {
363 ethframe.length = ETH_ERROR;
364
365 if ((ETH->DMASTS & BIT7) != (u32)RESET)
366 {
367 /* Clear RBUS ETHERNET DMA flag */
368 ETH->DMASTS = BIT7;
369 /* Resume DMA reception */
370 ETH->DMARXPD = 0;
371 }
372
373 /* Return error: OWN bit set */
374 return ethframe;
375 }
376
377 if(((DMARxDescToGet->Status & ETH_DMARXDESC_ERRS) == (u32)RESET) &&
378 ((DMARxDescToGet->Status & ETH_DMARXDESC_LDES) != (u32)RESET) &&
379 ((DMARxDescToGet->Status & ETH_DMARXDESC_FDES) != (u32)RESET))
380 {
381 /* Get the Frame Length of the received packet: substruct 4 bytes of the CRC */
382 ethframelength = ((DMARxDescToGet->Status & ETH_DMARXDESC_FL) >> 16) - 4;
383
384 /* Get the addrees of the actual buffer */
385 ethframe.buffer = DMARxDescToGet->Buffer1Addr;
386 }
387 else
388 {
389 /* Return ERROR */
390 ethframelength = ETH_ERROR;
391 }
392
393 ethframe.length = ethframelength;
394 ethframe.descriptor = DMARxDescToGet;
395
396 /* Selects the next DMA Rx descriptor list for next buffer to read */
397 DMARxDescToGet = (ETH_DMADescConfig_T*) (DMARxDescToGet->Buffer2NextDescAddr);
398
399 /* Return Frame */
400 return ethframe;
401}

413int ETH_TxPacket(u16 Length)
414{
416 if((DMATxDescToSet->Status & ETH_DMATXDESC_OWN) != (u32)RESET)
417 {
419 return ETH_ERROR;
420 }
421
423 DMATxDescToSet->ControlBufferSize = (Length & ETH_DMATXDESC_TXBS1);
424
426 DMATxDescToSet->Status |= ETH_DMATXDESC_LS | ETH_DMATXDESC_FS;
427
429 DMATxDescToSet->Status |= ETH_DMATXDESC_OWN;
430
432 if ((ETH->DMASTS & BIT2) != (u32)RESET)
433 {
435 ETH->DMASTS = BIT2;
437 ETH->DMATXPD = 0;
438 }
439
443 DMATxDescToSet = (ETH_DMADescConfig_T*) (DMATxDescToSet->Buffer2NextDescAddr);
444
446 return ETH_SUCCESS;
447}
