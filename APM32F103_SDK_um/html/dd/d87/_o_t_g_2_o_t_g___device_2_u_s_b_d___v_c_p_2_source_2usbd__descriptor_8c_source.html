337void USBD_ConfigSerialString(uint8_t *string)
338{
339 uint32_t serialNum1 = 0;
340 uint16_t serialNum2 = 0;
341 uint8_t index, movCnt = 0;
342 serialNum1 = (*(uint32_t *)UID_ADDRESS1) + (*(uint32_t *)UID_ADDRESS2);
343 serialNum2 = (uint16_t)((*(uint32_t *)UID_ADDRESS1) + (*(uint32_t *)UID_ADDRESS3));
344 for (index = 2; index <= 16; index += 2)
345 {
346 string[index] = USBD_NumberToChar((uint8_t)(serialNum1 >> movCnt) & 0x0F);
347 if (movCnt < 16)
348 {
349 string[index + 16] = USBD_NumberToChar(((uint8_t)(serialNum2 >> movCnt) & 0x0F));
350 }
351 movCnt += 4;
352 }
353}
