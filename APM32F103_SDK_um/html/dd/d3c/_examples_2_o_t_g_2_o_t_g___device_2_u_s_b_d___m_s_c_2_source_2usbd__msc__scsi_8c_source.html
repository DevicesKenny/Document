123uint8_t SCSI_CmdHandler(uint8_t lun, uint8_t* command)
124{
125 uint8_t ret = SCSI_OK;
126
127 switch (command[0])
128 {
129 case SCSI_CMD_TEST_UNIT_READY:
130 ret = SCSI_TestUnitReady(lun);
131 break;
132
133 case SCSI_CMD_INQUIRY:
134 ret = SCSI_Inquiry(lun, command);
135 break;
136
137 case SCSI_CMD_REQUEST_SENSE:
138 ret = SCSI_RequestSense(lun, command);
139 break;
140
141 case SCSI_CMD_READ_FORMAT_CAPACITIES:
142 ret = SCSI_ReadFormatCapacity(lun, command);
143 break;
144
145 case SCSI_CMD_READ_CAPACITY_10:
146 ret = SCSI_ReadCapacity10(lun, command);
147 break;
148
149 case SCSI_CMD_READ_10:
150 ret = SCSI_Read10(lun, command);
151 break;
152
153 case SCSI_CMD_WRITE10:
154 ret = SCSI_Write10(lun, command);
155 break;
156
157 case SCSI_CMD_VERIFY_10:
158 ret = SCSI_Verify10(lun, command);
159 break;
160 case SCSI_CMD_ALLOW_MEDIUM_REMOVAL:
161 case SCSI_CMD_START_STOP_UNIT:
162 ret = SCSI_StartStopUnit();
163 break;
164
165 case SCSI_CMD_MODE_SENSE_6:
166 ret = SCSI_ModeSense6 (lun, command);
167 break;
168
169 case SCSI_CMD_MODE_SENSE_10:
170 ret = SCSI_ModeSense10 (lun, command);
171 break;
172
173 default:
174 SCSI_PutSenseCode(lun, SCSI_SKEY_ILLEGAL_REQUEST,
175 SCSI_ASC_INVALID_CDB, 0);
176 ret = SCSI_FAIL;
177 }
178
179 return ret;
180}

244static uint8_t SCSI_Inquiry(uint8_t lun, uint8_t* command)
245{
246 uint16_t i;
247 uint8_t* pInquiryData;
248
249 if (command[1] & 0x01)
250 {
251 pInquiryData = (uint8_t*)s_page00InquiryData;
252 g_BOTInfo.dataLen = s_page00InquiryData[3] + 4;
253 }
254 else
255 {
256 pInquiryData = &g_storageCallBack.pInquiryData[lun * SCSI_INQUIRY_LENGTH];
257
258 g_BOTInfo.dataLen = USB_MIN((pInquiryData[4] + 5), command[4]);
259 }
260
261 for (i = 0; i < g_BOTInfo.dataLen; i++)
262 {
263 g_BOTInfo.data[i] = pInquiryData[i];
264 }
265
266 return SCSI_OK;
267}

278static uint8_t SCSI_RequestSense(uint8_t lun, uint8_t* command)
279{
280 uint8_t i = 0;
281
282 while (i < SCSI_REQUEST_SENSE_DATA_LEN)
283 {
284 g_BOTInfo.data[i++] = 0;
285 }
286
287 g_BOTInfo.data[0] = 0x70;
288 g_BOTInfo.data[7] = SCSI_REQUEST_SENSE_DATA_LEN - 6;
289
290 if (g_senseTxCnt != g_sensePutCnt)
291 {
292 g_BOTInfo.data[2] = g_scsiSenseCode[g_senseTxCnt].sensekey;
293 g_BOTInfo.data[12] = g_scsiSenseCode[g_senseTxCnt].ASC;
294 g_BOTInfo.data[13] = g_scsiSenseCode[g_senseTxCnt].ASCQ;
295
296 if ((++g_senseTxCnt) == SCSI_SENSE_LIST_NUMBER)
297 {
298 g_senseTxCnt = 0;
299 }
300 }
301
302 g_BOTInfo.dataLen = (SCSI_REQUEST_SENSE_DATA_LEN < command[4]) ? \
303 SCSI_REQUEST_SENSE_DATA_LEN : command[4];
304
305 return SCSI_OK;
306}

317static uint8_t SCSI_ReadFormatCapacity(uint8_t lun, uint8_t* command)
318{
319 uint16_t i = 0;
320 uint32_t blkSize;
321 uint32_t blkNbr;
322
323 while (i < 12)
324 {
325 g_BOTInfo.data[i++] = 0;
326 }
327
328 if (g_storageCallBack.ReadCapacity(lun, &blkNbr, &blkSize) != SCSI_OK)
329 {
330 SCSI_PutSenseCode(lun, SCSI_SKEY_NOT_READY,
331 SCSI_ASC_MEDIUM_NOT_PRESENT, 0);
332
333 return SCSI_FAIL;
334 }
335 else
336 {
337 blkNbr--;
338
339 g_BOTInfo.data[3] = 0x08;
340 g_BOTInfo.data[4] = (uint8_t)(blkNbr >> 24);
341 g_BOTInfo.data[5] = (uint8_t)(blkNbr >> 16);
342 g_BOTInfo.data[6] = (uint8_t)(blkNbr >> 8);
343 g_BOTInfo.data[7] = (uint8_t)(blkNbr);
344 g_BOTInfo.data[8] = 0x02;
345 g_BOTInfo.data[9] = (uint8_t)(blkSize >> 16);
346 g_BOTInfo.data[10] = (uint8_t)(blkSize >> 8);
347 g_BOTInfo.data[11] = (uint8_t)blkSize;
348
349 g_BOTInfo.dataLen = 12;
350
351 return SCSI_OK;
352 }

399static uint8_t SCSI_Read10(uint8_t lun, uint8_t* command)
400{
401 uint8_t ret = SCSI_OK;
402
403 if (g_BOTInfo.state == BOT_STATE_IDLE)
404 {
405 if ((g_BOTInfo.CBW.bmFlags & 0x80) != 0x80)
406 {
407 SCSI_PutSenseCode(g_BOTInfo.CBW.bLUN, SCSI_SKEY_ILLEGAL_REQUEST,
408 SCSI_ASC_INVALID_CDB, 0);
409
410 return SCSI_FAIL;
411 }
412
413 if (g_storageCallBack.CheckReady(lun) != SCSI_OK)
414 {
415 SCSI_PutSenseCode(lun, SCSI_SKEY_NOT_READY,
416 SCSI_ASC_MEDIUM_NOT_PRESENT, 0);
417
418 return SCSI_FAIL;
419 }
420
421 s_blkAddr = ((uint32_t)command[2] << 24) | \
422 ((uint32_t)command[3] << 16) | \
423 ((uint32_t)command[4] << 8) | \
424 (uint32_t)command[5];
425
426 s_blkLen = ((uint16_t)command[7] << 8 | (uint8_t )command[8]);
427
428 if (SCSI_CheckAddress(lun, s_blkAddr, s_blkLen) != SCSI_OK)
429 {
430 return SCSI_FAIL;
431 }
432
433 g_BOTInfo.state = BOT_STATE_DATA_IN;
434 s_blkAddr *= s_blkSize;
435 s_blkLen *= s_blkSize;
436
437 if (g_BOTInfo.CBW.dDataXferLen != s_blkLen)
438 {
439 SCSI_PutSenseCode(g_BOTInfo.CBW.bLUN, SCSI_SKEY_ILLEGAL_REQUEST,
440 SCSI_ASC_INVALID_CDB, 0);
441
442 return SCSI_FAIL;
443 }
444
445 }
446
447 g_BOTInfo.dataLen = MSC_MEDIA_PACKET;
448
449 ret = SCSI_Read(lun);
450
451 return ret;
452}

504 }

507 s_blkLen *= s_blkSize;
508

512 SCSI_ASC_INVALID_CDB, 0);
513
514 return SCSI_FAIL;
515 }
516

521
522 }
523 else
524 {
525 ret = SCSI_Write(lun);
526 }

629static uint8_t SCSI_Read(uint8_t lun)
630{
631 uint32_t len = USB_MIN(MSC_MEDIA_PACKET, s_blkLen);
632
633 if (g_storageCallBack.ReadData(len, g_BOTInfo.data, (s_blkAddr / s_blkSize),
634 (len / s_blkSize)) != SCSI_OK)
635 {
636 SCSI_PutSenseCode(lun, SCSI_SKEY_HARDWARE_ERROR,
637 SCSI_ASC_UNRECOVERED_READ_ERROR, 0);
638
639 return SCSI_FAIL;
640 }
641
642 USBD_TxData(MSC_IN_EP & 0x7F, g_BOTInfo.data, len);
643
644 s_blkAddr += len;
645 s_blkLen -= len;
646
647 g_BOTInfo.CSW.dDataResidue -= len;
648
649 if (s_blkLen == 0)
650 {
651 g_BOTInfo.state = BOT_STATE_LAST_DATA_IN;
652 }
653
654 return SCSI_OK;
655}
