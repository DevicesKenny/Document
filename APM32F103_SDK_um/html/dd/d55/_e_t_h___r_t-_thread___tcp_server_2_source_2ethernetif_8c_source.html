122static void
123low_level_init(struct netif *netif)
124{
125 int i;
126
127 /* set MAC hardware address length */
128 netif->hwaddr_len = ETHARP_HWADDR_LEN;
129
130 /* set MAC hardware address */
131 netif->hwaddr[0] = SetMACaddr[0];
132 netif->hwaddr[1] = SetMACaddr[1];
133 netif->hwaddr[2] = SetMACaddr[2];
134 netif->hwaddr[3] = SetMACaddr[3];
135 netif->hwaddr[4] = SetMACaddr[4];
136 netif->hwaddr[5] = SetMACaddr[5];
137
138 /* Config MAC Address */
139 ETH_ConfigMACAddress(ETH_MAC_ADDRESS0, SetMACaddr);
140
141 /* maximum transfer unit */
142 netif->mtu = 1500;
143
144 /* device capabilities */
145 /* don't set NETIF_FLAG_ETHARP if this device is not an ethernet one */
146 netif->flags = NETIF_FLAG_BROADCAST | NETIF_FLAG_ETHARP | NETIF_FLAG_LINK_UP;
147
148 /* Initialize Tx Descriptors list: Chain Mode */
149 ETH_ConfigDMATxDescChain(DMATxDscrTab, &Tx_Buff[0][0], ETH_TXBUFNB);
150 /* Initialize Rx Descriptors list: Chain Mode */
151 ETH_ConfigDMARxDescChain(DMARxDscrTab, &Rx_Buff[0][0], ETH_RXBUFNB);
152
153 /* Enable Ethernet Rx interrrupt */
154 for(i=0; i<ETH_RXBUFNB; i++)
155 {
156 ETH_EnableDMARxDescReceiveInterrupt(&DMARxDscrTab[i]);
157 }
158
159#ifdef HARDWARE_CHECKSUM
160 /* Enable the checksum insertion for the Tx frames */
161 for(i=0; i<ETH_TXBUFNB; i++)
162 {
163 ETH_ConfigDMATxDescChecksumInsertion(&DMATxDscrTab[i], ETH_DMATXDESC_CHECKSUMTCPUDPICMPFULL);
164 }
165#endif
166
167 eth_rx_sem = rt_sem_create("eth_rx_sem", 0, RT_IPC_FLAG_PRIO);
168
169 /* create the task that handles the ETH_MAC */
170 sys_thread_new("eth_rx",
171 ethernetif_input,
172 netif,
173 1024,
174 TCPIP_THREAD_PRIO);
175
176 /* Enable MAC and DMA transmission and reception */
177 ETH_Start();
178}

195static err_t
196low_level_output(struct netif *netif, struct pbuf *p)
197{
198 struct pbuf *q;
199 int l = 0;
200 u8 *buffer = (u8 *)(DMATxDescToSet->Buffer1Addr);
201
202 for(q = p; q != NULL; q = q->next)
203 {
204 memcpy((u8_t*)&buffer[l], q->payload, q->len);
205 l = l + q->len;
206 }
207
208 ETH_TxPacket(l);
209
210 return ERR_OK;
211}

221static struct pbuf *
222low_level_input(struct netif *netif)
223{
224 struct pbuf *p, *q;
225 u16_t len;
226 int l =0;
227 ETH_Frame_T frame;
228 u8 *buffer;
229 __IO ETH_DMADescConfig_T *DMARxNextDesc;
230 int i = 0;
231
232 p = NULL;
233 frame = ETH_ReadReceivedFrameInterrupt();
234 if(frame.length == 0)
235 {
236 return NULL;
237 }
238
239 /* Obtain the size of the packet and put it into the "len"
240 variable. */
241 len = frame.length;
242 buffer = (u8 *)frame.buffer;
243
244 /* check that frame has no error */
245 if ((frame.descriptor->Status & ETH_DMARXDESC_ERRS) == (uint32_t)RESET)
246 {
247 /* We allocate a pbuf chain of pbufs from the pool. */
248 p = pbuf_alloc(PBUF_RAW, len, PBUF_POOL);
249
250 if (p != NULL)
251 {
252 for (q = p; q != NULL; q = q->next)
253 {
254 memcpy((u8_t*)q->payload, (u8_t*)&buffer[l], q->len);
255 l = l + q->len;
256 }
257 }
258 }
259
260 /* Release descriptors to DMA */
261 /* Check if frame with multiple DMA buffer segments */
262 if (DMARxFraminfos->segCount > 1)
263 {
264 DMARxNextDesc = DMARxFraminfos->FS_RxDesc;
265 }
266 else
267 {
268 DMARxNextDesc = frame.descriptor;
269 }
270
271 /* Set Own bit in Rx descriptors: gives the buffers back to DMA */
272 for (i=0; i<DMARxFraminfos->segCount; i++)
273 {
274 DMARxNextDesc->Status = ETH_DMARXDESC_OWN;
275 DMARxNextDesc = (ETH_DMADescConfig_T *)(DMARxNextDesc->Buffer2NextDescAddr);
276 }
277
278 /* Clear Segment_Count */
279 DMARxFraminfos->segCount =0;
280
282 if ((ETH->DMASTS & BIT7) != (u32)RESET)
283 {
285 ETH->DMASTS = BIT7;
287 ETH->DMARXPD = 0;
288 }
289
290 return p;
291}

302void ethernetif_input(void *pParams)
303{
304 struct pbuf *p = NULL;
305 struct netif *netif = (struct netif*) pParams;
306
307 LWIP_DEBUGF(NETIF_DEBUG, ("ethernetif_input: IP input error\n"));
308
309 while(1)
310 {
311 if(rt_sem_take(eth_rx_sem, RT_WAITING_FOREVER) == RT_EOK)
312 {
313 while(1)
314 {
315 /* move received packet into a new pbuf */
316 p = low_level_input(netif);
317 if(p == NULL)
318 {
319 break;
320 }
321
322 /* full packet send to tcpip_thread to process */
323 if (netif->input(p, netif) != ERR_OK)
324 {
325 LWIP_DEBUGF(NETIF_DEBUG, ("ethernetif_input: IP input error\n"));
326 pbuf_free(p);
327 p = NULL;
328 }
329 }
330 }
331 }

346err_t
347ethernetif_init(struct netif *netif)
348{
349 struct ethernetif *ethernetif;
350
351 LWIP_ASSERT("netif != NULL", (netif != NULL));
352
353 ethernetif = mem_malloc(sizeof(struct ethernetif));
354 if (ethernetif == NULL)
355 {
356 LWIP_DEBUGF(NETIF_DEBUG, ("ethernetif_init: out of memory\n"));
357 return ERR_MEM;
358 }
359
360#if LWIP_NETIF_HOSTNAME
361 /* Initialize interface hostname */
362 netif->hostname = "lwip";
363#endif /* LWIP_NETIF_HOSTNAME */
364
365 /*
366 * Initialize the snmp variables and counters inside the struct netif.
367 * The last argument should be replaced with your link speed, in units
368 * of bits per second.
369 */
370 NETIF_INIT_SNMP(netif, snmp_ifType_ethernet_csmacd, 100000000);
371
372 netif->state = ethernetif;
373 netif->name[0] = IFNAME0;
374 netif->name[1] = IFNAME1;
375 /* We directly use etharp_output() here to save a function call.
376 * You can instead declare your own function an call etharp_output()
377 * from it if you have to do some checks before sending (e.g. if link
378 * is available...) */
379 netif->output = etharp_output;
380 netif->linkoutput = low_level_output;
381
382 ethernetif->ethaddr = (struct eth_addr *)&(netif->hwaddr[0]);
383
384 /* initialize the hardware */
385 low_level_init(netif);
386
387 return ERR_OK;
388}

399ETH_Frame_T ETH_RxPacket(void)
400{
401 u32 ethframelength = 0;
402 ETH_Frame_T ethframe = {0,0};
403
404 /* Check if the descriptor is owned by the ETHERNET DMA (when set) or CPU (when reset) */
405 if((DMARxDescToGet->Status & ETH_DMARXDESC_OWN) != (u32)RESET)
406 {
407 ethframe.length = ETH_ERROR;
408
409 if ((ETH->DMASTS & BIT7) != (u32)RESET)
410 {
411 /* Clear RBUS ETHERNET DMA flag */
412 ETH->DMASTS = BIT7;
413 /* Resume DMA reception */
414 ETH->DMARXPD = 0;
415 }
416
417 /* Return error: OWN bit set */
418 return ethframe;
419 }
420
421 if(((DMARxDescToGet->Status & ETH_DMARXDESC_ERRS) == (u32)RESET) &&
422 ((DMARxDescToGet->Status & ETH_DMARXDESC_LDES) != (u32)RESET) &&
423 ((DMARxDescToGet->Status & ETH_DMARXDESC_FDES) != (u32)RESET))
424 {
425 /* Get the Frame Length of the received packet: substruct 4 bytes of the CRC */
426 ethframelength = ((DMARxDescToGet->Status & ETH_DMARXDESC_FL) >> 16) - 4;
427
428 /* Get the addrees of the actual buffer */
429 ethframe.buffer = DMARxDescToGet->Buffer1Addr;
430 }
431 else
432 {
433 /* Return ERROR */
434 ethframelength = ETH_ERROR;
435 }
436
437 ethframe.length = ethframelength;
438 ethframe.descriptor = DMARxDescToGet;
439
440 /* Selects the next DMA Rx descriptor list for next buffer to read */
441 DMARxDescToGet = (ETH_DMADescConfig_T*) (DMARxDescToGet->Buffer2NextDescAddr);
442
443 /* Return Frame */
444 return ethframe;
445}

457int ETH_TxPacket(u16 Length)
458{
460 if((DMATxDescToSet->Status & ETH_DMATXDESC_OWN) != (u32)RESET)
461 {
463 return ETH_ERROR;
464 }
465
467 DMATxDescToSet->ControlBufferSize = (Length & ETH_DMATXDESC_TXBS1);
468
470 DMATxDescToSet->Status |= ETH_DMATXDESC_LS | ETH_DMATXDESC_FS;
471
473 DMATxDescToSet->Status |= ETH_DMATXDESC_OWN;
474
476 if ((ETH->DMASTS & BIT2) != (u32)RESET)
477 {
479 ETH->DMASTS = BIT2;
481 ETH->DMATXPD = 0;
482 }
483
487 DMATxDescToSet = (ETH_DMADescConfig_T*) (DMATxDescToSet->Buffer2NextDescAddr);
488
490 return ETH_SUCCESS;
491}
