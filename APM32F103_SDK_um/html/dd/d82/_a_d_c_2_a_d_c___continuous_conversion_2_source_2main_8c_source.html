90void ADC_Init(void)
91{
92 GPIO_Config_T gpioConfig;
93 ADC_Config_T adcConfig;
94
95 /* Enable GPIOA clock */
96 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA);
97
98 /* ADC channel 0 configuration */
99 GPIO_ConfigStructInit(&gpioConfig);
100 gpioConfig.mode = GPIO_MODE_ANALOG;
101 gpioConfig.pin = GPIO_PIN_0;
102 GPIO_Config(GPIOA, &gpioConfig);
103
104 /* ADCCLK = PCLK2/4 */
105 RCM_ConfigADCCLK(RCM_PCLK2_DIV_4);
106
107 /* Enable ADC clock */
108 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_ADC1);
109
110 /* ADC configuration */
111 ADC_Reset(ADC1);
112 ADC_ConfigStructInit(&adcConfig);
113 adcConfig.mode = ADC_MODE_INDEPENDENT;
114 adcConfig.scanConvMode = DISABLE;
115 adcConfig.continuosConvMode = ENABLE;
116 adcConfig.externalTrigConv = ADC_EXT_TRIG_CONV_None;
117 adcConfig.dataAlign = ADC_DATA_ALIGN_RIGHT;
118 /* channel number */
119 adcConfig.nbrOfChannel = 1;
120 ADC_Config(ADC1, &adcConfig);
121
122 /* ADC channel Convert configuration */
123 ADC_ConfigRegularChannel(ADC1, ADC_CHANNEL_0, 1, ADC_SAMPLETIME_13CYCLES5);
124
125 /* Enable complete conversion interupt */
126 ADC_EnableInterrupt(ADC1, ADC_INT_EOC);
127
128 /* NVIC configuration */
129 NVIC_EnableIRQRequest(ADC1_2_IRQn, 1, 1);
130
131 /* Enable ADC */
132 ADC_Enable(ADC1);
133
134 /* Enable ADC1 reset calibration register */
135 ADC_ResetCalibration(ADC1);
136 /* Check the end of ADC1 reset calibration register */
137 while(ADC_ReadResetCalibrationStatus(ADC1));
138
139 /* Start ADC1 calibration */
140 ADC_StartCalibration(ADC1);
141 /* Check the end of ADC1 calibration */
142 while(ADC_ReadCalibrationStartFlag(ADC1));
143
144 /* Start ADC1 Software Conversion */
145 ADC_EnableSoftwareStartConv(ADC1);
146}

157void ADC1_Isr(void)
158{
159 uint16_t adcData = 0;
160 uint16_t voltage = 0;
161
162 if (ADC_ReadIntFlag(ADC1, ADC_INT_EOC) == SET)
163 {
164 ADC_ClearIntFlag(ADC1, ADC_INT_EOC);
165 /* Read ADC Conversion value */
166 adcData = ADC_ReadConversionValue(ADC1);
167 /* voltage(mV) = adcData * (3300mV / 4095) */
168 voltage = (adcData * 3300) / 4095;
169 printf("\r\n voltage : %d mV\r\n", voltage);
170 }
171}
