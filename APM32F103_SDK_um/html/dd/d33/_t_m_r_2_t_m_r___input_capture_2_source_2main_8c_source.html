70int main(void)
71{
72 GPIO_Config_T gpioConfig;
73 USART_Config_T usartConfig;
74 TMR_ICConfig_T ICConfig;
75 TMR_BaseConfig_T baseConfig;
76
77 /* Init LED */
78 APM_MINI_LEDInit(LED2);
79
80 /* USART1 Tx:PA9 Rx:PA10 */
81 usartConfig.baudRate = 115200;
82 usartConfig.mode = USART_MODE_TX_RX;
83 usartConfig.hardwareFlow = USART_HARDWARE_FLOW_NONE;
84 usartConfig.parity = USART_PARITY_NONE;
85 usartConfig.stopBits = USART_STOP_BIT_1;
86 usartConfig.wordLength = USART_WORD_LEN_8B;
87 APM_MINI_COMInit(COM1,&usartConfig);
88
89 /* Enable Periph Clock */
90 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_TMR1);
91 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_TMR5);
92 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA);
93
94 /* TMR5 channel2 for PA1 */
95 gpioConfig.pin = GPIO_PIN_1;
96 gpioConfig.mode = GPIO_MODE_AF_PP;
97 gpioConfig.speed = GPIO_SPEED_50MHz;
98 GPIO_Config(GPIOA, &gpioConfig);
99
100 /* TMR5 PWM Input mode */
101 ICConfig.channel = TMR_CHANNEL_2;
102 ICConfig.polarity = TMR_IC_POLARITY_RISING;
103 ICConfig.selection = TMR_IC_SELECTION_DIRECT_TI;
104 ICConfig.prescaler = TMR_IC_PSC_1;
105 ICConfig.filter = 0x00;
106 TMR_ConfigPWM(TMR5, &ICConfig);
107
108 /* Enable TMR5 Interrupt */
109 TMR_Enable(TMR5);
110 NVIC_EnableIRQRequest(TMR5_IRQn, 0, 0);
111 TMR_EnableInterrupt(TMR5, TMR_INT_CC2);
112
113 /* Config TMR1 */
114 baseConfig.clockDivision = TMR_CLOCK_DIV_1;
115 baseConfig.countMode = TMR_COUNTER_MODE_UP;
116 baseConfig.division = 167;
117 baseConfig.period = 999;
118 baseConfig.repetitionCounter = 0;
119 TMR_ConfigTimeBase(TMR1, &baseConfig);
120
121 /* Enable TMR1 Interrupt */
122 TMR_Enable(TMR1);
123 TMR_EnableInterrupt(TMR1, TMR_INT_UPDATE);
124 NVIC_EnableIRQRequest(TMR1_UP_IRQn, 0, 0);
125
126 while(1)
127 {
128 printf("Frequency: %d Hz\r\n",Frequency);
129 APM_MINI_LEDToggle(LED2);
130 Delay();
131 }
132}

164int fputc(int ch, FILE* f)
165{
166 /* send a byte of data to the serial port */
167 USART_TxData(DEBUG_USART, (uint8_t)ch);
168
169 /* wait for the data to be send */
170 while (USART_ReadStatusFlag(DEBUG_USART, USART_FLAG_TXBE) == RESET);
171
172 return (ch);
173}
