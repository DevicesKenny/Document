167void LINInit(void)
168{
169 GPIO_Config_T GPIO_ConfigStruct;

172 /* System Clocks Configuration */
173 RCM_EnableAPB2PeriphClock(RCM_APB2_PERIPH_GPIOA | RCM_APB2_PERIPH_GPIOB | RCM_APB2_PERIPH_AFIO);
174 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_USART2);
175 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_USART3);
176
177 /* Configure the GPIO ports */
178 /* USART2 GPIO Configuration
179 PA2 ------> USART2_TX
180 PA3 ------> USART2_RX
181 */
182 GPIO_ConfigStruct.pin = GPIO_PIN_2;
183 GPIO_ConfigStruct.mode = GPIO_MODE_AF_PP;
184 GPIO_ConfigStruct.speed = GPIO_SPEED_50MHz;
185 GPIO_Config(GPIOA, &GPIO_ConfigStruct);
186
187 GPIO_ConfigStruct.pin = GPIO_PIN_3;
188 GPIO_ConfigStruct.mode = GPIO_MODE_IN_FLOATING;
189 GPIO_Config(GPIOA, &GPIO_ConfigStruct);
190
191 /* USART3 GPIO Configuration
192 PB10 ------> USART3_TX
193 PB11 ------> USART3_RX
194 */
195 GPIO_ConfigStruct.pin = GPIO_PIN_10;
196 GPIO_ConfigStruct.mode = GPIO_MODE_AF_PP;
197 GPIO_ConfigStruct.speed = GPIO_SPEED_50MHz;
198 GPIO_Config(GPIOB, &GPIO_ConfigStruct);
199
200 GPIO_ConfigStruct.pin = GPIO_PIN_11;
201 GPIO_ConfigStruct.mode = GPIO_MODE_IN_FLOATING;
202 GPIO_Config(GPIOB, &GPIO_ConfigStruct);
203
204 /* USARTx configuration */
205 /* USARTx configured as follow:
206 - BaudRate = 9600 baud
207 - Word Length = 8 Bits
208 - One Stop Bit
209 - No parity
210 - Hardware flow control disabled (RTS and CTS signals)
211 - Receive and transmit enabled
212 */
213 USART_ConfigStruct.baudRate = 9600;
214 USART_ConfigStruct.hardwareFlow = USART_HARDWARE_FLOW_NONE;
215 USART_ConfigStruct.mode = USART_MODE_TX_RX;
216 USART_ConfigStruct.parity = USART_PARITY_NONE;
217 USART_ConfigStruct.stopBits = USART_STOP_BIT_1;
218 USART_ConfigStruct.wordLength = USART_WORD_LEN_8B;
219

236

239 USART_EnableLIN(USART3);
240
241 /* Enable the USARTx */
242 USART_Enable(USART2);
243 USART_Enable(USART3);

421uint8_t LIN_CheckPID(uint8_t id)
422{
423 uint8_t returnpid ;
424 uint8_t P0 ;
425 uint8_t P1 ;
426
427 P0 = (((id)^(id>>1)^(id>>2)^(id>>4))&0x01)<<6 ;
428 P1 = ((~((id>>1)^(id>>3)^(id>>4)^(id>>5)))&0x01)<<7 ;
429
430 returnpid = id|P0|P1 ;
431
432 return returnpid ;
433}

445uint8_t LIN_Checksum(uint8_t id, uint8_t* data)
446{
447 uint8_t t ;
448 uint16_t sum ;
449
450 sum = data[0];
451
452 for (t=1; t<8; t++)
453 {
454 sum += data[t];
455
456 if (sum&0xff00)
457 {
458 sum&=0x00ff;
459 sum+=1;
460 }
461 }
462
463 /* In the case of diagnostic frames, the classic checksum is used. */
464 if (id != 0x3C)
465 {
466 sum+=LIN_CheckPID(id);
467
468 if (sum&0xff00)
469 {
470 sum&=0x00ff;
471 sum+=1;
472 }
473 }
474
475 sum = ~sum;
476 return (uint8_t)sum ;
477}

595void LIN_MasterISR(void)
596{
597 uint8_t ReceiveData;
598
599 if (USART_ReadIntFlag(USART2, USART_INT_RXBNE) == SET)
600 {
601 ReceiveData = USART_RxData(USART2);
602
603 MasterReceivedOverFlag = 0;
604
605 if (MasterReceivedProces == 0)
606 {
607 if (MasterReceiveDataProces<8)
608 {
609 MasterReceived.data[MasterReceiveDataProces] = ReceiveData ;
610 MasterReceiveDataProces += 1 ;
611
612 if (MasterReceiveDataProces == 8)
613 {
614 MasterReceiveDataProces = 0 ;
615 MasterReceivedProces = 1;
616 }
617 }
618 }
619 else
620 {
621 MasterReceived.CheckSum = ReceiveData;
622 MasterReceivedProces = 0;
623 MasterReceivedOverFlag = 1;
624 }
625 }

636void LIN_SlaveISR(void)
637{
638 uint8_t ReceiveData;
639 uint8_t ReceiveID;
640
641 if (USART_ReadIntFlag(USART3, USART_INT_RXBNE) == SET)
642 {
643 ReceiveData = USART_RxData(USART3);
644
645 SlaveReceivedOverFlag = 0;
646
647 if (SlaveReceivedProces == 0)
648 {
649 /* Sync frame processing. */
650 if (ReceiveData == 0x55)
651 {
652 SlaveReceivedProces = 1 ;
653 }
654 }
655 else if (SlaveReceivedProces == 1)
656 {
657 SlaveReceived.ID = ReceiveData ;
658 ReceiveID = SlaveReceived.ID&0x3f ;
659
660 /* The master reads the slave data */
661 if (ReceiveID == 0x33)
662 {
663 LIN_TxData(USART3, SlaveSend.data);
664 USART_TxData(USART3, LIN_Checksum(ReceiveID, SlaveSend.data));
665
666 /* wait for the data to be send */
667 while (USART_ReadStatusFlag(USART3, USART_FLAG_TXBE) == RESET);
668
669 SlaveReceivedProces = 0 ;
670 }
671
672 SlaveReceivedflag = 1 ;
673 SlaveReceivedProces = 2 ;
674 }
675 else if (SlaveReceivedProces == 2)
676 {
677 /* data frame processing */
678 if (SlaveReceivedDataProces<8)
679 {
680 SlaveReceived.data[SlaveReceivedDataProces] = ReceiveData ;
681 SlaveReceivedDataProces += 1 ;
682
683 if (SlaveReceivedDataProces == 8)
684 {
685 SlaveReceivedDataProces = 0 ;
686 SlaveReceivedProces = 3 ;
687 return ;
688 }
689 }
690 }
691 else if (SlaveReceivedProces == 3)
692 {
693 /* Check frame processing */
694 SlaveReceived.CheckSum = ReceiveData;
695 SlaveReceivedProces = 0 ;
696 SlaveReceivedOverFlag = 1;
697 }
698 }
