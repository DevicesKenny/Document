65int main(void)
66{
67 uint8_t state;
68
69 RCM_EnableAPB1PeriphClock(RCM_APB1_PERIPH_CAN1);
70 NVIC_Configuration();
71
72 /* Configures LED2 and LED3 */
73 APM_MINI_LEDInit(LED2);
74 APM_MINI_LEDInit(LED3);
75
76 /* Turns selected LED Off */
77 APM_MINI_LEDOff(LED2);
78 APM_MINI_LEDOff(LED3);
79
80 state = CAN_Polling();
81 if (state == FAILED)
82 {
83 /* Turn off LED2 */
84 APM_MINI_LEDOff(LED2);
85 }
86 else
87 {
88 /* Turn on LED2 */
89 APM_MINI_LEDOn(LED2);
90 }
91
92 /* CAN transmit at 500Kb/s and receive by interrupt in loopback mode */
93 state = CAN_Interrupt();
94
95 if (state == FAILED)
96 {
97 /* Turn off LED3 */
98 APM_MINI_LEDOff(LED3);
99 }
100 else
101 {
102 /* Turn on LED3 */
103 APM_MINI_LEDOn(LED3);
104 }
105 while (1)
106 {
107 }
108}

118uint8_t CAN_Polling(void)
119{
120 uint32_t i = 0;
121 CAN_Config_T CAN_ConfigStructure;
122 CAN_FilterConfig_T CAN_FilterStruct;
123 CAN_TxMessage_T TxMessage;
124 CAN_RxMessage_T RxMessage;
125 CAN_TX_MAILBIX_T TransmitMailbox = CAN_TX_MAILBIX_1;
126
127 /* CAN register init */
128 CAN_Reset(CANx);
129 CAN_ConfigStructInit(&CAN_ConfigStructure);
130
131 /* CAN init */
132 CAN_ConfigStructure.autoBusOffManage=DISABLE;
133 CAN_ConfigStructure.autoWakeUpMode=DISABLE;
134 CAN_ConfigStructure.nonAutoRetran=DISABLE;
135 CAN_ConfigStructure.rxFIFOLockMode=DISABLE;
136 CAN_ConfigStructure.txFIFOPriority=DISABLE;
137 CAN_ConfigStructure.mode=CAN_MODE_LOOPBACK;
138
139 /* Baudrate = 125kbps*/
140 CAN_ConfigStructure.syncJumpWidth=CAN_SJW_1;
141 CAN_ConfigStructure.timeSegment1=CAN_TIME_SEGMENT1_2;
142 CAN_ConfigStructure.timeSegment2=CAN_TIME_SEGMENT2_3;
143 CAN_ConfigStructure.prescaler=48;
144 CAN_Config(CANx, &CAN_ConfigStructure);
145
146 /* CAN filter init */
147 CAN_FilterStruct.filterNumber=0;
148 CAN_FilterStruct.filterMode=CAN_FILTER_MODE_IDMASK;
149 CAN_FilterStruct.filterScale=CAN_FILTER_SCALE_32BIT;
150 CAN_FilterStruct.filterIdHigh=0x0000;
151 CAN_FilterStruct.filterIdLow=0x0000;
152 CAN_FilterStruct.filterMaskIdHigh=0x0000;
153 CAN_FilterStruct.filterMaskIdLow=0x0000;
154 CAN_FilterStruct.filterFIFO=CAN_FILTER_FIFO_0;
155 CAN_FilterStruct.filterActivation=ENABLE;
156 CAN_ConfigFilter(CANx, &CAN_FilterStruct);
157
158 /* Transmit */
159 TxMessage.stdID=0x11;
160 TxMessage.remoteTxReq=CAN_RTXR_DATA;
161 TxMessage.typeID=CAN_TYPEID_STD;
162 TxMessage.dataLengthCode=2;
163 TxMessage.data[0]=0xCA;
164 TxMessage.data[1]=0xFE;
165
166 TransmitMailbox=(CAN_TX_MAILBIX_T)CAN_TxMessage(CANx, &TxMessage);
167 i = 0;
168 while((!CAN_TxMessageStatus(CANx, TransmitMailbox)) && (i != 0xFFFF))
169 {
170 i++;
171 }
172
173 i = 0;
174 while((CAN_PendingMessage(CANx, CAN_RX_FIFO_0) < 1) && (i != 0xFFFF))
175 {
176 i++;
177 }
178
179 /* Receive */
180 RxMessage.stdID=0x00;
181 RxMessage.typeID=CAN_TYPEID_STD;
182 RxMessage.dataLengthCode=0;
183 RxMessage.data[0]=0x00;
184 RxMessage.data[1]=0x00;
185 CAN_RxMessage(CANx, CAN_RX_FIFO_0, &RxMessage);
186
187 if (RxMessage.stdID!=0x11)
188 {
189 return FAILED;
190 }
191
192 if (RxMessage.typeID!=CAN_TYPEID_STD)
193 {
194 return FAILED;
195 }
196
197 if (RxMessage.dataLengthCode!=2)
198 {
199 return FAILED;
200 }
201
202 if ((RxMessage.data[0]<<8|RxMessage.data[1])!=0xCAFE)
203 {
204 return FAILED;
205 }
206
207 return PASSED;
208}

218uint8_t CAN_Interrupt(void)
219{
220 uint32_t i = 0;
221 CAN_Config_T CAN_ConfigStructure;
222 CAN_FilterConfig_T CAN_FilterStruct;
223 CAN_TxMessage_T TxMessage;
224
225 /* CAN register init */
226 CAN_Reset(CANx);
227 CAN_ConfigStructInit(&CAN_ConfigStructure);
228
229 /* CAN init */
230 CAN_ConfigStructure.autoBusOffManage=DISABLE;
231 CAN_ConfigStructure.autoWakeUpMode=DISABLE;
232 CAN_ConfigStructure.nonAutoRetran=DISABLE;
233 CAN_ConfigStructure.rxFIFOLockMode=DISABLE;
234 CAN_ConfigStructure.txFIFOPriority=DISABLE;
235 CAN_ConfigStructure.mode=CAN_MODE_LOOPBACK;
236 /* Baudrate = 500 Kbps */
237 CAN_ConfigStructure.syncJumpWidth=CAN_SJW_1;
238 CAN_ConfigStructure.timeSegment1=CAN_TIME_SEGMENT1_2;
239 CAN_ConfigStructure.timeSegment2=CAN_TIME_SEGMENT2_3;
240 CAN_ConfigStructure.prescaler=12;
241 CAN_Config(CANx, &CAN_ConfigStructure);
242
243 /* CAN filter init */
244 CAN_FilterStruct.filterNumber=1;
245 CAN_FilterStruct.filterMode=CAN_FILTER_MODE_IDMASK;
246 CAN_FilterStruct.filterScale=CAN_FILTER_SCALE_32BIT;
247 CAN_FilterStruct.filterIdHigh=0x0000;
248 CAN_FilterStruct.filterIdLow=0x0000;
249 CAN_FilterStruct.filterMaskIdHigh=0x0000;
250 CAN_FilterStruct.filterMaskIdLow=0x0000;
251 CAN_FilterStruct.filterFIFO=CAN_FILTER_FIFO_0;
252 CAN_FilterStruct.filterActivation=ENABLE;
253 CAN_ConfigFilter(CANx, &CAN_FilterStruct);
254
255 /* CAN FIFO0 message pending interrupt enable */
256 CAN_EnableInterrupt(CANx, CAN_INT_F0MP);
257
258 /* Transmit 1 message */
259 TxMessage.stdID=0x0;
260 TxMessage.extID = 0x1234;
261 TxMessage.remoteTxReq=CAN_RTXR_DATA;
262 TxMessage.typeID=CAN_TYPEID_EXT;
263 TxMessage.dataLengthCode=2;
264 TxMessage.data[0]=0xDE;
265 TxMessage.data[1]=0xCA;
266
267 CAN_TxMessage(CANx, &TxMessage);
268
269 /* Initialize the value that will be returned */
270 intFlag = 0xFF;
271
272 /* Receive message with interrupt handling */
273 i=0;
274 while((intFlag == 0xFF) && (i < 0xFFF))
275 {
276 i++;
277 }
278
279 if (i == 0xFFF)
280 {
281 intFlag=0;
282 }
283
284 /* Disable interrupt handling */
285 CAN_DisableInterrupt(CANx, CAN_INT_F0MP);
286
287 return intFlag;
288}
