187uint32_t ETH_Config(ETH_Config_T* ethConfig, uint16_t addr)
188{
189 uint32_t regValue = 0;
190 uint32_t hclk = 60000000;
191 __IO uint32_t timeout = 0, err = ETH_SUCCESS;
192
193 hclk = RCM_ReadHCLKFreq();
194
195 if ((hclk >= 20000000) && (hclk <= 35000000))
196 {
197 ETH->ADDR_B.CR = ETH_MACMIIAR_CR_DIV16;
198 }
199 else if ((hclk >= 35000000) && (hclk < 60000000))
200 {
201 ETH->ADDR_B.CR = ETH_MACMIIAR_CR_DIV26;
202 }
203 else if ((hclk >= 60000000) && (hclk < 100000000))
204 {
205 ETH->ADDR_B.CR = ETH_MACMIIAR_CR_DIV42;
206 }
207 else if ((hclk >= 100000000) && (hclk < 150000000))
208 {
209 ETH->ADDR_B.CR = ETH_MACMIIAR_CR_DIV62;
210 }
211 else
212 {
213 ETH->ADDR_B.CR = ETH_MACMIIAR_CR_DIV102;
214 }
215
216 /* PHY initialization and configuration */
217 if (!(ETH_WritePHYRegister(addr, PHY_BCR, PHY_RESET)))
218 {
219 /* Return ERROR in case of write timeout */
220 err = ETH_ERROR;
221 goto error;
222 }
223
224 ETH_Delay(PHY_RESET_DELAY);
225
226 if (ethConfig->autoNegotiation == ETH_AUTONEGOTIATION_ENABLE)
227 {
228 /* Wait for linked status */
229 do
230 {
231 timeout++ ;
232 }
233 while (!(ETH_ReadPHYRegister(addr, PHY_BSR) & PHY_LINKED_STATUS) && (timeout < PHY_READ_TIMEOUT));
234
235 /* Return ERROR in case of timeout */
236 if (timeout == PHY_READ_TIMEOUT)
237 {
238 err = ETH_ERROR;
239 goto error;
240 }
241
242 timeout = 0;
243 /* Enable Auto-Negotiation */
244 if (!(ETH_WritePHYRegister(addr, PHY_BCR, PHY_AUTONEGOTIATION)))
245 {
246 /* Return ERROR in case of write timeout */
247 err = ETH_ERROR;
248 }
249
250 /* Wait until the auto-negotiation will be completed */
251 do
252 {
253 timeout++;
254 }
255 while (!(ETH_ReadPHYRegister(addr, PHY_BSR) & PHY_AUTONEGO_COMPLETE) && (timeout < (uint32_t)PHY_READ_TIMEOUT));
256
257 /* Return ERROR in case of timeout */
258 if (timeout == PHY_READ_TIMEOUT)
259 {
260 err = ETH_ERROR;
261 goto error;
262 }
263
264 timeout = 0;
265 /* Read the result of the auto-negotiation */
266 regValue = ETH_ReadPHYRegister(addr, PHY_SR);
267
268 if ((regValue & PHY_DUPLEX_STATUS) != (uint32_t)RESET)
269 {
270 ethConfig->mode = ETH_MODE_FULLDUPLEX;
271 }
272 else
273 {
274 ethConfig->mode = ETH_MODE_HALFDUPLEX;
275 }
276 if (regValue & PHY_SPEED_STATUS)
277 {
278 ethConfig->speed = ETH_SPEED_10M;
279 }
280 else
281 {
282 ethConfig->speed = ETH_SPEED_100M;
283 }
284 }
285 else
286 {
287 if (!ETH_WritePHYRegister(addr, PHY_BCR, ((uint16_t)(ethConfig->speed << 8) |
288 (uint16_t)(ethConfig->mode << 13))))
289 {
290 err = ETH_ERROR;
291 }
292
293 ETH_Delay(PHY_CONFIG_DELAY);
294 }
295error:
296 if (err == ETH_ERROR)
297 {
298 ethConfig->speed = ETH_SPEED_100M;
299 ethConfig->mode = ETH_MODE_FULLDUPLEX;
300 }
301
302 /* ETHERNET MAC_CFG Configuration */
303 ETH->CFG_B.WDTDIS = ethConfig->watchDog;
304 ETH->CFG_B.JDIS = ethConfig->jabber;
305 ETH->CFG_B.IFG = ethConfig->interFrameGap;
306 ETH->CFG_B.DISCRS = ethConfig->carrierSense;
307 ETH->CFG_B.SSEL = ethConfig->speed;
308 ETH->CFG_B.DISRXO = ethConfig->receiveOwn;
309 ETH->CFG_B.LBM = ethConfig->loopbackMode;
310 ETH->CFG_B.DM = ethConfig->mode;
311 ETH->CFG_B.IPC = ethConfig->checksumOffload;
312 ETH->CFG_B.DISR = ethConfig->retryTransmission;
313 ETH->CFG_B.ACS = ethConfig->automaticPadCRCStrip;
314 ETH->CFG_B.BL = ethConfig->backOffLimit;
315 ETH->CFG_B.DC = ethConfig->deferralCheck;
316 ETH_Delay(ETH_REG_WRITE_DELAY);
317
318 /* ETHERNET MAC_FRAF Configuration */
319 ETH->FRAF_B.RXA = ethConfig->receiveAll;
320 ETH->FRAF |= ethConfig->sourceAddrFilter;
321 ETH->FRAF_B.PCTRLF = ethConfig->passControlFrames;
322 ETH->FRAF_B.DISBF = ethConfig->broadcastFramesReception;
323 ETH->FRAF_B.DAIF = ethConfig->destinationAddrFilter;
324 ETH->FRAF_B.PR = ethConfig->promiscuousMode;
325 ETH->FRAF |= ethConfig->multicastFramesFilter;
326 ETH->FRAF |= ethConfig->unicastFramesFilter;
327 ETH_Delay(ETH_REG_WRITE_DELAY);
328
329 /* ETHERNET MAC_HTH/HTL Configuration */
330 ETH->HTH = ethConfig->hashTableHigh;
331 ETH->HTL = ethConfig->hashTableLow;
332
333 /* ETHERNET MAC_FCTRL Configuration */
334 ETH->FCTRL_B.PT = ethConfig->pauseTime;
335 ETH->FCTRL_B.ZQPDIS = ethConfig->zeroQuantaPause;
336 ETH->FCTRL_B.PTSEL = ethConfig->pauseLowThreshold;
337 ETH->FCTRL_B.UNPFDETE = ethConfig->unicastPauseFrameDetect;
338 ETH->FCTRL_B.RXFCTRLEN = ethConfig->receiveFlowControl;
339 ETH->FCTRL_B.TXFCTRLEN = ethConfig->transmitFlowControl;
340 ETH_Delay(ETH_REG_WRITE_DELAY);
341
342 /* ETHERNET MAC_VLANT Configuration */
343 ETH->VLANT_B.VLANTCOMP = ethConfig->VLANTagComparison;
344 ETH->VLANT_B.VLANTID = ethConfig->VLANTagIdentifier;
345 ETH_Delay(ETH_REG_WRITE_DELAY);
346
347 /* ETHERNET DMA_OPMOD Configuration */
348 ETH->DMAOPMOD_B.DISDT = ethConfig->dropTCPIPChecksumErrorFrame;
349 ETH->DMAOPMOD_B.RXSF = ethConfig->receiveStoreForward;
350 ETH->DMAOPMOD_B.DISFRXF = ethConfig->flushReceivedFrame;
351 ETH->DMAOPMOD_B.TXSF = ethConfig->transmitStoreForward;
352 ETH->DMAOPMOD_B.TXTHCTRL = ethConfig->transmitThresholdControl;
353 ETH->DMAOPMOD_B.FERRF = ethConfig->forwardErrorFrames;
354 ETH->DMAOPMOD_B.FUF = ethConfig->forwardUndersizedGoodFrames;
355 ETH->DMAOPMOD_B.RXTHCTRL = ethConfig->receiveThresholdControl;
356 ETH->DMAOPMOD_B.OSECF = ethConfig->secondFrameOperate;
357 ETH_Delay(ETH_REG_WRITE_DELAY);
358
359 /* ETHERNET DMA_BMOD Configuration */
360 ETH->DMABMOD = RESET;
361 ETH->DMABMOD_B.AAL = ethConfig->addressAlignedBeats;
362 ETH->DMABMOD_B.FB = ethConfig->fixedBurst;
363 ETH->DMABMOD |= ethConfig->rxDMABurstLength;
364 ETH->DMABMOD |= ethConfig->txDMABurstLength;
365 ETH->DMABMOD_B.DSL = ethConfig->descriptorSkipLength;
366 ETH->DMABMOD |= ethConfig->DMAArbitration;
367 ETH->DMABMOD_B.USP = SET;
368 ETH_Delay(ETH_REG_WRITE_DELAY);
369
370 if (err == ETH_SUCCESS)
371 {
372 return ETH_SUCCESS;
373 }
374 else
375 {
376 return ETH_ERROR;
377 }

982{

984 ((DMARxDescToGet->Status & ETH_DMARXDESC_LDES) != (uint32_t)RESET))

989 DMARxFraminfos->FS_RxDesc = DMARxDescToGet;

991 DMARxFraminfos->LS_RxDesc = DMARxDescToGet;
992 return 1;
993 }
994 else if (((DMARxDescToGet->Status & ETH_DMARXDESC_OWN) == (uint32_t)RESET) &&
995 ((DMARxDescToGet->Status & ETH_DMARXDESC_FDES) != (uint32_t)RESET) &&
996 ((DMARxDescToGet->Status & ETH_DMARXDESC_LDES) == (uint32_t)RESET))

998 DMARxFraminfos->FS_RxDesc = DMARxDescToGet;
999 DMARxFraminfos->LS_RxDesc = NULL;
